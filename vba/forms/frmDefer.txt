Private mRowId As String
Private mTaskCode As String
Private mIsHoursFault As Boolean

Private Sub UserForm_Initialize()
    Dim taskDesc As String
    Dim intervalCode As String
    Dim dueHint As String
    Dim currentHHDisplay As String

    mRowId = GetPendingDeferRowId()
    If Len(mRowId) = 0 Then
        MsgBox "No deferred fault was passed to the Defer form.", vbExclamation, APP_NAME
        Unload Me
        Exit Sub
    End If

    If Not GetDeferSelectionMeta(mRowId, mTaskCode, taskDesc, intervalCode, _
                                 mIsHoursFault, dueHint, currentHHDisplay) Then
        MsgBox "Could not read selected deferred fault details.", vbExclamation, APP_NAME
        Unload Me
        Exit Sub
    End If

    lblTask.Caption = mTaskCode & " — " & taskDesc
    lblInterval.Caption = "Interval Type Code: " & intervalCode

    If mIsHoursFault Then
        lblDue.Caption = "Airframe hours due (HH):"
        lblDueHint.Caption = dueHint
        lblCurrent.Caption = "Current airframe HH: " & currentHHDisplay
    Else
        lblDue.Caption = "Date due:"
        lblDueHint.Caption = dueHint
        lblCurrent.Caption = ""
    End If

    txtDue.Value = ""
    txtSNOW.Value = ""
    txtAuthoriser.Value = ""
End Sub

Private Sub cmdApply_Click()
    If ApplyDeferralToSelection(mRowId, txtDue.Value, txtSNOW.Value, _
                                txtAuthoriser.Value, True) Then
        Unload Me
    End If
End Sub

Private Sub cmdCancel_Click()
    Unload Me
End Sub
