Option Explicit

' ========================================================================
' WORK PACKAGE USERFORM — Offline Aircraft SITS v2 (Phase 4/5)
' ========================================================================
' Create this UserForm manually in the VBA IDE and paste this code.
' See vba/forms/WORK_PACKAGE_FORM_LAYOUT.txt for control names and layout.
' ========================================================================

Private Sub UserForm_Initialize()
    RefreshForm
End Sub

Private Sub UserForm_Activate()
    CenterUserFormOnExcel Me
    LayoutWorkPackageListHeaders
End Sub

Private Sub cboORN_Change()
    PopulateTaskList
    UpdateActionButtons
End Sub

Private Sub lstTasks_Click()
    UpdateActionButtons
End Sub

Private Sub cmdDelete_Click()
    Dim orn As String

    orn = SelectedORN()
    If Len(orn) = 0 Then Exit Sub

    ' Hide before long delete so the modeless progress form can show.
    Me.Hide
    If DeleteWorkPackage(orn, True) Then
        RefreshForm
    End If
    Me.Show
End Sub

Private Sub cmdRemoveTask_Click()
    Dim orn As String
    Dim rowId As String

    orn = SelectedORN()
    rowId = SelectedTaskRowId()
    If Len(orn) = 0 Or Len(rowId) = 0 Then Exit Sub

    ' Hide before long remove so the modeless progress form can show.
    Me.Hide
    If RemoveTaskFromWorkPackage(orn, rowId, True) Then
        RefreshForm
    End If
    Me.Show
End Sub

Private Sub cmdCloseWP_Click()
    Dim orn As String

    orn = SelectedORN()
    If Len(orn) = 0 Then Exit Sub

    ' Hide before long close so the modeless progress form can show.
    Me.Hide
    If CloseWorkPackage(orn, True) Then
        RefreshForm
    End If
    Me.Show
End Sub

Private Sub cmdCloseForm_Click()
    Unload Me
End Sub

Private Sub RefreshForm()
    Dim orns As Collection
    Dim i As Long
    Dim previous As String

    previous = SelectedORN()
    txtOpenCount.Value = CStr(CountOpenWorkPackages())
    txtOpenCount.Locked = True
    txtOpenCount.Enabled = True

    cboORN.Clear
    Set orns = GetOpenWorkPackageORNs()
    For i = 1 To orns.Count
        cboORN.AddItem CStr(orns(i))
    Next i

    If Len(previous) > 0 Then
        For i = 0 To cboORN.ListCount - 1
            If StrComp(CStr(cboORN.List(i)), previous, vbTextCompare) = 0 Then
                cboORN.ListIndex = i
                Exit For
            End If
        Next i
    End If

    If cboORN.ListIndex < 0 And cboORN.ListCount > 0 Then
        cboORN.ListIndex = 0
    End If

    PopulateTaskList
    UpdateActionButtons
End Sub

Private Sub PopulateTaskList()
    Dim orn As String
    Dim tasks As Collection
    Dim i As Long
    Dim rowData As Variant

    lstTasks.Clear
    lstTasks.ColumnCount = 3
    lstTasks.ColumnWidths = "160 pt;280 pt;0 pt"
    LayoutWorkPackageListHeaders

    orn = SelectedORN()
    If Len(orn) = 0 Then Exit Sub

    Set tasks = GetWorkPackageTaskDisplayRows(orn)
    For i = 1 To tasks.Count
        rowData = tasks(i)
        lstTasks.AddItem CStr(rowData(0))
        lstTasks.List(lstTasks.ListCount - 1, 1) = CStr(rowData(1))
        lstTasks.List(lstTasks.ListCount - 1, 2) = CStr(rowData(2))
    Next i
End Sub

' ColumnHeads only works with RowSource. These lists use AddItem, so align
' Labels above lstTasks to the visible ColumnWidths instead.
Private Sub LayoutWorkPackageListHeaders()
    Const GAP As Double = 2
    Const HDR_HEIGHT As Double = 12
    Dim leftPos As Double
    Dim topPos As Double

    leftPos = lstTasks.Left
    topPos = lstTasks.Top - HDR_HEIGHT - GAP

    With lblHdrTaskCode
        .Caption = "Task Code"
        .Left = leftPos
        .Top = topPos
        .Width = 160
        .Height = HDR_HEIGHT
        .Font.Bold = True
    End With
    leftPos = leftPos + 160

    With lblHdrDescription
        .Caption = "Description"
        .Left = leftPos
        .Top = topPos
        .Width = 280
        .Height = HDR_HEIGHT
        .Font.Bold = True
    End With
End Sub

Private Sub UpdateActionButtons()
    Dim hasOrn As Boolean
    Dim hasTask As Boolean

    hasOrn = (Len(SelectedORN()) > 0)
    hasTask = (Len(SelectedTaskRowId()) > 0)
    cmdDelete.Enabled = hasOrn
    cmdCloseWP.Enabled = hasOrn
    cmdRemoveTask.Enabled = hasOrn And hasTask
End Sub

Private Function SelectedORN() As String
    If cboORN.ListIndex < 0 Then Exit Function
    SelectedORN = Trim$(CStr(cboORN.Value))
End Function

Private Function SelectedTaskRowId() As String
    If lstTasks.ListIndex < 0 Then Exit Function
    SelectedTaskRowId = Trim$(CStr(lstTasks.List(lstTasks.ListIndex, 2)))
End Function
