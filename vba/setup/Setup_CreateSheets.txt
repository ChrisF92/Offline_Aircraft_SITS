Attribute VB_Name = "modSetupSheets"
Option Explicit

' ========================================================================
' ONE-TIME SETUP — Create worksheets
' ========================================================================
' Run from a new macro-enabled workbook after importing CONFIG and modUtility.
' ========================================================================

Public Sub Setup_CreateSheets()
    Dim response As VbMsgBoxResult

    response = MsgBox("This will create all worksheets for " & APP_NAME & " v" & APP_VERSION & "." & vbCrLf & vbCrLf & _
                      "Continue?", vbQuestion + vbYesNo, "Setup — Sheets")
    If response <> vbYes Then Exit Sub

    SetExcelState "disable"
    Application.DisplayAlerts = False

    PrepareWorkbookForSetup
    ReuseSheetAs SHEET_DASHBOARD, xlSheetVisible, COLOR_TAB_VISIBLE
    CreateAdditionalSheets
    OrderSheets

    Application.DisplayAlerts = True
    SetExcelState "enable"

    MsgBox "Worksheets created successfully.", vbInformation, "Setup — Sheets"
End Sub

Private Sub CreateAdditionalSheets()
    AddSetupSheet SHEET_SITS, xlSheetVisible, COLOR_TAB_VISIBLE
    AddSetupSheet SHEET_USAGE, xlSheetVisible, COLOR_TAB_VISIBLE
    AddSetupSheet SHEET_AUDIT, xlSheetVisible, COLOR_TAB_SYSTEM
    AddSetupSheet SHEET_CONFIG, xlSheetVeryHidden, COLOR_TAB_DATA
    AddSetupSheet SHEET_DATA_SITS, xlSheetVeryHidden, COLOR_TAB_DATA
    AddSetupSheet SHEET_DATA_USAGE_BASELINE, xlSheetVeryHidden, COLOR_TAB_DATA
    AddSetupSheet SHEET_DATA_USAGE_LOG, xlSheetVeryHidden, COLOR_TAB_DATA
    AddSetupSheet SHEET_DATA_USAGE_CURRENT, xlSheetVeryHidden, COLOR_TAB_DATA
    AddSetupSheet SHEET_DATA_WP, xlSheetVeryHidden, COLOR_TAB_DATA
    AddSetupSheet SHEET_DATA_WP_TASKS, xlSheetVeryHidden, COLOR_TAB_DATA
    AddSetupSheet SHEET_WP_STATE, xlSheetVeryHidden, COLOR_TAB_SYSTEM
End Sub

Private Sub OrderSheets()
    Dim order As Variant
    Dim i As Long

    order = Array( _
        SHEET_DASHBOARD, SHEET_SITS, SHEET_USAGE, SHEET_AUDIT, _
        SHEET_CONFIG, SHEET_DATA_SITS, SHEET_DATA_USAGE_BASELINE, _
        SHEET_DATA_USAGE_LOG, SHEET_DATA_USAGE_CURRENT, _
        SHEET_DATA_WP, SHEET_DATA_WP_TASKS, SHEET_WP_STATE _
    )

    For i = LBound(order) To UBound(order)
        ThisWorkbook.Worksheets(CStr(order(i))).Move Before:=ThisWorkbook.Worksheets(i + 1)
    Next i
End Sub
