Attribute VB_Name = "modSetupSheets"
Option Explicit

' ========================================================================
' ONE-TIME SETUP — Create worksheets
' ========================================================================

Public Sub Setup_CreateSheets()
    Dim response As VbMsgBoxResult

    response = MsgBox("This will create all worksheets for " & APP_NAME & " v" & APP_VERSION & "." & vbCrLf & vbCrLf & _
                      "Continue?", vbQuestion + vbYesNo, "Setup — Sheets")
    If response <> vbYes Then Exit Sub

    SetExcelState "disable"
    Application.DisplayAlerts = False

    If ThisWorkbook.Worksheets.Count = 1 Then
        ReuseSheetAs SHEET_DASHBOARD
    Else
        PrepareWorkbookForSetup
        ReuseSheetAs SHEET_DASHBOARD
    End If

    CreateAdditionalSheets
    OrderSheets
    FinalizeSetupSheetLayout

    Application.DisplayAlerts = True
    SetExcelState "enable"

    MsgBox "Worksheets created successfully.", vbInformation, "Setup — Sheets"
End Sub

Private Sub CreateAdditionalSheets()
    AddSetupSheet SHEET_SITS
    AddSetupSheet SHEET_USAGE
    AddSetupSheet SHEET_AUDIT
    AddSetupSheet SHEET_CONFIG
    AddSetupSheet SHEET_DATA_SITS
    AddSetupSheet SHEET_DATA_USAGE_BASELINE
    AddSetupSheet SHEET_DATA_USAGE_LOG
    AddSetupSheet SHEET_DATA_USAGE_CURRENT
    AddSetupSheet SHEET_DATA_WP
    AddSetupSheet SHEET_DATA_WP_TASKS
    AddSetupSheet SHEET_WP_STATE
End Sub

Private Sub OrderSheets()
    Dim order As Variant
    Dim i As Long

    order = Array( _
        SHEET_DASHBOARD, SHEET_SITS, SHEET_USAGE, SHEET_AUDIT, _
        SHEET_CONFIG, SHEET_DATA_SITS, SHEET_DATA_USAGE_BASELINE, _
        SHEET_DATA_USAGE_LOG, SHEET_DATA_USAGE_CURRENT, _
        SHEET_DATA_WP, SHEET_DATA_WP_TASKS, SHEET_WP_STATE _
    )

    For i = LBound(order) To UBound(order)
        ThisWorkbook.Worksheets(CStr(order(i))).Move Before:=ThisWorkbook.Worksheets(i + 1)
    Next i
End Sub
