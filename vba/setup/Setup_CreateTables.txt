Option Explicit

' ========================================================================
' ONE-TIME SETUP — Create Excel tables
' ========================================================================

Public Sub Setup_CreateTables(Optional ByVal silent As Boolean = False)
    If Not silent Then
        If MsgBox("Create all data tables?", vbQuestion + vbYesNo, "Setup — Tables") <> vbYes Then Exit Sub
    End If

    SetExcelState "disable"
    On Error GoTo ErrorHandler

    BuildAllTables

    SetExcelState "enable"
    If Not silent Then MsgBox "Tables created successfully.", vbInformation, "Setup — Tables"
    Exit Sub

ErrorHandler:
    SetExcelState "enable"
    MsgBox "Error creating tables: " & Err.Number & " - " & Err.Description, vbCritical, "Setup — Tables"
End Sub

Private Sub BuildAllTables()
    CreateSITSTable
    CreateUsageBaselineTable
    CreateUsageLogTable
    CreateUsageCurrentTable
    CreateWorkPackageTables
    CreateConfigTables
    CreateAuditTable
    FormatViewSheets
    ApplyWorkbookDataFormats
End Sub

Private Sub CreateSITSTable()
    Dim ws As Worksheet
    Dim headers As Variant

    Set ws = GetWorksheet(SHEET_DATA_SITS)
    headers = Array( _
        HDR_SELECTOR, HDR_ORN_INWORK, HDR_ORN_COMPLETED, _
        HDR_HOURS_WHEN_COMPLETE, HDR_NUMBER_WHEN_COMPLETE, HDR_DATE_WHEN_COMPLETE, _
        HDR_INTERVAL_TYPE_CODE, HDR_INTERVAL_TYPE, _
        HDR_CURRENT_USAGE_HOURS, HDR_CURRENT_USAGE_NUMBER, HDR_END_ITEM_SERIAL, _
        HDR_HOURS_LIFE_REMAINING, HDR_NUMBER_LIFE_REMAINING, HDR_DAYS_WHEN_DUE, _
        HDR_EXTENSION, HDR_EXTENSION_SNOW, HDR_TASK_TYPE_CODE, HDR_ACTUAL_INTERVAL, _
        HDR_WARNING_INTERVAL, HDR_TASK_CODE, HDR_TASK_SEQUENCE, HDR_TASK_DESCRIPTION, _
        HDR_SERIAL, HDR_PROJECTED_DUE_DATE, HDR_SITS_ROW_ID, _
        HDR_BASELINE_LR_HOURS, HDR_BASELINE_LR_NUMBER, HDR_BASELINE_DAYS_DUE, _
        HDR_NEEDS_MANUAL_UPDATE _
    )

    CreateTableOnSheet ws, TBL_SITS, headers
End Sub

    )

Private Sub CreateWorkPackageTables()
    Dim ws As Worksheet

    Set ws = GetWorksheet(SHEET_DATA_WP)
    CreateTableOnSheet ws, TBL_WORK_PACKAGES, Array( _
        HDR_WP_ORN, HDR_WP_STATUS, HDR_WP_CREATED_AT, _
        HDR_WP_CREATED_BY, HDR_WP_COMPLETED_AT _
    )

    Set ws = GetWorksheet(SHEET_DATA_WP_TASKS)
    CreateTableOnSheet ws, TBL_WP_TASKS, Array( _
        HDR_WP_TASK_ORN, HDR_WP_TASK_ROW_ID, HDR_WP_TASK_COMPLETED _
    )

    Set ws = GetWorksheet(SHEET_WP_STATE)
    ws.Cells(1, 1).Value = "ORN"
    ws.Cells(1, 2).Value = "SITS_RowID"
    ws.Cells(1, 3).Value = "Checked"
    StyleHeaderRow ws.Range("A1:C1")
End Sub

Private Sub CreateConfigTables()
    Dim ws As Worksheet
    Dim tbl As ListObject

    Set ws = GetWorksheet(SHEET_CONFIG)
    ws.Cells.Clear

    ws.Cells(1, 1).Value = "Setting"
    ws.Cells(1, 2).Value = "Value"
    StyleHeaderRow ws.Range("A1:B1")
    Set tbl = ws.ListObjects.Add(xlSrcRange, ws.Range("A1:B1"), , xlYes)
    tbl.Name = TBL_CONFIG
    tbl.TableStyle = "TableStyleMedium2"

    ws.Cells(1, 4).Value = "SourceTaskCode"
    ws.Cells(1, 5).Value = "SourceSerial"
    ws.Cells(1, 6).Value = "TargetIntervalCode"
    ws.Cells(1, 7).Value = "Notes"
    StyleHeaderRow ws.Range("D1:G1")
    Set tbl = ws.ListObjects.Add(xlSrcRange, ws.Range("D1:G1"), , xlYes)
    tbl.Name = TBL_INTERVAL_MAP
    tbl.TableStyle = "TableStyleMedium4"

    ws.Range("I1").Value = "Notes"
    ws.Range("I2").Value = "Runtime settings are stored in tblConfig."
    ws.Range("I3").Value = "tblIntervalMap holds optional import mappings."
    ws.Range("I4").Value = "tblUsageSnapshot (col M) stores usage at last SITS import."

    CreateUsageSnapshotTable ws
End Sub

Private Sub CreateUsageSnapshotTable(ByVal ws As Worksheet)
    Dim tbl As ListObject

    ws.Cells(1, 13).Value = HDR_SNAPSHOT_INTERVAL_CODE
    ws.Cells(1, 14).Value = HDR_SNAPSHOT_VALUE
    ws.Cells(1, 15).Value = HDR_SNAPSHOT_FORMAT
    StyleHeaderRow ws.Range("M1:O1")

    If TableExists(ws, TBL_USAGE_SNAPSHOT) Then
        ws.ListObjects(TBL_USAGE_SNAPSHOT).Delete
    End If

    Set tbl = ws.ListObjects.Add(xlSrcRange, ws.Range("M1:O1"), , xlYes)
    tbl.Name = TBL_USAGE_SNAPSHOT
    tbl.TableStyle = "TableStyleMedium4"
End Sub

Private Sub CreateAuditTable()
    Dim ws As Worksheet

    Set ws = GetWorksheet(SHEET_AUDIT)
    CreateTableOnSheet ws, "tblAudit", Array( _
        "Timestamp", "User", "Action", "ORN", "Details", "AppVersion" _
    )
End Sub

Private Sub CreateTableOnSheet(ByVal ws As Worksheet, _
                               ByVal tableName As String, _
                               ByVal headers As Variant, _
                               Optional ByVal startCol As Long = 1)
    Dim col As Long
    Dim lastCol As Long
    Dim headerRange As Range
    Dim tbl As ListObject

    If TableExists(ws, tableName) Then
        ws.ListObjects(tableName).Delete
    End If

    ws.Cells.Clear
    lastCol = startCol + UBound(headers) - LBound(headers)

    For col = LBound(headers) To UBound(headers)
        ws.Cells(1, startCol + col).Value = headers(col)
    Next col

    Set headerRange = ws.Range(ws.Cells(1, startCol), ws.Cells(1, lastCol))
    StyleHeaderRow headerRange

    Set tbl = ws.ListObjects.Add(xlSrcRange, headerRange, , xlYes)
    tbl.Name = tableName
    tbl.TableStyle = "TableStyleMedium2"
End Sub

Private Sub StyleHeaderRow(ByVal headerRange As Range)
    With headerRange
        .Font.Bold = True
        .Font.Color = vbWhite
        .Interior.Color = COLOR_HEADER_BLUE
        .HorizontalAlignment = xlCenter
    End With
End Sub

Private Sub FormatViewSheets()
    FormatSITSView
    FormatUsageView
    FormatDashboardShell
End Sub

Private Sub FormatSITSView()
    Dim ws As Worksheet

    Set ws = GetWorksheet(SHEET_SITS)
    ws.Cells.Clear

    With ws.Range("A1")
        .Value = "SITS — Task View"
        .Font.Size = 14
        .Font.Bold = True
        .Font.Color = COLOR_HEADER_BLUE
    End With

    ws.Range("A2").Value = "Tail:"
    ws.Range("B2").Value = "(set during setup)"
    ws.Range("A3").Value = "Forecast hours (HH/E1/E2):"
    ws.Range("B3").Value = 50
    ws.Range("A4").Value = "Forecast days (DD):"
    ws.Range("B4").Value = 90
    ws.Range("A5").Value = "Use filter dropdowns on row 7 to hide interval types. Check tasks to select for future work packages."
End Sub

Private Sub FormatUsageView()
    Dim ws As Worksheet

    Set ws = GetWorksheet(SHEET_USAGE)
    ws.Cells.Clear

    With ws.Range("A1")
        .Value = "Usage — Current Totals & Sortie History"
        .Font.Size = 14
        .Font.Bold = True
        .Font.Color = COLOR_HEADER_BLUE
    End With

    ws.Range("A3").Value = "Run RefreshUsageView after setup, or log a sortie to refresh this sheet."
End Sub
