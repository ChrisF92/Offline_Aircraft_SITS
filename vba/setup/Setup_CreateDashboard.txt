Attribute VB_Name = "modSetupDashboard"
Option Explicit

' ========================================================================
' ONE-TIME SETUP — Dashboard layout
' ========================================================================

Public Sub Setup_CreateDashboard(Optional ByVal silent As Boolean = False)
    If Not silent Then
        If MsgBox("Format the Dashboard sheet?", vbQuestion + vbYesNo, "Setup — Dashboard") <> vbYes Then Exit Sub
    End If

    Dim ws As Worksheet
    Set ws = GetWorksheet(SHEET_DASHBOARD)

    SetExcelState "disable"
    ws.Cells.Clear
    ws.Columns("A").ColumnWidth = 2
    ws.Columns("B").ColumnWidth = 24
    ws.Columns("C").ColumnWidth = 18
    ws.Columns("D").ColumnWidth = 40

    With ws.Range("B2")
        .Value = APP_NAME
        .Font.Size = 18
        .Font.Bold = True
        .Font.Color = COLOR_HEADER_BLUE
    End With

    ws.Range("B3").Value = "Version " & APP_VERSION & "  |  " & AIRCRAFT_TYPE
    ws.Range("B3").Font.Color = RGB(89, 89, 89)

    WriteSectionHeader ws, "B5", "Aircraft Status"
    WriteLabelValue ws, "B6", "Tail Number:", "=IFERROR(INDEX(tblConfig[Value],MATCH(""" & CFG_TAIL_NUMBER & """,tblConfig[Setting],0)),"""" )"
    WriteLabelValue ws, "B7", "Setup Complete:", "=IFERROR(INDEX(tblConfig[Value],MATCH(""" & CFG_SETUP_COMPLETE & """,tblConfig[Setting],0)),FALSE)"
    WriteLabel ws, "B8", "Active WP (ORN):"
    ws.Range("C8").Value = ""
    ws.Range("C8").Interior.Color = COLOR_INPUT_HIGHLIGHT

    WriteSectionHeader ws, "B10", "Extensions"
    WriteLabel ws, "B11", "Extension SNOW:"
    ws.Range("C10").Interior.Color = COLOR_INPUT_HIGHLIGHT
    ws.Range("C10").Value = ""

    WriteSectionHeader ws, "B13", "Actions"
    ws.Range("B14").Value = "• Setup Wizard — assign macro RunSetupWizard"
    ws.Range("B15").Value = "• Re-import SITS — assign macro ImportSITSFromFile"
    ws.Range("B16").Value = "• Log Sortie — not yet available (Phase 2)"
    ws.Range("B17").Value = "• Work Packages — not yet available (Phase 4)"
    ws.Range("B18").Value = "• Extensions / Deferrals — not yet available (Phase 5)"
    ws.Range("B19").Value = "• View Audit Log — assign macro ViewAuditLog"

    WriteSectionHeader ws, "B21", "Import Requirements"
    ws.Range("B22").Value = "SITS Long Forecast: 9 required columns (strict — abort if any missing)."
    ws.Range("B23").Value = "Usage exports: Interval Type, Task Code, Life Used, Top Serial."
    ws.Range("B24").Value = "Engine usage must include SM and SN (WC/STARTS/ECU)."

    ws.Range("B26").Value = "Phase 1 — run RunSetupWizard after Phase 0 setup to import SITS and usage."
    ws.Range("B26").Font.Italic = True

    ws.Activate
    ws.Range("B2").Select
    SetExcelState "enable"
    If Not silent Then MsgBox "Dashboard formatted.", vbInformation, "Setup — Dashboard"
End Sub

Private Sub WriteSectionHeader(ByVal ws As Worksheet, ByVal cellAddress As String, ByVal title As String)
    With ws.Range(cellAddress)
        .Value = title
        .Font.Bold = True
        .Font.Size = 12
        .Font.Color = COLOR_HEADER_BLUE
    End With
End Sub

Private Sub WriteLabel(ByVal ws As Worksheet, ByVal labelCell As String, ByVal caption As String)
    ws.Range(labelCell).Value = caption
    ws.Range(labelCell).Font.Bold = True
End Sub

Private Sub WriteLabelValue(ByVal ws As Worksheet, ByVal labelCell As String, ByVal caption As String, ByVal formulaText As String)
    WriteLabel ws, labelCell, caption
    ws.Range("C" & Mid$(labelCell, 2)).Formula = formulaText
End Sub
