Option Explicit

' ========================================================================
' DASHBOARD — Layout, status, user actions, and Developer Tools
' Module Name property: modSetupDashboard
' ========================================================================
' RefreshDashboardLayout fully rebuilds the sheet (explicit Rebuild only).
' EnsureDashboardActionButtons only adds missing buttons so developer
' placement and column widths survive day-to-day use.
'
' Pre-setup / master: Setup Wizard + View Help under User Actions.
' After setup: Log Sortie, Work Packages, View Audit Log, View Help.
' Extend / Defer live on the SITS sheet only (not Dashboard).
' Import SITS and Refresh Usage live under Developer Tools.
'
' Developer Tools are hidden by default. Run EnableDevTools (Alt+F8) to
' show the whole section; HideDevTools (or the button) to hide it again.
' ========================================================================

Private Const BTN_WIDTH As Double = 140
Private Const BTN_HEIGHT As Double = 26
Private Const BTN_COL_GAP As Double = 12
Private Const BTN_ROW_HEIGHT As Double = 32

Private Const USER_BTN_START_ROW As Long = 14
Private Const DEV_HEADER_ROW As Long = 18
Private Const DEV_HELP_ROW As Long = 19
Private Const DEV_BTN_START_ROW As Long = 20
Private Const DEV_BTN_LAST_OFFSET As Long = 6

' Full rebuild — explicit "Rebuild Dashboard" only.
Public Sub RefreshDashboardLayout()
    Dim ws As Worksheet

    Set ws = GetWorksheet(SHEET_DASHBOARD)
    SetExcelState "disable"
    RemoveDashboardButtons ws
    ws.Cells.Clear
    ws.Columns("A").ColumnWidth = 2
    ws.Columns("B").ColumnWidth = 32
    ws.Columns("C").ColumnWidth = 18
    ws.Columns("D").ColumnWidth = 28

    With ws.Range("B2")
        .Value = APP_NAME
        .Font.Size = 18
        .Font.Bold = True
        .Font.Color = COLOR_HEADER_BLUE
    End With

    ws.Range("B3").Value = "Version " & APP_VERSION & "  |  " & AIRCRAFT_TYPE
    ws.Range("B3").Font.Color = RGB(89, 89, 89)
    WriteWorkbookStatusLine ws

    WriteSectionHeader ws, "B5", "Aircraft Status"
    WriteLabelValue ws, "B6", "Tail Number:", _
        "=IFERROR(INDEX(tblConfig[Value],MATCH(""" & CFG_TAIL_NUMBER & """,tblConfig[Setting],0)),"""")"
    WriteLabelValue ws, "B7", "Setup Complete:", _
        "=IFERROR(INDEX(tblConfig[Value],MATCH(""" & CFG_SETUP_COMPLETE & """,tblConfig[Setting],0)),FALSE)"
    RepairDashboardConfigFormulas

    WriteLabel ws, "B8", "Open work packages:"
    ws.Range("C8").Value = CountOpenWorkPackages()
    ws.Range("C8").NumberFormat = FORMAT_INTEGER
    ws.Range("C8").Interior.Color = COLOR_INPUT_HIGHLIGHT

    WriteLabel ws, "B9", "Tasks needing manual update:"
    ws.Range("C9").Value = CountManualUpdateTasks()
    ws.Range("C9").NumberFormat = FORMAT_INTEGER
    ws.Range("C9").Interior.Color = COLOR_MANUAL_BLUE
    SetCellCommentText ws.Range("C9"), BuildManualUpdateDashboardComment()

    WriteLabel ws, "B10", "Tasks in warning period:"
    ws.Range("C10").Value = CountWarningPeriodTasks()
    ws.Range("C10").NumberFormat = FORMAT_INTEGER
    ws.Range("C10").Interior.Color = COLOR_WARN_AMBER

    WriteLabel ws, "B11", "Tasks overdue:"
    ws.Range("C11").Value = CountOverdueTasks()
    ws.Range("C11").NumberFormat = FORMAT_INTEGER
    ws.Range("C11").Interior.Color = COLOR_WARN_PINK

    CenterStatusValueCells ws

    WriteSectionHeader ws, "B13", "User Actions"
    PrepareUserButtonRows ws
    SyncUserActionButtons ws, forceAdd:=True

    Setup_CreateNamedRanges True

    If AreDevToolsEnabled() Then
        ShowDashboardDevToolsSection ws, forceAdd:=True
    End If

    ws.Activate
    ws.Range("B2").Select
    SetExcelState "enable"
End Sub

' Day-to-day / post-setup: update status cells only — never clear or resize.
Public Sub RefreshDashboardStatusCounts()
    Dim ws As Worksheet
    Dim manualCount As Long
    Dim commentText As String

    Set ws = GetWorksheet(SHEET_DASHBOARD, showError:=False)
    If ws Is Nothing Then Exit Sub

    On Error Resume Next
    ws.Range("C8").Value = CountOpenWorkPackages()
    ws.Range("C8").NumberFormat = FORMAT_INTEGER

    manualCount = CountManualUpdateTasks()
    ws.Range("C9").Value = manualCount
    ws.Range("C9").NumberFormat = FORMAT_INTEGER

    commentText = BuildManualUpdateDashboardComment()
    SetCellCommentText ws.Range("C9"), commentText

    ws.Range("C10").Value = CountWarningPeriodTasks()
    ws.Range("C10").NumberFormat = FORMAT_INTEGER
    ws.Range("C10").Interior.Color = COLOR_WARN_AMBER

    ws.Range("C11").Value = CountOverdueTasks()
    ws.Range("C11").NumberFormat = FORMAT_INTEGER
    ws.Range("C11").Interior.Color = COLOR_WARN_PINK

    CenterStatusValueCells ws
    WriteWorkbookStatusLine ws
    Err.Clear
    On Error GoTo 0
End Sub

' Add any missing Dashboard buttons at default anchors; never move existing ones.
Public Sub EnsureDashboardActionButtons()
    Dim ws As Worksheet
    Dim errNum As Long
    Dim errDesc As String
    Dim suspended As Boolean

    Set ws = GetWorksheet(SHEET_DASHBOARD, showError:=False)
    If ws Is Nothing Then Exit Sub

    SetExcelState "disable"
    On Error GoTo ErrorHandler
    SuspendWorkbookProtection
    suspended = True

    PrepareUserButtonRows ws
    SyncUserActionButtons ws, forceAdd:=False
    If AreDevToolsEnabled() Then
        ShowDashboardDevToolsSection ws, forceAdd:=False
    Else
        HideDashboardDevToolsSection ws
    End If

    ResumeWorkbookProtection
    suspended = False
    SetExcelState "enable"
    Exit Sub

ErrorHandler:
    errNum = Err.Number
    errDesc = Err.Description
    If suspended Then ResumeWorkbookProtection
    SetExcelState "enable"
    Err.Raise errNum, "EnsureDashboardActionButtons", errDesc
End Sub

' Restore Tail / Setup Complete formulas if they were overwritten or lost table refs.
Public Sub RepairDashboardConfigFormulas()
    Dim ws As Worksheet
    Dim calcWas As XlCalculation

    Set ws = GetWorksheet(SHEET_DASHBOARD, showError:=False)
    If ws Is Nothing Then Exit Sub

    On Error Resume Next
    calcWas = Application.Calculation
    ' Ensure these lookups calculate immediately (avoid Excel "stale" strikethrough).
    Application.Calculation = xlCalculationAutomatic

    With ws.Range("C6")
        .Formula = "=IFERROR(INDEX(tblConfig[Value],MATCH(""" & CFG_TAIL_NUMBER & """,tblConfig[Setting],0)),"""")"
        .Font.Strikethrough = False
        .Font.Name = "Calibri"
        .Font.Size = 11
        .Font.Color = RGB(0, 0, 0)
        .HorizontalAlignment = xlCenter
        .VerticalAlignment = xlCenter
    End With
    With ws.Range("C7")
        .Formula = "=IFERROR(INDEX(tblConfig[Value],MATCH(""" & CFG_SETUP_COMPLETE & """,tblConfig[Setting],0)),FALSE)"
        .Font.Strikethrough = False
        .Font.Name = "Calibri"
        .Font.Size = 11
        .Font.Color = RGB(0, 0, 0)
        .HorizontalAlignment = xlCenter
        .VerticalAlignment = xlCenter
    End With

    ws.Range("C6:C7").Calculate
    Application.Calculate
    If calcWas <> 0 Then Application.Calculation = calcWas
    On Error GoTo 0
End Sub

' Alt+F8 → EnableDevTools — reveals the whole Developer Tools section.
Public Sub EnableDevTools()
    Dim ws As Worksheet

    SetConfigValue CFG_SHOW_DEV_TOOLS, True
    Set ws = GetWorksheet(SHEET_DASHBOARD, showError:=False)
    If ws Is Nothing Then
        MsgBox "Dashboard sheet not found.", vbExclamation, APP_NAME
        Exit Sub
    End If

    SetExcelState "disable"
    ShowDashboardDevToolsSection ws, forceAdd:=True
    SetExcelState "enable"
    MsgBox "Developer Tools are now visible on the Dashboard." & vbCrLf & _
           "Run HideDevTools (or click Hide Dev Tools) to hide them again.", _
           vbInformation, APP_NAME
End Sub

' Alt+F8 → HideDevTools — or use the button inside Developer Tools.
Public Sub HideDevTools()
    Dim ws As Worksheet

    SetConfigValue CFG_SHOW_DEV_TOOLS, False
    Set ws = GetWorksheet(SHEET_DASHBOARD, showError:=False)
    If ws Is Nothing Then Exit Sub

    SetExcelState "disable"
    HideDashboardDevToolsSection ws
    SetExcelState "enable"
End Sub

Public Function AreDevToolsEnabled() As Boolean
    AreDevToolsEnabled = CBool(GetConfigValue(CFG_SHOW_DEV_TOOLS, False))
End Function

Private Sub WriteWorkbookStatusLine(ByVal ws As Worksheet)
    With ws.Range("B4")
        If IsMasterTemplate() Then
            .Value = "MASTER TEMPLATE — save a copy for each aircraft, then run Setup Wizard on the copy."
            .Font.Color = RGB(192, 0, 0)
            .Font.Bold = True
        ElseIf IsSetupComplete() Then
            .Value = "Aircraft workbook — setup complete."
            .Font.Color = RGB(0, 112, 60)
        Else
            .Value = "Aircraft workbook — run Setup Wizard to import data for this tail."
            .Font.Color = RGB(192, 96, 0)
        End If
        .Font.Size = 10
    End With
End Sub

' Before Setup Wizard completes (and on the master), only Setup Wizard is shown.
Private Function IsPreSetupUserMode() As Boolean
    IsPreSetupUserMode = (Not IsSetupComplete()) Or IsMasterTemplate()
End Function

Private Sub SyncUserActionButtons(ByVal ws As Worksheet, ByVal forceAdd As Boolean)
    ' Extend / Defer belong on the SITS sheet only — strip any Dashboard leftovers.
    RemoveDashboardButtonForMacro ws, "RunExtendFromSelection"
    RemoveDashboardButtonForMacro ws, "RunDeferFromSelection"

    If IsPreSetupUserMode() Then
        RemoveDashboardButtonForMacro ws, "RunLogSortie"
        RemoveDashboardButtonForMacro ws, "RunManageWorkPackages"
        RemoveDashboardButtonForMacro ws, "ViewAuditLog"

        If forceAdd Then
            AddDashboardButtonAt ws, USER_BTN_START_ROW, 0, "Setup Wizard", "RunSetupWizard"
            AddDashboardButtonAt ws, USER_BTN_START_ROW, 1, "View Help", "ViewHelp"
        Else
            EnsureDashboardButtonAt ws, USER_BTN_START_ROW, 0, "Setup Wizard", "RunSetupWizard"
            EnsureDashboardButtonAt ws, USER_BTN_START_ROW, 1, "View Help", "ViewHelp"
        End If
    Else
        RemoveDashboardButtonForMacro ws, "RunSetupWizard"

        If forceAdd Then
            AddDashboardButtonAt ws, USER_BTN_START_ROW, 0, "Log Sortie", "RunLogSortie"
            AddDashboardButtonAt ws, USER_BTN_START_ROW, 1, "Work Packages", "RunManageWorkPackages"
            AddDashboardButtonAt ws, USER_BTN_START_ROW + 1, 0, "View Audit Log", "ViewAuditLog"
            AddDashboardButtonAt ws, USER_BTN_START_ROW + 1, 1, "View Help", "ViewHelp"
        Else
            EnsureDashboardButtonAt ws, USER_BTN_START_ROW, 0, "Log Sortie", "RunLogSortie"
            EnsureDashboardButtonAt ws, USER_BTN_START_ROW, 1, "Work Packages", "RunManageWorkPackages"
            EnsureDashboardButtonAt ws, USER_BTN_START_ROW + 1, 0, "View Audit Log", "ViewAuditLog"
            EnsureDashboardButtonAt ws, USER_BTN_START_ROW + 1, 1, "View Help", "ViewHelp"
        End If
    End If

    ' Import SITS / Refresh Usage are developer-only — remove leftovers when tools are hidden.
    If Not AreDevToolsEnabled() Then
        RemoveDashboardButtonForMacro ws, "RunImportSITSFromFile"
        RemoveDashboardButtonForMacro ws, "RefreshUsageView"
    End If
End Sub

Private Sub ShowDashboardDevToolsSection(ByVal ws As Worksheet, ByVal forceAdd As Boolean)
    WriteSectionHeader ws, "B" & DEV_HEADER_ROW, "Developer Tools"
    With ws.Range("B" & DEV_HELP_ROW)
        .Value = "Rebuild macros after pasting updated VBA modules. Rebuild Dashboard resets layout; " & _
                 "Ensure Missing Buttons only adds new ones. Reset to Master clears aircraft data without " & _
                 "touching formatting. Protect/Unprotect Workbook for layout edits. " & _
                 "Run HideDevTools to hide this section."
        .Font.Italic = True
        .Font.Size = 9
        .Font.Color = RGB(89, 89, 89)
    End With

    PrepareDevButtonRows ws

    If forceAdd Then
        RemoveDevToolButtons ws
        AddDevToolButtons ws
    Else
        EnsureDevToolButtons ws
    End If
End Sub

Private Sub HideDashboardDevToolsSection(ByVal ws As Worksheet)
    RemoveDevToolButtons ws
    On Error Resume Next
    ws.Range("B" & DEV_HEADER_ROW & ":D" & (DEV_BTN_START_ROW + DEV_BTN_LAST_OFFSET)).ClearContents
    ws.Range("B" & DEV_HEADER_ROW & ":D" & (DEV_BTN_START_ROW + DEV_BTN_LAST_OFFSET)).ClearComments
    On Error GoTo 0
End Sub

Private Sub AddDashboardButtonAt(ByVal ws As Worksheet, ByVal rowNum As Long, _
                                 ByVal colIndex As Long, ByVal caption As String, _
                                 ByVal macroName As String)
    Dim anchor As Range
    Dim btn As Button
    Dim leftPos As Double

    Set anchor = ws.Range("B" & rowNum)
    leftPos = anchor.Left + (colIndex * (BTN_WIDTH + BTN_COL_GAP))
    Set btn = ws.Buttons.Add(leftPos, anchor.Top, BTN_WIDTH, BTN_HEIGHT)
    btn.OnAction = macroName
    btn.Caption = caption
    btn.Placement = xlFreeFloating
End Sub

Private Sub EnsureDashboardButtonAt(ByVal ws As Worksheet, ByVal rowNum As Long, _
                                    ByVal colIndex As Long, ByVal caption As String, _
                                    ByVal macroName As String)
    If DashboardHasButtonForMacro(ws, macroName) Then Exit Sub
    AddDashboardButtonAt ws, rowNum, colIndex, caption, macroName
End Sub

Private Sub RemoveDashboardButtonForMacro(ByVal ws As Worksheet, ByVal macroName As String)
    Dim btn As Button
    Dim actionName As String
    Dim bangPos As Long

    On Error Resume Next
    For Each btn In ws.Buttons
        actionName = Trim$(CStr(btn.OnAction))
        bangPos = InStrRev(actionName, "!")
        If bangPos > 0 Then actionName = Mid$(actionName, bangPos + 1)
        If StrComp(actionName, macroName, vbTextCompare) = 0 Then
            btn.Delete
        End If
    Next btn
    On Error GoTo 0
End Sub

Private Function DashboardHasButtonForMacro(ByVal ws As Worksheet, ByVal macroName As String) As Boolean
    Dim btn As Button
    Dim actionName As String
    Dim bangPos As Long

    DashboardHasButtonForMacro = False
    On Error Resume Next
    For Each btn In ws.Buttons
        actionName = Trim$(CStr(btn.OnAction))
        bangPos = InStrRev(actionName, "!")
        If bangPos > 0 Then actionName = Mid$(actionName, bangPos + 1)
        If StrComp(actionName, macroName, vbTextCompare) = 0 Then
            DashboardHasButtonForMacro = True
            Exit Function
        End If
    Next btn
    On Error GoTo 0
End Function

Private Sub PrepareUserButtonRows(ByVal ws As Worksheet)
    ws.Rows(USER_BTN_START_ROW & ":" & (USER_BTN_START_ROW + 1)).RowHeight = BTN_ROW_HEIGHT
End Sub

Private Sub PrepareDevButtonRows(ByVal ws As Worksheet)
    ws.Rows(DEV_BTN_START_ROW & ":" & (DEV_BTN_START_ROW + DEV_BTN_LAST_OFFSET)).RowHeight = BTN_ROW_HEIGHT
End Sub

Private Sub RemoveDashboardButtons(ByVal ws As Worksheet)
    Dim btn As Button
    Dim shp As Shape

    On Error Resume Next
    For Each btn In ws.Buttons
        btn.Delete
    Next btn
    For Each shp In ws.Shapes
        If shp.Type = msoFormControl Then
            shp.Delete
        End If
    Next shp
    On Error GoTo 0
End Sub

Private Sub CenterStatusValueCells(ByVal ws As Worksheet)
    With ws.Range("C6:C11")
        .HorizontalAlignment = xlCenter
        .VerticalAlignment = xlCenter
    End With
End Sub

Private Sub WriteSectionHeader(ByVal ws As Worksheet, ByVal cellAddress As String, ByVal title As String)
    With ws.Range(cellAddress)
        .Value = title
        .Font.Bold = True
        .Font.Size = 12
        .Font.Color = COLOR_HEADER_BLUE
    End With
End Sub

Private Sub WriteLabel(ByVal ws As Worksheet, ByVal labelCell As String, ByVal caption As String)
    ws.Range(labelCell).Value = caption
    ws.Range(labelCell).Font.Bold = True
End Sub

Private Sub WriteLabelValue(ByVal ws As Worksheet, ByVal labelCell As String, _
                            ByVal caption As String, ByVal formulaText As String)
    WriteLabel ws, labelCell, caption
    ws.Range("C" & Mid$(labelCell, 2)).Formula = formulaText
End Sub
