Option Explicit

' ========================================================================
' ONE-TIME SETUP — Dashboard layout and action buttons
' ========================================================================
' RefreshDashboardLayout fully rebuilds the sheet (master setup / explicit
' Rebuild). EnsureDashboardActionButtons only adds missing buttons so
' developer placement and column widths survive day-to-day use.
' Setup Wizard is shown only until aircraft setup completes (kept on master).
' ========================================================================

Private Const BTN_WIDTH As Double = 150
Private Const BTN_HEIGHT As Double = 24
Private Const BTN_COL_GAP As Double = 10

Public Sub Setup_CreateDashboard(Optional ByVal silent As Boolean = False)
    If Not silent Then
        If MsgBox("Format the Dashboard sheet and create action buttons?", _
                  vbQuestion + vbYesNo, "Setup — Dashboard") <> vbYes Then Exit Sub
    End If

    RefreshDashboardLayout
    If Not silent Then MsgBox "Dashboard formatted.", vbInformation, "Setup — Dashboard"
End Sub

' Full rebuild — master setup and explicit "Rebuild Dashboard" only.
Public Sub RefreshDashboardLayout()
    Dim ws As Worksheet

    Set ws = GetWorksheet(SHEET_DASHBOARD)
    SetExcelState "disable"
    RemoveDashboardButtons ws
    ws.Cells.Clear
    ws.Columns("A").ColumnWidth = 2
    ws.Columns("B").ColumnWidth = 28
    ws.Columns("C").ColumnWidth = 24
    ws.Columns("D").ColumnWidth = 40

    With ws.Range("B2")
        .Value = APP_NAME
        .Font.Size = 18
        .Font.Bold = True
        .Font.Color = COLOR_HEADER_BLUE
    End With

    ws.Range("B3").Value = "Version " & APP_VERSION & "  |  " & AIRCRAFT_TYPE
    ws.Range("B3").Font.Color = RGB(89, 89, 89)
    WriteWorkbookStatusLine ws

    WriteSectionHeader ws, "B5", "Aircraft Status"
    WriteLabelValue ws, "B6", "Tail Number:", _
        "=IFERROR(INDEX(tblConfig[Value],MATCH(""" & CFG_TAIL_NUMBER & """,tblConfig[Setting],0)),"""" )"
    WriteLabelValue ws, "B7", "Setup Complete:", _
        "=IFERROR(INDEX(tblConfig[Value],MATCH(""" & CFG_SETUP_COMPLETE & """,tblConfig[Setting],0)),FALSE)"
    WriteLabel ws, "B8", "Open work packages:"
    ws.Range("C8").Value = CountOpenWorkPackages()
    ws.Range("C8").NumberFormat = FORMAT_INTEGER
    ws.Range("C8").Interior.Color = COLOR_INPUT_HIGHLIGHT

    WriteLabel ws, "B9", "Tasks needing manual update:"
    ws.Range("C9").Value = CountManualUpdateTasks()
    ws.Range("C9").NumberFormat = FORMAT_INTEGER
    ws.Range("C9").Interior.Color = COLOR_MANUAL_BLUE
    SetCellCommentText ws.Range("C9"), BuildManualUpdateDashboardComment()

    WriteSectionHeader ws, "B11", "User Actions"
    PrepareButtonRows ws
    If ShouldShowSetupWizardButton() Then
        AddDashboardButton ws, "B12", "Setup Wizard", "RunSetupWizard"
    End If
    AddDashboardButton ws, "C12", "Log Sortie", "RunLogSortie"
    AddDashboardButton ws, "B13", "Import SITS", "RunImportSITSFromFile"
    AddDashboardButton ws, "C13", "Refresh Usage", "RefreshUsageView"
    AddDashboardButton ws, "B14", "Work Packages", "RunManageWorkPackages"
    AddDashboardButton ws, "C14", "View Audit Log", "ViewAuditLog"
    AddDashboardButton ws, "B15", "Extend", "RunExtendFromSelection"
    AddDashboardButton ws, "C15", "Defer", "RunDeferFromSelection"

    WriteSectionHeader ws, "B17", "Developer Tools"
    ws.Range("B18").Value = "Rebuild macros after pasting updated VBA modules. Rebuild Dashboard resets layout; Ensure Missing Buttons only adds new ones."
    ws.Range("B18").Font.Italic = True
    ws.Range("B18").Font.Size = 9
    ws.Range("B18").Font.Color = RGB(89, 89, 89)

    AddDashboardButton ws, "B19", "Repair Formats", "RepairWorkbookFormats"
    AddDashboardButton ws, "C19", "Repair Visibility", "RepairSetupSheetVisibility"
    AddDashboardButton ws, "B20", "Validate Structure", "RunValidateWorkbookStructure"
    AddDashboardButton ws, "C20", "Rebuild Dashboard", "RefreshDashboardLayout"
    AddDashboardButton ws, "B21", "Ensure Missing Buttons", "EnsureDashboardActionButtons"
    AddDashboardButton ws, "C21", "Phase 0 Setup", "Setup_RunAll"
    AddDashboardButton ws, "B22", "Rebuild SITS Layout", "RebuildSITSViewLayout"
    AddDashboardButton ws, "C22", "Format Usage Matrix", "Setup_FormatUsageMatrix"

    WriteSectionHeader ws, "B24", "Import Requirements"
    ws.Range("B25").Value = "SITS Long Forecast: 9 required columns (strict — abort if any missing)."
    ws.Range("B26").Value = "Usage exports: Interval Type, Task Code, Life Used, Top Serial."
    ws.Range("B27").Value = "Engine usage must include SM and SN (WC/STARTS/ECU)."
    ws.Range("B28").Value = "ORN format: ####TAILDDMMYY (e.g. 7034ZZ383081125)."
    ws.Range("B29").Value = "Extend: same Interval Type + Actual Interval. Defer: one PT2/PT3 at a time."
    ws.Range("B30").Value = "Log Sortie: enter Usage only (Current/New Total are calculated). HUMS Op Number optional."
    ws.Range("B31").Value = "WARE counters IL/IR/HE/FT appear on the form once present in the usage import."
    ws.Range("B32").Value = "Layout tip: tweak column widths / button positions on the Master; refresh does not reset them."

    ws.Activate
    ws.Range("B2").Select
    SetExcelState "enable"
End Sub

' Day-to-day / post-setup: update status cells only — never clear or resize.
Public Sub RefreshDashboardStatusCounts()
    Dim ws As Worksheet
    Dim manualCount As Long
    Dim commentText As String

    Set ws = GetWorksheet(SHEET_DASHBOARD, showError:=False)
    If ws Is Nothing Then Exit Sub

    On Error Resume Next
    ws.Range("C8").Value = CountOpenWorkPackages()
    ws.Range("C8").NumberFormat = FORMAT_INTEGER

    manualCount = CountManualUpdateTasks()
    ws.Range("C9").Value = manualCount
    ws.Range("C9").NumberFormat = FORMAT_INTEGER

    commentText = BuildManualUpdateDashboardComment()
    SetCellCommentText ws.Range("C9"), commentText
    WriteWorkbookStatusLine ws
    On Error GoTo 0
End Sub

' Add any missing Dashboard buttons at default anchors; never move existing ones.
Public Sub EnsureDashboardActionButtons()
    Dim ws As Worksheet

    Set ws = GetWorksheet(SHEET_DASHBOARD, showError:=False)
    If ws Is Nothing Then Exit Sub

    SetExcelState "disable"
    SyncSetupWizardButton ws
    EnsureDashboardButton ws, "C12", "Log Sortie", "RunLogSortie"
    EnsureDashboardButton ws, "B13", "Import SITS", "RunImportSITSFromFile"
    EnsureDashboardButton ws, "C13", "Refresh Usage", "RefreshUsageView"
    EnsureDashboardButton ws, "B14", "Work Packages", "RunManageWorkPackages"
    EnsureDashboardButton ws, "C14", "View Audit Log", "ViewAuditLog"
    EnsureDashboardButton ws, "B15", "Extend", "RunExtendFromSelection"
    EnsureDashboardButton ws, "C15", "Defer", "RunDeferFromSelection"
    EnsureDashboardButton ws, "B19", "Repair Formats", "RepairWorkbookFormats"
    EnsureDashboardButton ws, "C19", "Repair Visibility", "RepairSetupSheetVisibility"
    EnsureDashboardButton ws, "B20", "Validate Structure", "RunValidateWorkbookStructure"
    EnsureDashboardButton ws, "C20", "Rebuild Dashboard", "RefreshDashboardLayout"
    EnsureDashboardButton ws, "B21", "Ensure Missing Buttons", "EnsureDashboardActionButtons"
    EnsureDashboardButton ws, "C21", "Phase 0 Setup", "Setup_RunAll"
    EnsureDashboardButton ws, "B22", "Rebuild SITS Layout", "RebuildSITSViewLayout"
    EnsureDashboardButton ws, "C22", "Format Usage Matrix", "Setup_FormatUsageMatrix"
    SetExcelState "enable"
End Sub

Private Sub WriteWorkbookStatusLine(ByVal ws As Worksheet)
    With ws.Range("B4")
        If IsMasterTemplate() Then
            .Value = "MASTER TEMPLATE — save a copy for each aircraft, then run Setup Wizard on the copy."
            .Font.Color = RGB(192, 0, 0)
            .Font.Bold = True
        ElseIf IsSetupComplete() Then
            .Value = "Aircraft workbook — setup complete."
            .Font.Color = RGB(0, 112, 60)
        Else
            .Value = "Aircraft workbook — run Setup Wizard to import data for this tail."
            .Font.Color = RGB(192, 96, 0)
        End If
        .Font.Size = 10
    End With
End Sub

' Setup Wizard stays on the master template and unfinished aircraft copies only.
Private Function ShouldShowSetupWizardButton() As Boolean
    ShouldShowSetupWizardButton = (Not IsSetupComplete()) Or IsMasterTemplate()
End Function

Private Sub SyncSetupWizardButton(ByVal ws As Worksheet)
    If ShouldShowSetupWizardButton() Then
        EnsureDashboardButton ws, "B12", "Setup Wizard", "RunSetupWizard"
    Else
        RemoveDashboardButtonForMacro ws, "RunSetupWizard"
    End If
End Sub

Private Sub AddDashboardButton(ByVal ws As Worksheet, ByVal anchorCell As String, _
                               ByVal caption As String, ByVal macroName As String)
    Dim anchor As Range
    Dim btn As Button

    Set anchor = ws.Range(anchorCell)
    Set btn = ws.Buttons.Add(anchor.Left, anchor.Top, BTN_WIDTH, BTN_HEIGHT)
    btn.OnAction = macroName
    btn.Caption = caption
    btn.Placement = xlFreeFloating
End Sub

Private Sub EnsureDashboardButton(ByVal ws As Worksheet, ByVal anchorCell As String, _
                                  ByVal caption As String, ByVal macroName As String)
    If DashboardHasButtonForMacro(ws, macroName) Then Exit Sub
    AddDashboardButton ws, anchorCell, caption, macroName
End Sub

Private Sub RemoveDashboardButtonForMacro(ByVal ws As Worksheet, ByVal macroName As String)
    Dim btn As Button
    Dim actionName As String
    Dim bangPos As Long

    On Error Resume Next
    For Each btn In ws.Buttons
        actionName = Trim$(CStr(btn.OnAction))
        bangPos = InStrRev(actionName, "!")
        If bangPos > 0 Then actionName = Mid$(actionName, bangPos + 1)
        If StrComp(actionName, macroName, vbTextCompare) = 0 Then
            btn.Delete
        End If
    Next btn
    On Error GoTo 0
End Sub

Private Function DashboardHasButtonForMacro(ByVal ws As Worksheet, ByVal macroName As String) As Boolean
    Dim btn As Button
    Dim actionName As String
    Dim bangPos As Long

    DashboardHasButtonForMacro = False
    On Error Resume Next
    For Each btn In ws.Buttons
        actionName = Trim$(CStr(btn.OnAction))
        bangPos = InStrRev(actionName, "!")
        If bangPos > 0 Then actionName = Mid$(actionName, bangPos + 1)
        If StrComp(actionName, macroName, vbTextCompare) = 0 Then
            DashboardHasButtonForMacro = True
            Exit Function
        End If
    Next btn
    On Error GoTo 0
End Function

Private Sub PrepareButtonRows(ByVal ws As Worksheet)
    ws.Rows("12:15").RowHeight = 22
    ws.Rows("19:22").RowHeight = 22
End Sub

Private Sub RemoveDashboardButtons(ByVal ws As Worksheet)
    Dim btn As Button
    Dim shp As Shape

    On Error Resume Next
    For Each btn In ws.Buttons
        btn.Delete
    Next btn
    For Each shp In ws.Shapes
        If shp.Type = msoFormControl Then
            shp.Delete
        End If
    Next shp
    On Error GoTo 0
End Sub

Private Sub WriteSectionHeader(ByVal ws As Worksheet, ByVal cellAddress As String, ByVal title As String)
    With ws.Range(cellAddress)
        .Value = title
        .Font.Bold = True
        .Font.Size = 12
        .Font.Color = COLOR_HEADER_BLUE
    End With
End Sub

Private Sub WriteLabel(ByVal ws As Worksheet, ByVal labelCell As String, ByVal caption As String)
    ws.Range(labelCell).Value = caption
    ws.Range(labelCell).Font.Bold = True
End Sub

Private Sub WriteLabelValue(ByVal ws As Worksheet, ByVal labelCell As String, _
                            ByVal caption As String, ByVal formulaText As String)
    WriteLabel ws, labelCell, caption
    ws.Range("C" & Mid$(labelCell, 2)).Formula = formulaText
End Sub
