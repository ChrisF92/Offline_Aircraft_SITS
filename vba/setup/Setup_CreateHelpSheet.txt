Option Explicit

' ========================================================================
' ONE-TIME SETUP — Help / user guide sheet
' ========================================================================
' Module Name property: modSetupHelp
'
' Setup_CreateHelpSheet — builds (or rebuilds) the Help sheet that explains
' how to use each part of the workbook. Safe to re-run; content is replaced.
'
' ViewHelp — activates the Help sheet (Dashboard "View Help" button).
' ========================================================================

Public Sub Setup_CreateHelpSheet(Optional ByVal silent As Boolean = False)
    Dim ws As Worksheet
    Dim errNum As Long
    Dim errDesc As String

    If Not silent Then
        If MsgBox("Build the Help sheet with operator guidance for " & _
                  APP_NAME & " v" & APP_VERSION & "?" & vbCrLf & vbCrLf & _
                  "Existing Help sheet contents will be replaced.", _
                  vbQuestion + vbYesNo, "Setup — Help Sheet") <> vbYes Then Exit Sub
    End If

    On Error GoTo ErrorHandler
    SetExcelState "disable"
    Application.DisplayAlerts = False
    SuspendWorkbookProtection

    EnsureHelpSheetExists
    Set ws = GetWorksheet(SHEET_HELP)
    If ws Is Nothing Then
        Err.Raise vbObjectError + 5100, , "Help sheet could not be created."
    End If

    BuildHelpSheetContent ws

    On Error Resume Next
    ThisWorkbook.Worksheets(SHEET_HELP).Tab.Color = COLOR_TAB_VISIBLE
    On Error GoTo ErrorHandler

    If Not silent Then
        ' Standalone upgrade: place Help after Audit Log without a full Phase 0 reorder.
        PlaceHelpSheetAmongVisibleTabs
    End If

    ResumeWorkbookProtection
    Application.DisplayAlerts = True
    SetExcelState "enable"

    If Not silent Then
        MsgBox "Help sheet built successfully." & vbCrLf & vbCrLf & _
               "Open it from the Dashboard (View Help) or the Help tab.", _
               vbInformation, "Setup — Help Sheet"
        ViewHelp
    End If
    Exit Sub

ErrorHandler:
    errNum = Err.Number
    errDesc = Err.Description
    Application.DisplayAlerts = True
    SetExcelState "enable"
    If Not silent Then
        ForceResumeWorkbookProtection
        MsgBox "Help sheet setup failed." & vbCrLf & vbCrLf & _
               "Error " & errNum & ": " & errDesc, _
               vbCritical, "Setup — Help Sheet"
    Else
        Err.Raise errNum, "Setup_CreateHelpSheet", errDesc
    End If
End Sub

Public Sub ViewHelp()
    Dim ws As Worksheet

    Set ws = GetWorksheet(SHEET_HELP, showError:=False)
    If ws Is Nothing Then
        MsgBox "Help sheet not found." & vbCrLf & vbCrLf & _
               "Run Setup_CreateHelpSheet (Dev Tools: Rebuild Help Sheet) once.", _
               vbExclamation, APP_NAME
        Exit Sub
    End If

    ws.Visible = xlSheetVisible
    ws.Activate
End Sub

Private Sub EnsureHelpSheetExists()
    Dim ws As Worksheet

    If SheetExists(SHEET_HELP) Then Exit Sub

    UnprotectWorkbookStructure
    Set ws = ThisWorkbook.Worksheets.Add(After:=ThisWorkbook.Worksheets(ThisWorkbook.Worksheets.Count))
    ws.Name = SHEET_HELP
    ws.Visible = xlSheetVisible
End Sub

Private Sub PlaceHelpSheetAmongVisibleTabs()
    Dim wsHelp As Worksheet
    Dim wsAudit As Worksheet

    On Error Resume Next
    Set wsHelp = ThisWorkbook.Worksheets(SHEET_HELP)
    Set wsAudit = ThisWorkbook.Worksheets(SHEET_AUDIT)
    If Not wsHelp Is Nothing And Not wsAudit Is Nothing Then
        UnprotectWorkbookStructure
        If wsHelp.Index <> wsAudit.Index + 1 Then
            wsHelp.Move After:=wsAudit
        End If
    End If
    On Error GoTo 0
End Sub

Private Sub WriteHelpSection(ByVal ws As Worksheet, ByRef rowNum As Long, ByVal title As String)
    rowNum = rowNum + 1
    With ws.Range("B" & rowNum)
        .Value = title
        .Font.Bold = True
        .Font.Size = 12
        .Font.Color = COLOR_HEADER_BLUE
    End With
    ws.Range("C" & rowNum).ClearContents
    rowNum = rowNum + 1
End Sub

Private Sub WriteHelpRow(ByVal ws As Worksheet, ByRef rowNum As Long, _
                         ByVal topic As String, ByVal detail As String)
    ws.Range("B" & rowNum).Value = topic
    ws.Range("B" & rowNum).Font.Bold = True
    ws.Range("C" & rowNum).Value = detail
    rowNum = rowNum + 1
End Sub

Private Sub AutoFitHelpRows(ByVal ws As Worksheet, ByVal firstRow As Long, ByVal lastRow As Long)
    Dim r As Long
    Dim detailLen As Long
    Dim lineEstimate As Long

    For r = firstRow To lastRow
        If Len(Trim$(CStr(ws.Range("C" & r).Value))) = 0 Then
            ws.Rows(r).RowHeight = 18
        Else
            detailLen = Len(CStr(ws.Range("C" & r).Value))
            lineEstimate = Application.WorksheetFunction.Max(2, Int(detailLen / 95) + 1)
            ws.Rows(r).RowHeight = Application.WorksheetFunction.Min(90, 15 + (lineEstimate * 13))
        End If
    Next r
End Sub
