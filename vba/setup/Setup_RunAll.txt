Option Explicit

' ========================================================================
' ONE-TIME SETUP — Build master workbook skeleton (no aircraft data)
' Build marker: Phase-4-work-packages-v1
' ========================================================================

Private mSetupStep As String

Public Sub Setup_RunAll()
    Dim response As VbMsgBoxResult

    response = MsgBox("This builds the " & APP_NAME & " v" & APP_VERSION & _
                      " master template skeleton." & vbCrLf & vbCrLf & _
                      "Run this once on a blank workbook, then save as your master copy." & vbCrLf & _
                      "For each aircraft: Save As a new file, then run Setup Wizard." & vbCrLf & vbCrLf & _
                      "WARNING: Existing sheets and data in this workbook will be removed." & vbCrLf & vbCrLf & _
                      "Save a backup first. Continue?", _
                      vbExclamation + vbYesNo, "Setup — Run All")

    If response <> vbYes Then Exit Sub

    On Error GoTo ErrorHandler
    SetExcelState "disable"
    Application.DisplayAlerts = False

    mSetupStep = "Building sheets"
    BuildSheetsSilent

    mSetupStep = "Creating tables"
    Setup_CreateTables True

    mSetupStep = "Seeding configuration"
    Setup_CreateConfig True

    mSetupStep = "Creating named ranges"
    Setup_CreateNamedRanges True

    mSetupStep = "Formatting dashboard"
    Setup_CreateDashboard True

    mSetupStep = "Finalizing sheet layout"
    FinalizeSetupSheetLayout

    Application.DisplayAlerts = True
    SetExcelState "enable"

    mSetupStep = "Validating workbook"
    If ValidateWorkbookStructure() Then
        MsgBox "Master template setup completed successfully." & vbCrLf & vbCrLf & _
               "Next steps:" & vbCrLf & _
               "1. Save this workbook as your master copy (e.g. Wildcat_SITS_Master.xlsm)" & vbCrLf & _
               "2. For each aircraft: File > Save As > <TAIL>_SITS_v2.xlsm" & vbCrLf & _
               "3. On the aircraft copy, run Setup Wizard from the Dashboard", _
               vbInformation, "Setup Complete"
        WriteMasterTemplateAuditEntry
    End If

    Exit Sub

ErrorHandler:
    Application.DisplayAlerts = True
    SetExcelState "enable"
    MsgBox "Setup failed during: " & mSetupStep & vbCrLf & vbCrLf & _
           "Error " & Err.Number & ": " & Err.Description, _
           vbCritical, "Setup — Run All"
End Sub

Private Sub BuildSheetsSilent()
    If ThisWorkbook.Worksheets.Count = 1 Then
        ReuseSheetAs SHEET_DASHBOARD
    Else
        PrepareWorkbookForSetup
        ReuseSheetAs SHEET_DASHBOARD
    End If

    AddSetupSheet SHEET_SITS
    AddSetupSheet SHEET_USAGE
    AddSetupSheet SHEET_AUDIT
    AddSetupSheet SHEET_CONFIG
    AddSetupSheet SHEET_DATA_SITS
    AddSetupSheet SHEET_DATA_USAGE_BASELINE
    AddSetupSheet SHEET_DATA_USAGE_LOG
    AddSetupSheet SHEET_DATA_USAGE_CURRENT
    AddSetupSheet SHEET_DATA_WP
    AddSetupSheet SHEET_DATA_WP_TASKS
    AddSetupSheet SHEET_WP_STATE

    OrderSetupSheets
End Sub

Private Sub WriteMasterTemplateAuditEntry()
    On Error Resume Next

    Dim ws As Worksheet
    Dim tbl As ListObject
    Dim row As ListRow
    Dim userName As String

    Set ws = ThisWorkbook.Worksheets(SHEET_AUDIT)
    Set tbl = ws.ListObjects("tblAudit")
    If ws Is Nothing Or tbl Is Nothing Then Exit Sub

    userName = Environ$("USERNAME")
    If Len(userName) = 0 Then userName = "Unknown"

    Set row = tbl.ListRows.Add
    row.Range.Cells(1, 1).Value = Now
    row.Range.Cells(1, 2).Value = userName
    row.Range.Cells(1, 3).Value = "SETUP_MASTER"
    row.Range.Cells(1, 4).Value = ""
    row.Range.Cells(1, 5).Value = "Master template skeleton created"
    row.Range.Cells(1, 6).Value = APP_VERSION

    On Error GoTo 0
End Sub
