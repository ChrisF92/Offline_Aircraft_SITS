Option Explicit

' ========================================================================
' ONE-TIME SETUP — Build master workbook skeleton (no aircraft data)
' Build marker: Phase-5-extensions-deferrals-v1
'
' Setup_RunAll      — full scaffold rebuild (sheets, tables, formatting)
' ResetWorkbookToMaster — same empty-master data state, keep all formatting
' ========================================================================

Private mSetupStep As String

Public Sub Setup_RunAll()
    Dim response As VbMsgBoxResult

    response = MsgBox("This builds the " & APP_NAME & " v" & APP_VERSION & _
                      " master template skeleton." & vbCrLf & vbCrLf & _
                      "Run this once on a blank workbook, then save as your master copy." & vbCrLf & _
                      "For each aircraft: Save As a new file, then run Setup Wizard." & vbCrLf & vbCrLf & _
                      "WARNING: Existing sheets and data in this workbook will be removed." & vbCrLf & vbCrLf & _
                      "Save a backup first. Continue?", _
                      vbExclamation + vbYesNo, "Setup — Run All")

    If response <> vbYes Then Exit Sub

    On Error GoTo ErrorHandler
    SetExcelState "disable"
    Application.DisplayAlerts = False

    ' Structure/sheet protection blocks create/delete/hide during rebuild.
    RemoveWorkbookProtection

    mSetupStep = "Building sheets"
    BuildSheetsSilent

    mSetupStep = "Creating tables"
    Setup_CreateTables True

    mSetupStep = "Seeding configuration"
    Setup_CreateConfig True

    mSetupStep = "Creating named ranges"
    Setup_CreateNamedRanges True

    mSetupStep = "Formatting dashboard"
    Setup_CreateDashboard True

    mSetupStep = "Building Help sheet"
    Setup_CreateHelpSheet True

    mSetupStep = "Finalizing sheet layout"
    FinalizeSetupSheetLayout

    Application.DisplayAlerts = True
    SetExcelState "enable"

    mSetupStep = "Validating workbook"
    If ValidateWorkbookStructure() Then
        MsgBox "Master template setup completed successfully." & vbCrLf & vbCrLf & _
               "Next steps:" & vbCrLf & _
               "1. Save this workbook as your master copy (e.g. Wildcat_SITS_Master.xlsm)" & vbCrLf & _
               "2. For each aircraft: File > Save As > <TAIL>_SITS_v2.xlsm" & vbCrLf & _
               "3. On the aircraft copy, run Setup Wizard from the Dashboard", _
               vbInformation, "Setup Complete"
        WriteMasterTemplateAuditEntry
    End If

    Exit Sub

ErrorHandler:
    Application.DisplayAlerts = True
    SetExcelState "enable"
    MsgBox "Setup failed during: " & mSetupStep & vbCrLf & vbCrLf & _
           "Error " & Err.Number & ": " & Err.Description, _
           vbCritical, "Setup — Run All"
End Sub

' Reset aircraft data / config back to an empty master template without
' rebuilding sheets, column widths, button placement, or other formatting.
Public Sub ResetWorkbookToMaster()
    Dim response As VbMsgBoxResult
    Dim showDevTools As Boolean

    response = MsgBox("Reset this workbook to an empty master template?" & vbCrLf & vbCrLf & _
                      "Clears SITS, usage, work packages, sortie log, and audit data," & vbCrLf & _
                      "and restores master configuration (no tail / setup incomplete)." & vbCrLf & vbCrLf & _
                      "Column widths, colours, and button placement are preserved." & vbCrLf & _
                      "(Use Phase 0 Setup if you need a full formatting rebuild.)" & vbCrLf & vbCrLf & _
                      "Save a backup first. Continue?", _
                      vbExclamation + vbYesNo + vbDefaultButton2, "Reset to Master")

    If response <> vbYes Then Exit Sub

    On Error GoTo ErrorHandler

    If Not ValidateWorkbookStructure() Then Exit Sub

    showDevTools = AreDevToolsEnabled()

    SetExcelState "disable"
    Application.DisplayAlerts = False
    SuspendWorkbookProtection

    mSetupStep = "Clearing operational data"
    ClearOperationalTableData

    mSetupStep = "Resetting configuration"
    Setup_CreateConfig True
    ' Keep Dev Tools visible if the operator launched this from that section.
    SetConfigValue CFG_SHOW_DEV_TOOLS, showDevTools

    mSetupStep = "Refreshing views"
    ApplyWorkbookDataFormats
    RefreshUsageView
    RefreshSITSView
    EnsureDashboardActionButtons
    RefreshDashboardStatusCounts

    LogAction "RESET_MASTER", "", "Data cleared to master template; formatting preserved"
    ResumeWorkbookProtection
    Application.DisplayAlerts = True
    SetExcelState "enable"

    MsgBox "Workbook reset to empty master template." & vbCrLf & vbCrLf & _
           "Formatting was left unchanged." & vbCrLf & vbCrLf & _
           "Next steps:" & vbCrLf & _
           "1. Save this workbook as your master copy (or keep polishing layout)" & vbCrLf & _
           "2. For each aircraft: File > Save As > <TAIL>_SITS_v2.xlsm" & vbCrLf & _
           "3. On the aircraft copy, run Setup Wizard from the Dashboard", _
           vbInformation, "Reset to Master"
    Exit Sub

ErrorHandler:
    Dim errNum As Long
    Dim errDesc As String
    errNum = Err.Number
    errDesc = Err.Description
    On Error Resume Next
    ForceResumeWorkbookProtection
    Application.DisplayAlerts = True
    SetExcelState "enable"
    On Error GoTo 0
    MsgBox "Reset failed during: " & mSetupStep & vbCrLf & vbCrLf & _
           "Error " & errNum & ": " & errDesc, _
           vbCritical, "Reset to Master"
End Sub

Private Sub ClearOperationalTableData()
    ClearNamedTableData SHEET_DATA_SITS, TBL_SITS
    ClearNamedTableData SHEET_DATA_USAGE_BASELINE, TBL_USAGE_BASELINE
    ClearNamedTableData SHEET_DATA_USAGE_CURRENT, TBL_USAGE_CURRENT
    ClearNamedTableData SHEET_DATA_USAGE_LOG, TBL_USAGE_LOG
    ClearNamedTableData SHEET_DATA_WP, TBL_WORK_PACKAGES
    ClearNamedTableData SHEET_DATA_WP_TASKS, TBL_WP_TASKS
    ClearNamedTableData SHEET_CONFIG, TBL_USAGE_SNAPSHOT
    ClearNamedTableData SHEET_AUDIT, "tblAudit"
    ClearWpStateScratchData
End Sub

Private Sub ClearNamedTableData(ByVal sheetName As String, ByVal tableName As String)
    Dim ws As Worksheet
    Dim tbl As ListObject

    Set ws = GetWorksheet(sheetName, showError:=False)
    If ws Is Nothing Then Exit Sub

    Set tbl = GetTable(ws, tableName, showError:=False)
    ClearTableData tbl
End Sub

Private Sub ClearWpStateScratchData()
    Dim ws As Worksheet
    Dim lastRow As Long

    Set ws = GetWorksheet(SHEET_WP_STATE, showError:=False)
    If ws Is Nothing Then Exit Sub

    On Error Resume Next
    lastRow = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row
    If lastRow < 2 Then
        If ws.UsedRange.Rows.Count + ws.UsedRange.Row - 1 > 1 Then
            lastRow = ws.UsedRange.Rows.Count + ws.UsedRange.Row - 1
        End If
    End If
    If lastRow >= 2 Then
        ' Clear values only — keep header styling from Phase 0 / master polish.
        ws.Range(ws.Cells(2, 1), ws.Cells(lastRow, 3)).ClearContents
    End If
    On Error GoTo 0
End Sub

Private Sub BuildSheetsSilent()
    ' Prior Phase 0 runs leave data sheets hidden. Unhide before teardown so
    ' Excel will allow deleting every sheet except the last one.
    UnhideAllWorksheets

    If ThisWorkbook.Worksheets.Count = 1 Then
        ReuseSheetAs SHEET_DASHBOARD
    Else
        PrepareWorkbookForSetup
        ReuseSheetAs SHEET_DASHBOARD
    End If

    AddSetupSheet SHEET_SITS
    AddSetupSheet SHEET_USAGE
    AddSetupSheet SHEET_AUDIT
    AddSetupSheet SHEET_HELP
    AddSetupSheet SHEET_CONFIG
    AddSetupSheet SHEET_DATA_SITS
    AddSetupSheet SHEET_DATA_USAGE_BASELINE
    AddSetupSheet SHEET_DATA_USAGE_LOG
    AddSetupSheet SHEET_DATA_USAGE_CURRENT
    AddSetupSheet SHEET_DATA_WP
    AddSetupSheet SHEET_DATA_WP_TASKS
    AddSetupSheet SHEET_WP_STATE

    OrderSetupSheets
End Sub

Private Sub WriteMasterTemplateAuditEntry()
    On Error Resume Next

    Dim ws As Worksheet
    Dim tbl As ListObject
    Dim row As ListRow
    Dim userName As String

    Set ws = ThisWorkbook.Worksheets(SHEET_AUDIT)
    Set tbl = ws.ListObjects("tblAudit")
    If ws Is Nothing Or tbl Is Nothing Then Exit Sub

    userName = Environ$("USERNAME")
    If Len(userName) = 0 Then userName = "Unknown"

    Set row = tbl.ListRows.Add
    row.Range.Cells(1, 1).Value = Now
    row.Range.Cells(1, 2).Value = userName
    row.Range.Cells(1, 3).Value = "SETUP_MASTER"
    row.Range.Cells(1, 4).Value = ""
    row.Range.Cells(1, 5).Value = "Master template skeleton created"
    row.Range.Cells(1, 6).Value = APP_VERSION

    On Error GoTo 0
End Sub
