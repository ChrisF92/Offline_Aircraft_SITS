Option Explicit

' ========================================================================
' NAMED RANGES — Dashboard status count names
' Module Name property: modSetupNames
'
' Called by Rebuild Dashboard. Safe to re-run as a repair.
' ========================================================================

Public Sub Setup_CreateNamedRanges(Optional ByVal silent As Boolean = False)
    If Not silent Then
        If MsgBox("Create / refresh dashboard named ranges?", _
                  vbQuestion + vbYesNo, "Named Ranges") <> vbYes Then Exit Sub
    End If

    SetExcelState "disable"
    On Error GoTo ErrorHandler

    DeleteNamedRangeIfExists NAMED_ACTIVE_WP
    DeleteNamedRangeIfExists NAMED_EXTENSION_SNOW
    DeleteNamedRangeIfExists NAMED_OPEN_WP_COUNT
    DeleteNamedRangeIfExists NAMED_MANUAL_UPDATE_COUNT
    DeleteNamedRangeIfExists NAMED_WARNING_COUNT
    DeleteNamedRangeIfExists NAMED_OVERDUE_COUNT

    ThisWorkbook.Names.Add Name:=NAMED_OPEN_WP_COUNT, RefersTo:="=" & SHEET_DASHBOARD & "!$C$8"
    ThisWorkbook.Names.Add Name:=NAMED_MANUAL_UPDATE_COUNT, RefersTo:="=" & SHEET_DASHBOARD & "!$C$9"
    ThisWorkbook.Names.Add Name:=NAMED_WARNING_COUNT, RefersTo:="=" & SHEET_DASHBOARD & "!$C$10"
    ThisWorkbook.Names.Add Name:=NAMED_OVERDUE_COUNT, RefersTo:="=" & SHEET_DASHBOARD & "!$C$11"

    SetExcelState "enable"
    If Not silent Then MsgBox "Named ranges created.", vbInformation, "Named Ranges"
    Exit Sub

ErrorHandler:
    SetExcelState "enable"
    MsgBox "Error creating named ranges: " & Err.Description, vbCritical, "Named Ranges"
End Sub

Private Sub DeleteNamedRangeIfExists(ByVal rangeName As String)
    On Error Resume Next
    ThisWorkbook.Names(rangeName).Delete
    On Error GoTo 0
End Sub
