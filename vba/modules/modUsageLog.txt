Private Const USAGE_VIEW_START_ROW As Long = 5

Public Sub ShowLogSortieForm()
    If Not ValidateSetupReady() Then Exit Sub
    frmLogSortie.Show
End Sub

' Dashboard-friendly alias.
Public Sub RunLogSortie()
    ShowLogSortieForm
End Sub

' Resolve the baseline/current counter used on the Log Sortie form for one
' interval code. Prefers an Airframe-sourced row when duplicates exist (e.g. OS).
Public Function ResolveSortieUsageCounter(ByVal intervalCode As String, _
                                          ByRef taskCode As String, _
                                          ByRef formatType As String, _
                                          ByRef currentValue As Variant, _
                                          ByRef isAvailable As Boolean) As Boolean
    Dim wsBaseline As Worksheet
    Dim tblBaseline As ListObject
    Dim row As ListRow
    Dim code As String
    Dim sourceName As String
    Dim candidateTask As String
    Dim candidateFormat As String
    Dim airframeTask As String
    Dim airframeFormat As String
    Dim foundAny As Boolean
    Dim foundAirframe As Boolean

    ResolveSortieUsageCounter = False
    taskCode = ""
    formatType = ""
    currentValue = Empty
    isAvailable = False

    code = UCase$(Trim$(intervalCode))
    If Len(code) = 0 Then Exit Function

    Set wsBaseline = GetWorksheet(SHEET_DATA_USAGE_BASELINE, showError:=False)
    If wsBaseline Is Nothing Then Exit Function

    Set tblBaseline = GetTable(wsBaseline, TBL_USAGE_BASELINE, showError:=False)
    If tblBaseline Is Nothing Or tblBaseline.DataBodyRange Is Nothing Then Exit Function

    For Each row In tblBaseline.ListRows
        If StrComp(UCase$(Trim$(CStr(row.Range.Cells(1, 1).Value))), code, vbBinaryCompare) <> 0 Then
            GoTo NextBaselineRow
        End If

        candidateTask = Trim$(CStr(row.Range.Cells(1, 2).Value))
        candidateFormat = ResolveUsageFormatType(code, candidateTask, _
            Trim$(CStr(row.Range.Cells(1, 5).Value)), _
            Trim$(CStr(row.Range.Cells(1, 3).Value)))
        sourceName = Trim$(CStr(row.Range.Cells(1, 6).Value))

        If Not foundAny Then
            taskCode = candidateTask
            formatType = candidateFormat
            foundAny = True
        End If

        If StrComp(sourceName, "Airframe", vbTextCompare) = 0 Then
            airframeTask = candidateTask
            airframeFormat = candidateFormat
            foundAirframe = True
            Exit For
        End If

NextBaselineRow:
    Next row

    If Not foundAny Then
        ' Known sortie counter not yet imported — still show on the form.
        formatType = StoredUsageFormatLabel(code)
        currentValue = 0
        ResolveSortieUsageCounter = True
        isAvailable = False
        Exit Function
    End If

    If foundAirframe Then
        taskCode = airframeTask
        formatType = airframeFormat
    End If

    currentValue = UsageValueAsDayFraction(GetCurrentUsageValue(code, taskCode), formatType)
    isAvailable = True
    ResolveSortieUsageCounter = True
End Function

Public Function CommitSortie(ByVal sortieDate As Date, _
                             ByVal sortieRef As String, _
                             ByVal pending As Object, _
                             Optional ByVal showDialogs As Boolean = True) As Boolean
    Const PROC_NAME As String = "CommitSortie"

    Dim mergeKey As Variant
    Dim parts As Variant
    Dim intervalCode As String
    Dim taskCode As String
    Dim incrementValue As Variant
    Dim formatType As String
    Dim failReason As String
    Dim loggedCount As Long
    Dim currentValue As Variant
    Dim newValue As Variant

    CommitSortie = False

    If Not ValidateSetupReady() Then Exit Function
    If pending Is Nothing Or pending.Count = 0 Then
        If showDialogs Then MsgBox "Enter at least one usage value before submitting.", vbExclamation, APP_NAME
        Exit Function
    End If

    If sortieDate > Date Then
        If showDialogs Then
            If MsgBox("Sortie date is in the future. Continue anyway?", _
                      vbQuestion + vbYesNo, APP_NAME) <> vbYes Then Exit Function
        Else
            Exit Function
        End If
    End If

    On Error GoTo ErrorHandler
    SetExcelState "disable"

    For Each mergeKey In pending.Keys
        parts = Split(CStr(mergeKey), "|")
        If UBound(parts) < 1 Then GoTo NextItem

        intervalCode = parts(0)
        taskCode = parts(1)
        incrementValue = pending(mergeKey)

        formatType = GetUsageFormatType(intervalCode, taskCode)
        If Len(formatType) = 0 Then
            failReason = "Usage counter not found in baseline: " & intervalCode & _
                         ". Import usage that includes this interval type, then try again."
            GoTo ValidationFailed
        End If
        formatType = ResolveUsageFormatType(intervalCode, taskCode, formatType)

        If Not ValidateUsageIncrement(incrementValue, formatType, failReason) Then GoTo ValidationFailed

        currentValue = UsageValueAsDayFraction(GetCurrentUsageValue(intervalCode, taskCode), formatType)
        newValue = AddUsageValues(currentValue, incrementValue, formatType)

        AppendUsageLogRow sortieDate, sortieRef, intervalCode, taskCode, incrementValue, formatType, ""
        SetCurrentUsageValue intervalCode, taskCode, newValue, formatType
        loggedCount = loggedCount + 1

NextItem:
    Next mergeKey

    ApplyUsageTableFormats
    RefreshUsageView
    RefreshSITS
    SetExcelState "enable"

    LogAction "SORTIE_LOGGED", "", _
              "Sortie " & Format$(sortieDate, FORMAT_DATE) & _
              IIf(Len(Trim$(sortieRef)) > 0, " (" & sortieRef & ")", "") & _
              "; increments=" & loggedCount

    CommitSortie = True

    If showDialogs Then
        MsgBox "Logged " & loggedCount & " usage increment(s) for sortie " & _
               Format$(sortieDate, FORMAT_DATE) & ".", vbInformation, APP_NAME
    End If
    Exit Function

ValidationFailed:
    SetExcelState "enable"
    If showDialogs Then MsgBox failReason, vbExclamation, APP_NAME
    Exit Function

ErrorHandler:
    SetExcelState "enable"
    LogError PROC_NAME, Err.Number, Err.Description
    If showDialogs Then
        MsgBox "Sortie log error: " & Err.Number & " - " & Err.Description, vbCritical, APP_NAME
    End If
End Function

Public Sub RefreshUsageView()
    Dim ws As Worksheet
    Dim wsBaseline As Worksheet
    Dim tblBaseline As ListObject
    Dim row As ListRow
    Dim outputRow As Long
    Dim lastRow As Long
    Dim intervalCode As String
    Dim taskCode As String
    Dim currentValue As Variant
    Dim formatType As String

    Set ws = GetWorksheet(SHEET_USAGE, showError:=False)
    Set wsBaseline = GetWorksheet(SHEET_DATA_USAGE_BASELINE, showError:=False)
    If ws Is Nothing Or wsBaseline Is Nothing Then Exit Sub

    Set tblBaseline = GetTable(wsBaseline, TBL_USAGE_BASELINE, showError:=False)
    If tblBaseline Is Nothing Then Exit Sub

    SetExcelState "disable"

    ' Preserve column widths / shell chrome — clear data area only.
    EnsureUsageViewShellExists ws
    ws.Range("B2").Value = GetTailNumber()

    lastRow = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row
    If lastRow < USAGE_VIEW_START_ROW Then lastRow = USAGE_VIEW_START_ROW
    On Error Resume Next
    If ws.UsedRange.Rows.Count + ws.UsedRange.Row - 1 > lastRow Then
        lastRow = ws.UsedRange.Rows.Count + ws.UsedRange.Row - 1
    End If
    On Error GoTo 0
    ws.Range(ws.Cells(USAGE_VIEW_START_ROW, 1), ws.Cells(lastRow, 3)).Clear

    ws.Range("A" & USAGE_VIEW_START_ROW).Value = HDR_BASE_INTERVAL_CODE
    ws.Range("B" & USAGE_VIEW_START_ROW).Value = HDR_CUR_CURRENT_VALUE
    ws.Range("C" & USAGE_VIEW_START_ROW).Value = HDR_BASE_FORMAT

    With ws.Range("A" & USAGE_VIEW_START_ROW & ":C" & USAGE_VIEW_START_ROW)
        .Font.Bold = True
        .Font.Color = vbWhite
        .Interior.Color = COLOR_HEADER_BLUE
    End With

    outputRow = USAGE_VIEW_START_ROW + 1

    If Not tblBaseline.DataBodyRange Is Nothing Then
        For Each row In tblBaseline.ListRows
            intervalCode = Trim$(CStr(row.Range.Cells(1, 1).Value))
            taskCode = Trim$(CStr(row.Range.Cells(1, 2).Value))
            formatType = StoredUsageFormatLabel(intervalCode)
            currentValue = GetCurrentUsageValue(intervalCode, taskCode)

            ws.Cells(outputRow, 1).Value = intervalCode
            ws.Cells(outputRow, 2).Value = currentValue
            ws.Cells(outputRow, 2).NumberFormat = UsageFormatPatternForInterval(intervalCode, currentValue)
            ws.Cells(outputRow, 3).Value = formatType

            outputRow = outputRow + 1
        Next row
    End If

    ' Do not AutoFit — Master / developer column widths must survive refresh.
    SetExcelState "enable"
End Sub

Private Sub EnsureUsageViewShellExists(ByVal ws As Worksheet)
    If Len(Trim$(CStr(ws.Range("A1").Value))) > 0 Then Exit Sub

    With ws.Range("A1")
        .Value = "Usage — Current Totals"
        .Font.Size = 14
        .Font.Bold = True
        .Font.Color = COLOR_HEADER_BLUE
    End With

    ws.Range("A2").Value = "Tail:"
    ws.Range("B2").Value = GetTailNumber()
    ws.Range("A3").Value = "Log sorties with RunLogSortie."
End Sub

Public Function ParseUsageIncrement(ByVal rawIncrement As Variant, _
                                    ByVal formatType As String, _
                                    ByRef failReason As String) As Variant
    Dim parsedTime As Variant

    If IsEmpty(rawIncrement) Or Len(Trim$(CStr(rawIncrement))) = 0 Then
        failReason = "Increment cannot be blank."
        Exit Function
    End If

    Select Case UCase$(Trim$(formatType))
        Case FORMAT_TYPE_HOURS
            parsedTime = ParseTimeUsageValue(rawIncrement, CStr(rawIncrement))
            If IsEmpty(parsedTime) Then
                failReason = "Enter flight/usage time as h:mm (e.g. 1:30)."
                Exit Function
            End If
            If CDbl(parsedTime) <= 0 Then
                failReason = "Increment must be greater than zero."
                Exit Function
            End If
            ParseUsageIncrement = parsedTime

        Case FORMAT_TYPE_MINUTES
            parsedTime = ParseMinutesSecondsValue(rawIncrement, CStr(rawIncrement))
            If IsEmpty(parsedTime) Then
                failReason = "Enter duration as m:ss (e.g. 1:30)."
                Exit Function
            End If
            If CDbl(parsedTime) <= 0 Then
                failReason = "Increment must be greater than zero."
                Exit Function
            End If
            ParseUsageIncrement = parsedTime

        Case Else
            If Not IsNumeric(rawIncrement) Then
                failReason = "Enter a numeric increment."
                Exit Function
            End If
            If CDbl(rawIncrement) <= 0 Then
                failReason = "Increment must be greater than zero."
                Exit Function
            End If
            ParseUsageIncrement = CDbl(rawIncrement)
    End Select
End Function

Public Function ValidateUsageIncrement(ByVal incrementValue As Variant, _
                                       ByVal formatType As String, _
                                       ByRef failReason As String) As Boolean
    ValidateUsageIncrement = Not IsEmpty(ParseUsageIncrement(incrementValue, formatType, failReason))
End Function

Public Function AddUsageValues(ByVal currentValue As Variant, _
                               ByVal incrementValue As Variant, _
                               ByVal formatType As String) As Variant
    AddUsageValues = CDbl(NzUsageNumber(currentValue)) + CDbl(incrementValue)
End Function

Public Function GetCurrentUsageValue(ByVal intervalCode As String, _
                                     ByVal taskCode As String) As Variant
    Dim ws As Worksheet
    Dim tbl As ListObject
    Dim row As ListRow

    Set ws = GetWorksheet(SHEET_DATA_USAGE_CURRENT, showError:=False)
    If ws Is Nothing Then Exit Function

    Set tbl = GetTable(ws, TBL_USAGE_CURRENT, showError:=False)
    If tbl Is Nothing Or tbl.DataBodyRange Is Nothing Then Exit Function

    For Each row In tbl.ListRows
        If StrComp(CStr(row.Range.Cells(1, 1).Value), intervalCode, vbTextCompare) = 0 And _
           StrComp(CStr(row.Range.Cells(1, 2).Value), taskCode, vbTextCompare) = 0 Then
            GetCurrentUsageValue = row.Range.Cells(1, 3).Value
            Exit Function
        End If
    Next row
End Function

Public Function GetUsageFormatType(ByVal intervalCode As String, _
                                   ByVal taskCode As String) As String
    Dim ws As Worksheet
    Dim tbl As ListObject
    Dim row As ListRow

    Set ws = GetWorksheet(SHEET_DATA_USAGE_BASELINE, showError:=False)
    If ws Is Nothing Then Exit Function

    Set tbl = GetTable(ws, TBL_USAGE_BASELINE, showError:=False)
    If tbl Is Nothing Or tbl.DataBodyRange Is Nothing Then Exit Function

    For Each row In tbl.ListRows
        If StrComp(CStr(row.Range.Cells(1, 1).Value), intervalCode, vbTextCompare) = 0 And _
           StrComp(CStr(row.Range.Cells(1, 2).Value), taskCode, vbTextCompare) = 0 Then
            GetUsageFormatType = ResolveUsageFormatType(intervalCode, taskCode, _
                Trim$(CStr(row.Range.Cells(1, 5).Value)), _
                Trim$(CStr(row.Range.Cells(1, 3).Value)))
            Exit Function
        End If
    Next row
End Function

Private Sub AppendUsageLogRow(ByVal sortieDate As Date, _
                              ByVal sortieRef As String, _
                              ByVal intervalCode As String, _
                              ByVal taskCode As String, _
                              ByVal incrementValue As Variant, _
                              ByVal formatType As String, _
                              ByVal correctedFrom As String)
    Dim ws As Worksheet
    Dim tbl As ListObject
    Dim row As ListRow
    Dim userName As String

    Set ws = GetWorksheet(SHEET_DATA_USAGE_LOG)
    Set tbl = GetTable(ws, TBL_USAGE_LOG)

    userName = Environ$("USERNAME")
    If Len(userName) = 0 Then userName = "Unknown"

    Set row = tbl.ListRows.Add
    With row.Range
        .Cells(1, 1).Value = sortieDate
        .Cells(1, 1).NumberFormat = FORMAT_DATE
        .Cells(1, 2).Value = sortieRef
        .Cells(1, 3).Value = intervalCode
        .Cells(1, 4).Value = taskCode
        WriteFormattedUsageValue .Cells(1, 5), incrementValue, formatType
        .Cells(1, 6).Value = userName
        .Cells(1, 7).Value = Now
        .Cells(1, 7).NumberFormat = FORMAT_DATETIME
        .Cells(1, 8).Value = correctedFrom
    End With
End Sub

Private Sub SetCurrentUsageValue(ByVal intervalCode As String, _
                                 ByVal taskCode As String, _
                                 ByVal newValue As Variant, _
                                 ByVal formatType As String)
    Dim ws As Worksheet
    Dim tbl As ListObject
    Dim row As ListRow

    Set ws = GetWorksheet(SHEET_DATA_USAGE_CURRENT)
    Set tbl = GetTable(ws, TBL_USAGE_CURRENT)

    For Each row In tbl.ListRows
        If StrComp(CStr(row.Range.Cells(1, 1).Value), intervalCode, vbTextCompare) = 0 And _
           StrComp(CStr(row.Range.Cells(1, 2).Value), taskCode, vbTextCompare) = 0 Then
            WriteUsageCurrentCell row.Range.Cells(1, 3), newValue, formatType
            Exit Sub
        End If
    Next row

    Set row = tbl.ListRows.Add
    row.Range.Cells(1, 1).Value = intervalCode
    row.Range.Cells(1, 2).Value = taskCode
    WriteUsageCurrentCell row.Range.Cells(1, 3), newValue, formatType
End Sub

Private Function NzUsageNumber(ByVal value As Variant) As Double
    If IsEmpty(value) Or Len(Trim$(CStr(value))) = 0 Then
        NzUsageNumber = 0#
    Else
        NzUsageNumber = CDbl(value)
    End If
End Function
