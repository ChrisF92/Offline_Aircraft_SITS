' Apply Horizon values from the UserForm (or any caller). Refreshes the view.
Public Function ApplyHorizonSettings(ByVal hoursText As String, _
                                     ByVal daysText As String, _
                                     ByVal numberText As String, _
                                     Optional ByVal showDialogs As Boolean = True) As Boolean
    Dim hoursValue As Double
    Dim daysValue As Long
    Dim numberValue As Long
    Dim failReason As String
    Dim ws As Worksheet

    ApplyHorizonSettings = False

    If Not ParseHorizonHours(hoursText, hoursValue, failReason) Then
        If showDialogs Then MsgBox failReason, vbExclamation, APP_NAME
        Exit Function
    End If
    If Not ParseHorizonWholeNumber(daysText, "Days", daysValue, failReason) Then
        If showDialogs Then MsgBox failReason, vbExclamation, APP_NAME
        Exit Function
    End If
    If Not ParseHorizonWholeNumber(numberText, "Number", numberValue, failReason) Then
        If showDialogs Then MsgBox failReason, vbExclamation, APP_NAME
        Exit Function
    End If

    SetConfigValue CFG_FORECAST_HOURS, hoursValue
    SetConfigValue CFG_FORECAST_DAYS, daysValue
    SetConfigValue CFG_FORECAST_NUMBER, numberValue

    ' Prevent SaveForecastSettingsFromView from writing stale B3/B4/B5 over config.
    Set ws = GetWorksheet(SHEET_SITS, showError:=False)
    If Not ws Is Nothing Then ClearSITSViewLegacyForecastCells ws

    LogAction "SITS_HORIZON_SET", "", _
              "Hours=" & CStr(hoursValue) & "; Days=" & CStr(daysValue) & _
              "; Number=" & CStr(numberValue)

    RefreshSITSViewOnly
    ApplyHorizonSettings = True
End Function

' Form-control checkbox OnAction — highlight / unhighlight the linked SITS view row.
' Resolve the control via Application.Caller (name may not stay SITS_SEL_*), then
' read Value / LinkedCell so we do not depend on a stale name or cell alone.
Public Sub SITSSelectorCheckboxChanged()
    Dim ws As Worksheet
    Dim cb As CheckBox
    Dim linkCell As Range
    Dim viewRow As Long
    Dim selected As Boolean
    Dim callerName As String
    Dim linkAddr As String
    Dim valueKnown As Boolean

    On Error Resume Next
    callerName = CStr(Application.Caller)
    On Error GoTo 0
    If Len(callerName) = 0 Then Exit Sub

    Set ws = Nothing
    On Error Resume Next
    Set ws = ActiveSheet
    If ws Is Nothing Then Set ws = GetWorksheet(SHEET_SITS, showError:=False)
    On Error GoTo 0
    If ws Is Nothing Then Exit Sub
    If StrComp(ws.Name, SHEET_SITS, vbTextCompare) <> 0 Then
        Set ws = GetWorksheet(SHEET_SITS, showError:=False)
        If ws Is Nothing Then Exit Sub
    End If

    Set cb = Nothing
    On Error Resume Next
    Set cb = ws.CheckBoxes(callerName)
    On Error GoTo 0
    If cb Is Nothing Then Exit Sub

    ' Prefer the control Value (xlOn=1 / xlOff=-4146); fall back to LinkedCell.
    selected = False
    valueKnown = False
    On Error Resume Next
    If cb.Value = 1 Then
        selected = True
        valueKnown = True
    ElseIf cb.Value = xlOff Then
        selected = False
        valueKnown = True
    End If
    linkAddr = CStr(cb.LinkedCell)
    On Error GoTo 0

    Set linkCell = Nothing
    If Len(linkAddr) > 0 Then
        On Error Resume Next
        ' LinkedCell is usually sheet-qualified ('SITS'!A8) — resolve via Application.Range.
        Set linkCell = Application.Range(linkAddr)
        If linkCell Is Nothing Then Set linkCell = ws.Range(linkAddr)
        On Error GoTo 0
        If Not linkCell Is Nothing Then
            If Not linkCell.Parent Is ws Then Set linkCell = Nothing
        End If
    End If

    If linkCell Is Nothing Then
        ' Fall back to row inferred from the control name or Top position.
        viewRow = 0
        If StrComp(Left$(cb.Name, 8), "SITS_SEL_", vbTextCompare) = 0 Then
            On Error Resume Next
            viewRow = CLng(Mid$(cb.Name, 9))
            On Error GoTo 0
        End If
        If viewRow < SITS_VIEW_DATA_START_ROW Then
            viewRow = EstimateSITSViewRowFromCheckboxTop(ws, cb.Top)
        End If
        If viewRow < SITS_VIEW_DATA_START_ROW Then Exit Sub
        Set linkCell = ws.Cells(viewRow, SITS_SELECTOR_COL)
    Else
        viewRow = linkCell.Row
    End If

    If viewRow < SITS_VIEW_DATA_START_ROW Then Exit Sub
    If StrComp(Trim$(CStr(linkCell.Value)), SITS_IN_WORK_TEXT, vbTextCompare) = 0 Then Exit Sub

    If Not valueKnown Then selected = ConvertToBoolean(linkCell.Value)

    ApplySITSViewRowSelectionHighlight ws, viewRow, selected
End Sub

' Called from ThisWorkbook.Workbook_SheetChange when Selector LinkedCells change.
Public Sub HandleSITSSelectorSheetChange(ByVal Sh As Object, ByVal Target As Range)
    Dim ws As Worksheet
    Dim intersectRange As Range
    Dim cell As Range
    Dim lastRow As Long
    Dim selected As Boolean

    If Sh Is Nothing Or Target Is Nothing Then Exit Sub
    If StrComp(Sh.Name, SHEET_SITS, vbTextCompare) <> 0 Then Exit Sub

    Set ws = Sh
    lastRow = ws.Cells(ws.Rows.Count, SITS_SELECTOR_COL).End(xlUp).Row
    If lastRow < SITS_VIEW_DATA_START_ROW Then Exit Sub

    Set intersectRange = Intersect(Target, _
        ws.Range(ws.Cells(SITS_VIEW_DATA_START_ROW, SITS_SELECTOR_COL), _
                 ws.Cells(lastRow, SITS_SELECTOR_COL)))
    If intersectRange Is Nothing Then Exit Sub

    For Each cell In intersectRange.Cells
        If StrComp(Trim$(CStr(cell.Value)), SITS_IN_WORK_TEXT, vbTextCompare) = 0 Then GoTo NextSelectorChange
        selected = ConvertToBoolean(cell.Value)
        ApplySITSViewRowSelectionHighlight ws, cell.Row, selected
NextSelectorChange:
    Next cell
End Sub

Private Function EstimateSITSViewRowFromCheckboxTop(ByVal ws As Worksheet, _
                                                    ByVal topPos As Double) As Long
    Dim firstTop As Double
    Dim rowHeightPx As Double
    Dim estimatedRow As Long
    Dim lastRow As Long

    EstimateSITSViewRowFromCheckboxTop = 0
    If ws Is Nothing Then Exit Function

    firstTop = ws.Cells(SITS_VIEW_DATA_START_ROW, SITS_SELECTOR_COL).Top
    rowHeightPx = ws.Cells(SITS_VIEW_DATA_START_ROW, SITS_SELECTOR_COL).Height
    If rowHeightPx <= 0# Then rowHeightPx = 15#

    lastRow = ws.Cells(ws.Rows.Count, SITS_SELECTOR_COL).End(xlUp).Row
    If lastRow < SITS_VIEW_DATA_START_ROW Then Exit Function

    estimatedRow = SITS_VIEW_DATA_START_ROW + CLng((topPos - firstTop) / rowHeightPx)
    If estimatedRow < SITS_VIEW_DATA_START_ROW Then estimatedRow = SITS_VIEW_DATA_START_ROW
    If estimatedRow > lastRow Then estimatedRow = lastRow
    EstimateSITSViewRowFromCheckboxTop = estimatedRow
End Function

    RefreshSITSView
End Sub

' Mark visible SITS rows as IN WORK (remove checkbox, show text) without rebuild.
Public Sub MarkSITSViewRowsInWork(ByVal rowIds As Object)
    Dim wsView As Worksheet
    Dim viewMap As Object
    Dim rowId As Variant
    Dim viewRow As Long

    If rowIds Is Nothing Or rowIds.Count = 0 Then Exit Sub
    Set wsView = GetWorksheet(SHEET_SITS, showError:=False)
    If wsView Is Nothing Then Exit Sub

    Set viewMap = BuildSITSViewRowIdMap(wsView)
    For Each rowId In rowIds.Keys
        If viewMap.Exists(CStr(rowId)) Then
            viewRow = CLng(viewMap(CStr(rowId)))
            SetSITSViewSelectorInWork wsView, viewRow
            ' Selection highlight must not linger after the row leaves the selector pool.
            ApplySITSViewRowSelectionHighlight wsView, viewRow, False
        End If
    Next rowId
End Sub

' Targeted post-close view update: refresh/remove only affected rows.
' closedRowIds = tasks reforecasted or flagged manual; deletedRowIds = PT2/PT3 removed.
Public Sub ApplySITSViewAfterWorkPackageClose(ByVal closedRowIds As Object, _
                                              ByVal deletedRowIds As Object)
    Dim wsView As Worksheet
    Dim wsData As Worksheet
    Dim tbl As ListObject
    Dim viewMap As Object
    Dim rowsToRemove As Collection
    Dim rowId As Variant
    Dim viewRow As Long
    Dim sitsRow As ListRow
    Dim intervalCode As String
    Dim lifeHoursFraction As Double
    Dim daysUntilDue As Long
    Dim forecastHours As Double
    Dim forecastDays As Long
    Dim forecastNumber As Long
    Dim warnHoursFraction As Double
    Dim warnDays As Long
    Dim colManualFlag As Long
    Dim colInterval As Long
    Dim includeRow As Boolean
    Dim i As Long
    Dim removeRow As Long
    Dim viewColumns As Collection
    Dim sitsIndex As Object
    Dim colMap As Object
    Dim colLRHours As Long
    Dim colLRNumber As Long
    Dim colDaysDue As Long

    Set wsView = GetWorksheet(SHEET_SITS, showError:=False)
    Set wsData = GetWorksheet(SHEET_DATA_SITS, showError:=False)
    If wsView Is Nothing Or wsData Is Nothing Then Exit Sub

    Set tbl = GetTable(wsData, TBL_SITS, showError:=False)
    If tbl Is Nothing Then Exit Sub

    Set viewMap = BuildSITSViewRowIdMap(wsView)
    Set sitsIndex = BuildSITSRowIdIndex(tbl)
    Set rowsToRemove = New Collection
    Set viewColumns = BuildSITSViewColumns()

    forecastHours = GetForecastHoursSetting()
    forecastDays = GetForecastDaysSetting()
    forecastNumber = GetForecastNumberSetting()
    warnHoursFraction = ParseWarnHoursSetting()
    warnDays = CLng(NzSITSNumber(GetConfigValue(CFG_WARN_DAYS, 7)))

    Set colMap = BuildTableColumnIndexMap(tbl)
    colManualFlag = LookupColumnIndex(colMap, HDR_NEEDS_MANUAL_UPDATE)
    colInterval = LookupColumnIndex(colMap, HDR_INTERVAL_TYPE_CODE)
    colLRHours = LookupColumnIndex(colMap, HDR_HOURS_LIFE_REMAINING)
    colLRNumber = LookupColumnIndex(colMap, HDR_NUMBER_LIFE_REMAINING)
    colDaysDue = LookupColumnIndex(colMap, HDR_DAYS_WHEN_DUE)

    ' PT2/PT3 deleted from data — remove from view if present.
    If Not deletedRowIds Is Nothing Then
        For Each rowId In deletedRowIds.Keys
            If viewMap.Exists(CStr(rowId)) Then
                rowsToRemove.Add CLng(viewMap(CStr(rowId)))
            End If
        Next rowId
    End If

    ' Closed / reforecasted tasks: update in place, or remove if outside forecast.
    If Not closedRowIds Is Nothing Then
        For Each rowId In closedRowIds.Keys
            If Not viewMap.Exists(CStr(rowId)) Then GoTo NextClosedView

            viewRow = CLng(viewMap(CStr(rowId)))
            Set sitsRow = ResolveSITSRowById(sitsIndex, tbl, CStr(rowId))
            If sitsRow Is Nothing Then
                rowsToRemove.Add viewRow
                GoTo NextClosedView
            End If

            intervalCode = UCase$(Trim$(CStr(sitsRow.Range.Cells(1, colInterval).Value)))
            includeRow = TaskPassesForecastFilter(sitsRow, tbl, intervalCode, _
                forecastHours, forecastDays, forecastNumber, lifeHoursFraction, daysUntilDue, _
                colLRHours, colLRNumber, colDaysDue)

            If Not includeRow Then
                rowsToRemove.Add viewRow
            Else
                RewriteSITSViewDataRow wsView, viewRow, sitsRow, tbl, intervalCode, viewColumns, colMap
                ApplySITSViewRowWarnings wsView, viewRow, SITS_VISIBLE_COLUMN_COUNT, _
                    intervalCode, lifeHoursFraction, daysUntilDue, _
                    warnHoursFraction, warnDays, sitsRow, colManualFlag
                SetSITSViewSelectorCheckbox wsView, viewRow
            End If
NextClosedView:
        Next rowId
    End If

    ' Remove view rows bottom-up so indexes stay valid.
    If rowsToRemove.Count > 0 Then
        SortLongCollectionDescending rowsToRemove
        For i = 1 To rowsToRemove.Count
            removeRow = CLng(rowsToRemove(i))
            DeleteSITSViewCheckboxOnRow wsView, removeRow
            wsView.Rows(removeRow).Delete
        Next i
        ' Row deletes shift shapes (xlMove) but leave LinkedCell stale —
        ' remap existing checkboxes instead of recreating every control.
        RemapSITSViewSelectorLinkedCells wsView
    End If
End Sub

' skipTableFormats:=True for day-to-day paths (sortie / extend / defer) where
' column NumberFormats were already applied at import or repair.
Public Sub RecalculateSITS(Optional ByVal skipTableFormats As Boolean = False)
    Dim wsData As Worksheet
    Dim tbl As ListObject
    Dim data As Variant
    Dim rowCount As Long
    Dim rowIndex As Long
    Dim intervalCode As String
    Dim taskCode As String
    Dim formatType As String
    Dim currentUsage As Variant
    Dim snapshotUsage As Variant
    Dim usageDelta As Double
    Dim baselineHours As Variant
    Dim baselineNumber As Variant
    Dim newLifeHours As Double
    Dim newLifeNumber As Double
    Dim colInterval As Long
    Dim colTaskCode As Long
    Dim colCurHours As Long
    Dim colCurNumber As Long
    Dim colLRHours As Long
    Dim colLRNumber As Long
    Dim colDaysDue As Long
    Dim colBaseHours As Long
    Dim colBaseNumber As Long
    Dim colBaseDue As Long
    Dim colProjected As Long
    Dim usageByInterval As Object
    Dim snapshotByInterval As Object
    Dim snapshotDate As Variant
    Dim colMap As Object

    Set wsData = GetWorksheet(SHEET_DATA_SITS, showError:=False)
    If wsData Is Nothing Then Exit Sub

    Set tbl = GetTable(wsData, TBL_SITS, showError:=False)
    If tbl Is Nothing Then Exit Sub

    EnsureSITSBaselineColumns tbl
    EnsureWorkPackageSITSColumns tbl

    Set colMap = BuildTableColumnIndexMap(tbl)
    colInterval = LookupColumnIndex(colMap, HDR_INTERVAL_TYPE_CODE)
    colTaskCode = LookupColumnIndex(colMap, HDR_TASK_CODE)
    colCurHours = LookupColumnIndex(colMap, HDR_CURRENT_USAGE_HOURS)
    colCurNumber = LookupColumnIndex(colMap, HDR_CURRENT_USAGE_NUMBER)
    colLRHours = LookupColumnIndex(colMap, HDR_HOURS_LIFE_REMAINING)
    colLRNumber = LookupColumnIndex(colMap, HDR_NUMBER_LIFE_REMAINING)
    colDaysDue = LookupColumnIndex(colMap, HDR_DAYS_WHEN_DUE)
    colBaseHours = LookupColumnIndex(colMap, HDR_BASELINE_LR_HOURS)
    colBaseNumber = LookupColumnIndex(colMap, HDR_BASELINE_LR_NUMBER)
    colBaseDue = LookupColumnIndex(colMap, HDR_BASELINE_DAYS_DUE)
    colProjected = LookupColumnIndex(colMap, HDR_PROJECTED_DUE_DATE)

    If tbl.DataBodyRange Is Nothing Then Exit Sub
    If colInterval = 0 Or colTaskCode = 0 Then Exit Sub

    data = NormalizeTableDataArray(tbl.DataBodyRange.Value, tbl.ListRows.Count, tbl.ListColumns.Count)
    If Not IsArray(data) Then Exit Sub
    rowCount = UBound(data, 1)

    SetExcelState "disable"
    Set usageByInterval = BuildUsageCurrentLookup()
    Set snapshotByInterval = BuildUsageSnapshotLookup()
    snapshotDate = GetConfigValue(CFG_SITS_SNAPSHOT_DATE, "")

    For rowIndex = 1 To rowCount
        intervalCode = UCase$(Trim$(CStr(data(rowIndex, colInterval))))
        taskCode = Trim$(CStr(data(rowIndex, colTaskCode)))

        If intervalCode = TYPE_DD Then
            ApplyDDTaskRecalcToArray data, rowIndex, colCurHours, colCurNumber, _
                colLRHours, colLRNumber, colDaysDue, colBaseDue, colProjected, snapshotDate
        ElseIf IsHourUsageInterval(intervalCode) Or _
               (IsHourBasedIntervalType(intervalCode) And Not IsSecondsCountInterval(intervalCode)) Then
            formatType = FORMAT_TYPE_HOURS
            currentUsage = LookupDictUsage(usageByInterval, intervalCode, formatType)
            snapshotUsage = LookupDictSnapshot(snapshotByInterval, intervalCode)

            If colCurHours > 0 Then
                If IsMissingUsageValue(currentUsage) Then
                    data(rowIndex, colCurHours) = Empty
                Else
                    data(rowIndex, colCurHours) = CoerceUsageStorageValue( _
                        currentUsage, formatType)
                End If
            End If
            If colCurNumber > 0 Then data(rowIndex, colCurNumber) = Empty

            baselineHours = Empty
            If colBaseHours > 0 Then baselineHours = data(rowIndex, colBaseHours)
            If IsEmptyBaselineValue(baselineHours) Then
                If colLRHours > 0 Then baselineHours = data(rowIndex, colLRHours)
                If Not IsEmptyBaselineValue(baselineHours) And colBaseHours > 0 Then
                    data(rowIndex, colBaseHours) = CoerceUsageStorageValue( _
                        baselineHours, formatType)
                End If
            End If

            If Not IsEmptyBaselineValue(baselineHours) And Not IsMissingUsageValue(currentUsage) And _
               Not IsMissingUsageValue(snapshotUsage) And colLRHours > 0 Then
                usageDelta = UsageValueAsDayFraction(currentUsage, formatType) - _
                             UsageValueAsDayFraction(snapshotUsage, formatType)
                newLifeHours = UsageValueAsDayFraction(baselineHours, formatType) - usageDelta
                data(rowIndex, colLRHours) = newLifeHours
            End If

            If colDaysDue > 0 Then data(rowIndex, colDaysDue) = Empty
        Else
            formatType = ResolveUsageFormatType(intervalCode, taskCode)
            currentUsage = LookupDictUsage(usageByInterval, intervalCode, formatType)
            snapshotUsage = LookupDictSnapshot(snapshotByInterval, intervalCode)

            If colCurHours > 0 Then data(rowIndex, colCurHours) = Empty
            If colCurNumber > 0 Then
                If IsMissingUsageValue(currentUsage) Then
                    data(rowIndex, colCurNumber) = "---"
                ElseIf StrComp(formatType, FORMAT_TYPE_NUMBER, vbTextCompare) = 0 Then
                    data(rowIndex, colCurNumber) = CDbl(NzSITSNumber(currentUsage))
                Else
                    data(rowIndex, colCurNumber) = CoerceUsageStorageValue( _
                        currentUsage, formatType)
                End If
            End If

            baselineNumber = Empty
            If colBaseNumber > 0 Then baselineNumber = data(rowIndex, colBaseNumber)
            If IsEmptyBaselineValue(baselineNumber) Then
                If colLRNumber > 0 Then baselineNumber = data(rowIndex, colLRNumber)
                If Not IsEmptyBaselineValue(baselineNumber) And colBaseNumber > 0 Then
                    data(rowIndex, colBaseNumber) = CDbl(baselineNumber)
                End If
            End If

            If Not IsEmptyBaselineValue(baselineNumber) And Not IsMissingUsageValue(currentUsage) And _
               Not IsMissingUsageValue(snapshotUsage) And colLRNumber > 0 Then
                usageDelta = CDbl(NzSITSNumber(currentUsage)) - CDbl(NzSITSNumber(snapshotUsage))
                newLifeNumber = CDbl(NzSITSNumber(baselineNumber)) - usageDelta
                data(rowIndex, colLRNumber) = newLifeNumber
            End If

            If colDaysDue > 0 Then data(rowIndex, colDaysDue) = Empty
        End If
    Next rowIndex

    tbl.DataBodyRange.Value = data
    ApplyRecalcColumnNumberFormats tbl, colCurHours, colCurNumber, colLRHours, _
        colLRNumber, colDaysDue, colBaseHours, colBaseNumber, colBaseDue

    If Not skipTableFormats Then ApplySITSTableFormats
    SetExcelState "enable"
End Sub

' Column-wide formats after bulk value write — cheap vs per-cell NumberFormat.
Private Sub ApplyRecalcColumnNumberFormats(ByVal tbl As ListObject, _
                                           ByVal colCurHours As Long, _
                                           ByVal colCurNumber As Long, _
                                           ByVal colLRHours As Long, _
                                           ByVal colLRNumber As Long, _
                                           ByVal colDaysDue As Long, _
                                           ByVal colBaseHours As Long, _
                                           ByVal colBaseNumber As Long, _
                                           ByVal colBaseDue As Long)
    On Error Resume Next
    If colCurHours > 0 Then tbl.ListColumns(colCurHours).DataBodyRange.NumberFormat = FORMAT_TIME_HOURS
    If colCurNumber > 0 Then tbl.ListColumns(colCurNumber).DataBodyRange.NumberFormat = FORMAT_DECIMAL
    If colLRHours > 0 Then tbl.ListColumns(colLRHours).DataBodyRange.NumberFormat = FORMAT_TIME_HOURS
    If colLRNumber > 0 Then tbl.ListColumns(colLRNumber).DataBodyRange.NumberFormat = FORMAT_DECIMAL
    If colDaysDue > 0 Then tbl.ListColumns(colDaysDue).DataBodyRange.NumberFormat = FORMAT_DECIMAL
    If colBaseHours > 0 Then tbl.ListColumns(colBaseHours).DataBodyRange.NumberFormat = FORMAT_TIME_HOURS
    If colBaseNumber > 0 Then tbl.ListColumns(colBaseNumber).DataBodyRange.NumberFormat = FORMAT_DECIMAL
    ' Absolute DD due dates and day-counts may share this column — General displays both.
    If colBaseDue > 0 Then tbl.ListColumns(colBaseDue).DataBodyRange.NumberFormat = FORMAT_GENERAL
    On Error GoTo 0
End Sub

' Excel returns a 1D array for a single-row DataBodyRange — normalize to 2D.
Private Function NormalizeTableDataArray(ByVal raw As Variant, _
                                         ByVal rowCount As Long, _
                                         ByVal colCount As Long) As Variant
    Dim data As Variant
    Dim c As Long

    If rowCount < 1 Or colCount < 1 Then Exit Function

    If Not IsArray(raw) Then
        ReDim data(1 To 1, 1 To 1)
        data(1, 1) = raw
        NormalizeTableDataArray = data
        Exit Function
    End If

    If rowCount = 1 Then
        c = 0
        On Error Resume Next
        c = UBound(raw, 2)
        If Err.Number <> 0 Then c = 0
        On Error GoTo 0
        If c >= 1 Then
            NormalizeTableDataArray = raw
            Exit Function
        End If

        ReDim data(1 To 1, 1 To colCount)
        For c = 1 To colCount
            data(1, c) = raw(c)
        Next c
        NormalizeTableDataArray = data
        Exit Function
    End If

    NormalizeTableDataArray = raw
End Function

Public Sub RefreshSITSView()
    Dim wsView As Worksheet

    Dim colDaysDue As Long
    Dim colTaskCode As Long
    Dim colIntervalCached As Long

    On Error GoTo CleanFail

    Set wsView = GetWorksheet(SHEET_SITS, showError:=False)
    Set wsData = GetWorksheet(SHEET_DATA_SITS, showError:=False)

    Set tbl = GetTable(wsData, TBL_SITS, showError:=False)

    Set viewColumns = BuildSITSViewColumns()
    forecastHours = GetForecastHoursSetting()

    forecastNumber = GetForecastNumberSetting()

    colManualFlag = LookupColumnIndex(colMap, HDR_NEEDS_MANUAL_UPDATE)
    colIntervalCached = LookupColumnIndex(colMap, HDR_INTERVAL_TYPE_CODE)
    colLRHours = LookupColumnIndex(colMap, HDR_HOURS_LIFE_REMAINING)
    colLRNumber = LookupColumnIndex(colMap, HDR_NUMBER_LIFE_REMAINING)
    colDaysDue = LookupColumnIndex(colMap, HDR_DAYS_WHEN_DUE)
    colTaskCode = LookupColumnIndex(colMap, HDR_TASK_CODE)

    SetExcelState "disable"
    Application.StatusBar = "Refreshing SITS view…"
    SuspendWorkbookProtection
    suspended = True

    ' Keep existing selector checkboxes — SyncSITSViewSelectorCheckboxes reuses them
    ' after the data rewrite (delete/recreate is the slow part of a full refresh).

    On Error Resume Next
    If wsView.AutoFilterMode Then wsView.AutoFilterMode = False
    ' Avoid Activate/FreezePanes churn unless the SITS sheet is already active.
    If ActiveSheet Is wsView Then ActiveWindow.FreezePanes = False
    On Error GoTo CleanFail

    ' Clear data/header rows only — preserve shell chrome and column widths.
    ClearSITSViewDataArea wsView, viewColumns.Count
    EnsureSITSViewShellExists wsView
    TidySITSViewShellChrome wsView
    wsView.Range("B2").Value = GetTailNumber()
    EnsureSITSViewActionButtons wsView

    outputCol = 1
    For Each columnName In viewColumns
        wsView.Cells(SITS_VIEW_HEADER_ROW, outputCol).Value = CStr(columnName)
        outputCol = outputCol + 1
    Next columnName

    With wsView.Range(wsView.Cells(SITS_VIEW_HEADER_ROW, 1), _
                      wsView.Cells(SITS_VIEW_HEADER_ROW, viewColumns.Count))
        .Font.Bold = True
        .Font.Color = vbWhite
        .Interior.Color = COLOR_HEADER_BLUE
        .WrapText = False
    End With

    outputRow = SITS_VIEW_DATA_START_ROW

    If Not tbl.DataBodyRange Is Nothing Then
        Dim sortEntries() As SITSViewSortEntry
        Dim sortCount As Long
        Dim entryIndex As Long
        Dim includeRow As Boolean
        Dim data As Variant
        Dim dataRowCount As Long
        Dim dataRowIndex As Long
        Dim colOrn As Long
        Dim colDesc As Long

        data = NormalizeTableDataArray(tbl.DataBodyRange.Value, _
            tbl.ListRows.Count, tbl.ListColumns.Count)
        If Not IsArray(data) Then GoTo AfterDataBody
        dataRowCount = UBound(data, 1)
        colOrn = LookupColumnIndex(colMap, HDR_ORN_INWORK)
        colDesc = LookupColumnIndex(colMap, HDR_TASK_DESCRIPTION)

        sortCount = 0
        ReDim sortEntries(1 To dataRowCount)

        For dataRowIndex = 1 To dataRowCount
            intervalCode = UCase$(Trim$(CStr(data(dataRowIndex, colIntervalCached))))

            includeRow = TaskPassesForecastFilterFromData(data, dataRowIndex, intervalCode, _
                forecastHours, forecastDays, forecastNumber, lifeHoursFraction, daysUntilDue, _
                colLRHours, colLRNumber, colDaysDue)
            If Not includeRow Then GoTo NextCollectRow

            sortCount = sortCount + 1
            sortEntries(sortCount).ListRowIndex = dataRowIndex
            sortEntries(sortCount).IntervalCode = intervalCode
            sortEntries(sortCount).GroupOrder = SITSIntervalGroupOrder(intervalCode)
            sortEntries(sortCount).SortKey = SITSTaskSortKeyFromData(data, dataRowIndex, intervalCode, _
                colLRHours, colLRNumber, colDaysDue, colTaskCode)
            sortEntries(sortCount).LifeHoursFraction = lifeHoursFraction
            sortEntries(sortCount).DaysUntilDue = daysUntilDue

NextCollectRow:
        Next dataRowIndex

        If sortCount > 0 Then
            ReDim Preserve sortEntries(1 To sortCount)
            SortSITSViewEntries sortEntries

            Dim colCount As Long
            Dim values() As Variant
            Dim formats() As Variant
            Dim comments() As String
            Dim colNames() As String
            Dim dataRange As Range
            Dim cellValue As Variant
            Dim cellFormat As String
            Dim commentText As String
            Dim descCol As Long
            Dim hasComments As Boolean
            Dim arrayCol As Long
            Dim warnColor As Long
            Dim isManual As Boolean

            colCount = viewColumns.Count
            ReDim values(1 To sortCount, 1 To colCount)
            ReDim formats(1 To sortCount, 1 To colCount)
            ReDim comments(1 To sortCount)
            ReDim colNames(1 To colCount)

            arrayCol = 1
            For Each columnName In viewColumns
                colNames(arrayCol) = CStr(columnName)
                If StrComp(colNames(arrayCol), HDR_TASK_DESCRIPTION, vbTextCompare) = 0 Then
                    descCol = arrayCol
                End If
                arrayCol = arrayCol + 1
            Next columnName

            Application.StatusBar = "Building SITS view data…"
            For entryIndex = 1 To sortCount
                dataRowIndex = sortEntries(entryIndex).ListRowIndex
                intervalCode = sortEntries(entryIndex).IntervalCode

                For arrayCol = 1 To colCount
                    BuildSITSViewCellPayloadFromData data, dataRowIndex, intervalCode, _
                        colNames(arrayCol), colMap, colOrn, _
                        cellValue, cellFormat, commentText
                    If IsEmpty(cellValue) Then
                        values(entryIndex, arrayCol) = Empty
                    Else
                        values(entryIndex, arrayCol) = cellValue
                    End If
                    formats(entryIndex, arrayCol) = cellFormat
                Next arrayCol

                ' Comments are not in Value arrays — read only for visible description cells.
                If descCol > 0 And colDesc > 0 Then
                    commentText = GetCellCommentText(tbl.DataBodyRange.Cells(dataRowIndex, colDesc))
                    If Len(commentText) > 0 Then
                        comments(entryIndex) = commentText
                        hasComments = True
                    End If
                End If
            Next entryIndex

            lastDataRow = SITS_VIEW_DATA_START_ROW + sortCount - 1
            Set dataRange = wsView.Range( _
                wsView.Cells(SITS_VIEW_DATA_START_ROW, 1), _
                wsView.Cells(lastDataRow, colCount))
            dataRange.Value = values
            dataRange.NumberFormat = formats
            dataRange.WrapText = False

            If hasComments And descCol > 0 Then
                For entryIndex = 1 To sortCount
                    If Len(comments(entryIndex)) > 0 Then
                        SetCellCommentText _
                            wsView.Cells(SITS_VIEW_DATA_START_ROW + entryIndex - 1, descCol), _
                            comments(entryIndex)
                    End If
                Next entryIndex
            End If

            Application.StatusBar = "Applying SITS warning colours…"
            For entryIndex = 1 To sortCount
                dataRowIndex = sortEntries(entryIndex).ListRowIndex
                isManual = False
                If colManualFlag > 0 Then
                    isManual = ConvertToBoolean(data(dataRowIndex, colManualFlag))
                End If
                warnColor = ResolveSITSViewWarningColorValues( _
                    sortEntries(entryIndex).IntervalCode, warnHoursFraction, warnDays, _
                    isManual, data, dataRowIndex, colLRHours, colDaysDue)
                If warnColor <> 0 Then
                    wsView.Range( _
                        wsView.Cells(SITS_VIEW_DATA_START_ROW + entryIndex - 1, 1), _
                        wsView.Cells(SITS_VIEW_DATA_START_ROW + entryIndex - 1, SITS_VISIBLE_COLUMN_COUNT) _
                        ).Interior.Color = warnColor
                End If
            Next entryIndex

            outputRow = lastDataRow + 1
        End If
AfterDataBody:
    End If

    lastDataRow = outputRow - 1
    If lastDataRow >= SITS_VIEW_DATA_START_ROW Then
        With wsView.Range(wsView.Cells(SITS_VIEW_DATA_START_ROW, 1), _
                          wsView.Cells(lastDataRow, viewColumns.Count))
            .WrapText = False
        End With
    End If

    ' Preserve developer column widths; only apply visibility + text alignment.
    ApplySITSViewColumnPresentation wsView, lastDataRow, viewColumns.Count

    If lastDataRow >= SITS_VIEW_DATA_START_ROW Then
        Application.StatusBar = "Syncing SITS selectors…"
        SyncSITSViewSelectorCheckboxes wsView, SITS_VIEW_DATA_START_ROW, lastDataRow
    Else
        ClearSITSViewSelectorCheckboxes wsView
    End If

    ' Range.Clear above resets cell Locked=True; restore selector unlocks so
    ' Form Control checkboxes can write LinkedCells on a protected sheet.
    UnlockSITSSelectorInputs wsView

    ApplySITSViewAutoFilter wsView, lastDataRow, SITS_VISIBLE_COLUMN_COUNT
    If ActiveSheet Is wsView Then ApplySITSViewFreezePanes wsView
    ResumeWorkbookProtection
    suspended = False
    Application.StatusBar = False
    SetExcelState "enable"
    Exit Sub

CleanFail:
    Application.StatusBar = False
    If suspended Then ResumeWorkbookProtection
    SetExcelState "force-enable"
End Sub

Public Sub SnapshotUsageForSITS()
    Dim wsCurrent As Worksheet
    Dim tblCurrent As ListObject
    Dim wsConfig As Worksheet
    Dim tblSnapshot As ListObject
    Dim row As ListRow
    Dim intervalCode As String
    Dim taskCode As String
    Dim formatType As String
    Dim newRow As ListRow
    Dim seen As Object

    Set wsCurrent = GetWorksheet(SHEET_DATA_USAGE_CURRENT, showError:=False)
    If wsCurrent Is Nothing Then Exit Sub

    Set tblCurrent = GetTable(wsCurrent, TBL_USAGE_CURRENT, showError:=False)
    If tblCurrent Is Nothing Then Exit Sub

    Set wsConfig = GetWorksheet(SHEET_CONFIG, showError:=False)
    If wsConfig Is Nothing Then Exit Sub

    On Error GoTo SnapshotFail
    SuspendWorkbookProtection

    EnsureUsageSnapshotTable wsConfig
    Set tblSnapshot = GetTable(wsConfig, TBL_USAGE_SNAPSHOT, showError:=False)
    If tblSnapshot Is Nothing Then
        ResumeWorkbookProtection
        Exit Sub
    End If

    ClearTableData tblSnapshot
    Set seen = CreateObject("Scripting.Dictionary")

    If Not tblCurrent.DataBodyRange Is Nothing Then
        For Each row In tblCurrent.ListRows
            intervalCode = UCase$(Trim$(CStr(row.Range.Cells(1, 1).Value)))
            If Len(intervalCode) = 0 Then GoTo NextUsageRow
            If seen.Exists(intervalCode) Then GoTo NextUsageRow

            taskCode = Trim$(CStr(row.Range.Cells(1, 2).Value))
            formatType = ResolveUsageFormatType(intervalCode, taskCode)
            seen.Add intervalCode, True

            Set newRow = tblSnapshot.ListRows.Add
            newRow.Range.Cells(1, 1).Value = intervalCode
            WriteSnapshotValueCell newRow.Range.Cells(1, 2), row.Range.Cells(1, 3).Value, formatType
            newRow.Range.Cells(1, 3).Value = formatType
NextUsageRow:
        Next row
    End If

    SetConfigValue CFG_SITS_SNAPSHOT_DATE, Date
    ResumeWorkbookProtection
    Exit Sub

SnapshotFail:
    ForceResumeWorkbookProtection
End Sub

Public Sub CaptureTaskBaselineFromImport(ByVal tbl As ListObject, ByVal newRow As ListRow, _
                                         ByVal intervalCode As String, _
                                         Optional ByVal colMap As Object = Nothing)
    Dim colLRHours As Long
    Dim colLRNumber As Long
    Dim colDaysDue As Long
    Dim colBaseHours As Long
    Dim colBaseNumber As Long
    Dim colBaseDue As Long
    Dim colProjected As Long
    Dim dueValue As Variant
    Dim projectedDue As Variant

    EnsureSITSBaselineColumns tbl

    ' Baseline columns may have been added — rebuild map if caller passed a stale one.
    If Not colMap Is Nothing Then
        If LookupColumnIndex(colMap, HDR_BASELINE_LR_HOURS) = 0 Then
            Set colMap = BuildTableColumnIndexMap(tbl)
        End If
    End If

    colLRHours = ResolveTableColumnIndex(tbl, HDR_HOURS_LIFE_REMAINING, colMap)
    colLRNumber = ResolveTableColumnIndex(tbl, HDR_NUMBER_LIFE_REMAINING, colMap)
    colDaysDue = ResolveTableColumnIndex(tbl, HDR_DAYS_WHEN_DUE, colMap)
    colBaseHours = ResolveTableColumnIndex(tbl, HDR_BASELINE_LR_HOURS, colMap)
    colBaseNumber = ResolveTableColumnIndex(tbl, HDR_BASELINE_LR_NUMBER, colMap)
    colBaseDue = ResolveTableColumnIndex(tbl, HDR_BASELINE_DAYS_DUE, colMap)

    If intervalCode = TYPE_DD Then
        colProjected = ResolveTableColumnIndex(tbl, HDR_PROJECTED_DUE_DATE, colMap)
        If colProjected > 0 Then
            projectedDue = newRow.Range.Cells(1, colProjected).Value
            If LooksLikeAbsoluteDueDate(projectedDue) Then
                newRow.Range.Cells(1, colBaseDue).Value = CDate(projectedDue)
                newRow.Range.Cells(1, colBaseDue).NumberFormat = FORMAT_DATE
                Exit Sub
            End If
        End If

        dueValue = newRow.Range.Cells(1, colDaysDue).Value
        If LooksLikeAbsoluteDueDate(dueValue) Then
            newRow.Range.Cells(1, colBaseDue).Value = CDate(dueValue)
            newRow.Range.Cells(1, colBaseDue).NumberFormat = FORMAT_DATE
        ElseIf IsNumeric(dueValue) Then
            newRow.Range.Cells(1, colBaseDue).Value = CDbl(dueValue)
            newRow.Range.Cells(1, colBaseDue).NumberFormat = FORMAT_DECIMAL
        End If
        Exit Sub
    End If

    If IsHourUsageInterval(intervalCode) Or _
       (IsHourBasedIntervalType(intervalCode) And Not IsSecondsCountInterval(intervalCode)) Then
        If Not IsEmpty(newRow.Range.Cells(1, colLRHours).Value) Then
            newRow.Range.Cells(1, colBaseHours).Value = newRow.Range.Cells(1, colLRHours).Value
            newRow.Range.Cells(1, colBaseHours).NumberFormat = newRow.Range.Cells(1, colLRHours).NumberFormat
        End If
    Else
        If Not IsEmpty(newRow.Range.Cells(1, colLRNumber).Value) Then
            newRow.Range.Cells(1, colBaseNumber).Value = newRow.Range.Cells(1, colLRNumber).Value
            newRow.Range.Cells(1, colBaseNumber).NumberFormat = newRow.Range.Cells(1, colLRNumber).NumberFormat
        End If
    End If
End Sub

Private Sub ApplyDDTaskRecalcToArray(ByRef data As Variant, _
                                     ByVal rowIndex As Long, _
                                     ByVal colCurHours As Long, _
                                     ByVal colCurNumber As Long, _
                                     ByVal colLRHours As Long, _
                                     ByVal colLRNumber As Long, _
                                     ByVal colDaysDue As Long, _
                                     ByVal colBaseDue As Long, _
                                     ByVal colProjected As Long, _
                                     ByVal snapshotDate As Variant)
    Dim baselineDue As Variant
    Dim projectedDue As Variant
    Dim elapsedDays As Long
    Dim daysRemaining As Double
    Dim absoluteDue As Date
    Dim haveAbsoluteDue As Boolean

    If colCurHours > 0 Then data(rowIndex, colCurHours) = Empty
    If colCurNumber > 0 Then data(rowIndex, colCurNumber) = Empty
    If colLRHours > 0 Then data(rowIndex, colLRHours) = Empty
    If colLRNumber > 0 Then data(rowIndex, colLRNumber) = Empty

    haveAbsoluteDue = False
    If colProjected > 0 Then
        projectedDue = data(rowIndex, colProjected)
        If LooksLikeAbsoluteDueDate(projectedDue) Then
            absoluteDue = CDate(projectedDue)
            haveAbsoluteDue = True
        End If
    End If

    baselineDue = Empty
    If colBaseDue > 0 Then baselineDue = data(rowIndex, colBaseDue)
    If Not haveAbsoluteDue Then
        If LooksLikeAbsoluteDueDate(baselineDue) Then
            absoluteDue = CDate(baselineDue)
            haveAbsoluteDue = True
        ElseIf IsEmptyBaselineValue(baselineDue) Then
            If colDaysDue > 0 Then baselineDue = data(rowIndex, colDaysDue)
            If LooksLikeAbsoluteDueDate(baselineDue) Then
                absoluteDue = CDate(baselineDue)
                haveAbsoluteDue = True
            End If
        End If
    End If

    If haveAbsoluteDue Then
        If colBaseDue > 0 Then data(rowIndex, colBaseDue) = absoluteDue
        daysRemaining = CDbl(DateDiff("d", Date, absoluteDue))
        If colDaysDue > 0 Then data(rowIndex, colDaysDue) = daysRemaining
        Exit Sub
    End If

    If IsEmptyBaselineValue(baselineDue) Then
        If colDaysDue > 0 Then baselineDue = data(rowIndex, colDaysDue)
        If IsNumeric(baselineDue) And Len(Trim$(CStr(baselineDue))) > 0 And colBaseDue > 0 Then
            data(rowIndex, colBaseDue) = CDbl(baselineDue)
        End If
    End If

    If IsNumeric(baselineDue) And Len(Trim$(CStr(baselineDue))) > 0 Then
        daysRemaining = CDbl(baselineDue)
        If IsDate(snapshotDate) Then
            elapsedDays = DateDiff("d", CDate(snapshotDate), Date)
            daysRemaining = daysRemaining - CDbl(elapsedDays)
        End If
        If colDaysDue > 0 Then data(rowIndex, colDaysDue) = daysRemaining
    End If
End Sub

Private Sub ApplyDDTaskRecalc(ByVal row As ListRow, _
                              ByVal tbl As ListObject, _
                              ByVal colCurHours As Long, _
                              ByVal colCurNumber As Long, _
                              ByVal colLRHours As Long, _
                              ByVal colLRNumber As Long, _
                              ByVal colDaysDue As Long, _
                              ByVal colBaseDue As Long)
    Dim baselineDue As Variant
    Dim projectedDue As Variant
    Dim snapshotDate As Variant
    Dim elapsedDays As Long
    Dim daysRemaining As Double
    Dim absoluteDue As Date
    Dim colProjected As Long
    Dim haveAbsoluteDue As Boolean

    row.Range.Cells(1, colCurHours).ClearContents
    row.Range.Cells(1, colCurNumber).ClearContents
    row.Range.Cells(1, colLRHours).ClearContents
    row.Range.Cells(1, colLRNumber).ClearContents

    haveAbsoluteDue = False
    colProjected = GetTableColumnIndex(tbl, HDR_PROJECTED_DUE_DATE)
    If colProjected > 0 Then
        projectedDue = row.Range.Cells(1, colProjected).Value
        If LooksLikeAbsoluteDueDate(projectedDue) Then
            absoluteDue = CDate(projectedDue)
            haveAbsoluteDue = True
        End If
    End If

    baselineDue = row.Range.Cells(1, colBaseDue).Value
    If Not haveAbsoluteDue Then
        If LooksLikeAbsoluteDueDate(baselineDue) Then
            absoluteDue = CDate(baselineDue)
            haveAbsoluteDue = True
        ElseIf IsEmptyBaselineValue(baselineDue) Then
            baselineDue = row.Range.Cells(1, colDaysDue).Value
            If LooksLikeAbsoluteDueDate(baselineDue) Then
                absoluteDue = CDate(baselineDue)
                haveAbsoluteDue = True
            End If
        End If
    End If

    If haveAbsoluteDue Then
        ' Prefer absolute due date (Projected Due Date / baseline date).
        ' Day-count baselines freeze at 0 once due and never go negative.
        If colBaseDue > 0 Then
            row.Range.Cells(1, colBaseDue).Value = absoluteDue
            row.Range.Cells(1, colBaseDue).NumberFormat = FORMAT_DATE
        End If
        daysRemaining = CDbl(DateDiff("d", Date, absoluteDue))
        row.Range.Cells(1, colDaysDue).Value = daysRemaining
        row.Range.Cells(1, colDaysDue).NumberFormat = FORMAT_DECIMAL
        Exit Sub
    End If

    ' Fallback: imported day-count baseline minus elapsed days since snapshot.
    If IsEmptyBaselineValue(baselineDue) Then
        baselineDue = row.Range.Cells(1, colDaysDue).Value
        If IsNumeric(baselineDue) And Len(Trim$(CStr(baselineDue))) > 0 Then
            row.Range.Cells(1, colBaseDue).Value = CDbl(baselineDue)
            row.Range.Cells(1, colBaseDue).NumberFormat = FORMAT_DECIMAL
        End If
    End If

    If IsNumeric(baselineDue) And Len(Trim$(CStr(baselineDue))) > 0 Then
        snapshotDate = GetConfigValue(CFG_SITS_SNAPSHOT_DATE, "")
        daysRemaining = CDbl(baselineDue)
        If IsDate(snapshotDate) Then
            elapsedDays = DateDiff("d", CDate(snapshotDate), Date)
            daysRemaining = daysRemaining - CDbl(elapsedDays)
        End If
        row.Range.Cells(1, colDaysDue).Value = daysRemaining
        row.Range.Cells(1, colDaysDue).NumberFormat = FORMAT_DECIMAL
    End If
End Sub

Private Function SITSIntervalGroupOrder(ByVal intervalCode As String) As Long
    Select Case UCase$(Trim$(intervalCode))
        Case TYPE_E1
            SITSIntervalGroupOrder = 1
        Case TYPE_E2
            SITSIntervalGroupOrder = 2
        Case TYPE_HH
            SITSIntervalGroupOrder = 3
        Case TYPE_DD
            SITSIntervalGroupOrder = 4
        Case Else
            SITSIntervalGroupOrder = 5
    End Select
End Function

Private Function SITSTaskSortKey(ByVal row As ListRow, _
                                 ByVal tbl As ListObject, _
                                 ByVal intervalCode As String, _
                                 Optional ByVal colLRHours As Long = 0, _
                                 Optional ByVal colLRNumber As Long = 0, _
                                 Optional ByVal colDaysDue As Long = 0, _
                                 Optional ByVal colTaskCode As Long = 0) As Double
    Dim dueValue As Variant

    Select Case UCase$(Trim$(intervalCode))
        Case TYPE_E1, TYPE_E2, TYPE_HH
            If colLRHours = 0 Then colLRHours = GetTableColumnIndex(tbl, HDR_HOURS_LIFE_REMAINING)
            If colLRHours > 0 And IsNumeric(row.Range.Cells(1, colLRHours).Value) Then
                SITSTaskSortKey = CDbl(row.Range.Cells(1, colLRHours).Value)
                Exit Function
            End If

        Case TYPE_DD
            If colDaysDue = 0 Then colDaysDue = GetTableColumnIndex(tbl, HDR_DAYS_WHEN_DUE)
            dueValue = row.Range.Cells(1, colDaysDue).Value
            If LooksLikeAbsoluteDueDate(dueValue) Then
                SITSTaskSortKey = CDbl(CDate(dueValue))
                Exit Function
            ElseIf IsNumeric(dueValue) Then
                SITSTaskSortKey = CDbl(dueValue)
                Exit Function
            End If

        Case Else
            If colLRNumber = 0 Then colLRNumber = GetTableColumnIndex(tbl, HDR_NUMBER_LIFE_REMAINING)
            If colLRNumber > 0 And IsNumeric(row.Range.Cells(1, colLRNumber).Value) Then
                SITSTaskSortKey = CDbl(row.Range.Cells(1, colLRNumber).Value)
                Exit Function
            End If

            If colLRHours = 0 Then colLRHours = GetTableColumnIndex(tbl, HDR_HOURS_LIFE_REMAINING)
            If colLRHours > 0 And IsNumeric(row.Range.Cells(1, colLRHours).Value) Then
                SITSTaskSortKey = CDbl(row.Range.Cells(1, colLRHours).Value)
                Exit Function
            End If

            If colTaskCode = 0 Then colTaskCode = GetTableColumnIndex(tbl, HDR_TASK_CODE)
            If colTaskCode > 0 Then
                SITSTaskSortKey = 1E+15 + Len(CStr(row.Range.Cells(1, colTaskCode).Value))
                Exit Function
            End If
    End Select

    SITSTaskSortKey = SITS_SORT_KEY_MISSING
End Function

Private Function SITSTaskSortKeyFromData(ByRef data As Variant, _
                                         ByVal rowIndex As Long, _
                                         ByVal intervalCode As String, _
                                         ByVal colLRHours As Long, _
                                         ByVal colLRNumber As Long, _
                                         ByVal colDaysDue As Long, _
                                         ByVal colTaskCode As Long) As Double
    Dim dueValue As Variant

    Select Case UCase$(Trim$(intervalCode))
        Case TYPE_E1, TYPE_E2, TYPE_HH
            If colLRHours > 0 And IsNumeric(data(rowIndex, colLRHours)) Then
                SITSTaskSortKeyFromData = CDbl(data(rowIndex, colLRHours))
                Exit Function
            End If

        Case TYPE_DD
            If colDaysDue > 0 Then
                dueValue = data(rowIndex, colDaysDue)
                If LooksLikeAbsoluteDueDate(dueValue) Then
                    SITSTaskSortKeyFromData = CDbl(CDate(dueValue))
                    Exit Function
                ElseIf IsNumeric(dueValue) Then
                    SITSTaskSortKeyFromData = CDbl(dueValue)
                    Exit Function
                End If
            End If

        Case Else
            If colLRNumber > 0 And IsNumeric(data(rowIndex, colLRNumber)) Then
                SITSTaskSortKeyFromData = CDbl(data(rowIndex, colLRNumber))
                Exit Function
            End If

            If colLRHours > 0 And IsNumeric(data(rowIndex, colLRHours)) Then
                SITSTaskSortKeyFromData = CDbl(data(rowIndex, colLRHours))
                Exit Function
            End If

            If colTaskCode > 0 Then
                SITSTaskSortKeyFromData = 1E+15 + Len(CStr(data(rowIndex, colTaskCode)))
                Exit Function
            End If
    End Select

    SITSTaskSortKeyFromData = SITS_SORT_KEY_MISSING
End Function

Private Sub SortSITSViewEntries(ByRef entries() As SITSViewSortEntry)
    If UBound(entries) <= LBound(entries) Then Exit Sub
    QuickSortSITSViewEntries entries, LBound(entries), UBound(entries)
End Sub

Private Sub QuickSortSITSViewEntries(ByRef entries() As SITSViewSortEntry, _
                                     ByVal lowIndex As Long, _
                                     ByVal highIndex As Long)
    Dim i As Long
    Dim j As Long
    Dim pivot As SITSViewSortEntry
    Dim temp As SITSViewSortEntry

    i = lowIndex
    j = highIndex
    pivot = entries((lowIndex + highIndex) \ 2)

    Do While i <= j
        Do While SITSViewEntryLess(entries(i), pivot)
            i = i + 1
        Loop
        Do While SITSViewEntryLess(pivot, entries(j))
            j = j - 1
        Loop
        If i <= j Then
            temp = entries(i)
            entries(i) = entries(j)
            entries(j) = temp
            i = i + 1
            j = j - 1
        End If
    Loop

    If lowIndex < j Then QuickSortSITSViewEntries entries, lowIndex, j
    If i < highIndex Then QuickSortSITSViewEntries entries, i, highIndex
End Sub

Private Function SITSViewEntryLess(ByRef leftEntry As SITSViewSortEntry, _
                                   ByRef rightEntry As SITSViewSortEntry) As Boolean
    If leftEntry.GroupOrder < rightEntry.GroupOrder Then
        SITSViewEntryLess = True
    ElseIf leftEntry.GroupOrder > rightEntry.GroupOrder Then
        SITSViewEntryLess = False
    ElseIf leftEntry.SortKey < rightEntry.SortKey Then
        SITSViewEntryLess = True
    ElseIf leftEntry.SortKey > rightEntry.SortKey Then
        SITSViewEntryLess = False
    Else
        SITSViewEntryLess = (leftEntry.ListRowIndex < rightEntry.ListRowIndex)
    End If
End Function

Private Function TaskPassesForecastFilter(ByVal row As ListRow, _
                                          ByVal tbl As ListObject, _
                                          ByVal intervalCode As String, _
                                          ByVal forecastHours As Double, _
                                          ByVal forecastDays As Long, _
                                          ByVal forecastNumber As Long, _
                                          ByRef lifeHoursFraction As Double, _
                                          ByRef daysUntilDue As Long, _
                                          Optional ByVal colLRHours As Long = 0, _
                                          Optional ByVal colLRNumber As Long = 0, _
                                          Optional ByVal colDaysDue As Long = 0) As Boolean
    Dim dueValue As Variant
    Dim lifeNumber As Double

    lifeHoursFraction = -1#
    daysUntilDue = -1

    Select Case intervalCode
        Case TYPE_HH, TYPE_E1, TYPE_E2
            If colLRHours = 0 Then colLRHours = GetTableColumnIndex(tbl, HDR_HOURS_LIFE_REMAINING)
            If colLRHours > 0 And IsNumeric(row.Range.Cells(1, colLRHours).Value) Then
                lifeHoursFraction = CDbl(row.Range.Cells(1, colLRHours).Value)
            End If
            If forecastHours <= 0# Then
                TaskPassesForecastFilter = True
            ElseIf lifeHoursFraction < 0# Then
                TaskPassesForecastFilter = True
            Else
                TaskPassesForecastFilter = (lifeHoursFraction <= (forecastHours / 24#))
            End If

        Case TYPE_DD
            If colDaysDue = 0 Then colDaysDue = GetTableColumnIndex(tbl, HDR_DAYS_WHEN_DUE)
            dueValue = row.Range.Cells(1, colDaysDue).Value
            If LooksLikeAbsoluteDueDate(dueValue) Then
                daysUntilDue = CLng(DateDiff("d", Date, CDate(dueValue)))
            ElseIf IsNumeric(dueValue) Then
                daysUntilDue = CLng(CDbl(dueValue))
            End If
            If forecastDays <= 0 Then
                TaskPassesForecastFilter = True
            ElseIf daysUntilDue < 0 Then
                TaskPassesForecastFilter = True
            Else
                TaskPassesForecastFilter = (daysUntilDue <= forecastDays)
            End If

        Case Else
            If IsNumberBasedSITSInterval(intervalCode) Then
                If colLRNumber = 0 Then colLRNumber = GetTableColumnIndex(tbl, HDR_NUMBER_LIFE_REMAINING)
                lifeNumber = -1#
                If colLRNumber > 0 And IsNumeric(row.Range.Cells(1, colLRNumber).Value) Then
                    lifeNumber = CDbl(row.Range.Cells(1, colLRNumber).Value)
                End If
                If forecastNumber <= 0 Then
                    TaskPassesForecastFilter = True
                ElseIf lifeNumber < 0# Then
                    TaskPassesForecastFilter = True
                Else
                    TaskPassesForecastFilter = (lifeNumber <= forecastNumber)
                End If
            Else
                TaskPassesForecastFilter = True
            End If
    End Select
End Function

Private Function TaskPassesForecastFilterFromData(ByRef data As Variant, _
                                                  ByVal rowIndex As Long, _
                                                  ByVal intervalCode As String, _
                                                  ByVal forecastHours As Double, _
                                                  ByVal forecastDays As Long, _
                                                  ByVal forecastNumber As Long, _
                                                  ByRef lifeHoursFraction As Double, _
                                                  ByRef daysUntilDue As Long, _
                                                  ByVal colLRHours As Long, _
                                                  ByVal colLRNumber As Long, _
                                                  ByVal colDaysDue As Long) As Boolean
    Dim dueValue As Variant
    Dim lifeNumber As Double

    lifeHoursFraction = -1#
    daysUntilDue = -1

    Select Case intervalCode
        Case TYPE_HH, TYPE_E1, TYPE_E2
            If colLRHours > 0 And IsNumeric(data(rowIndex, colLRHours)) Then
                lifeHoursFraction = CDbl(data(rowIndex, colLRHours))
            End If
            If forecastHours <= 0# Then
                TaskPassesForecastFilterFromData = True
            ElseIf lifeHoursFraction < 0# Then
                TaskPassesForecastFilterFromData = True
            Else
                TaskPassesForecastFilterFromData = (lifeHoursFraction <= (forecastHours / 24#))
            End If

        Case TYPE_DD
            If colDaysDue > 0 Then
                dueValue = data(rowIndex, colDaysDue)
                If LooksLikeAbsoluteDueDate(dueValue) Then
                    daysUntilDue = CLng(DateDiff("d", Date, CDate(dueValue)))
                ElseIf IsNumeric(dueValue) Then
                    daysUntilDue = CLng(CDbl(dueValue))
                End If
            End If
            If forecastDays <= 0 Then
                TaskPassesForecastFilterFromData = True
            ElseIf daysUntilDue < 0 Then
                TaskPassesForecastFilterFromData = True
            Else
                TaskPassesForecastFilterFromData = (daysUntilDue <= forecastDays)
            End If

        Case Else
            If IsNumberBasedSITSInterval(intervalCode) Then
                lifeNumber = -1#
                If colLRNumber > 0 And IsNumeric(data(rowIndex, colLRNumber)) Then
                    lifeNumber = CDbl(data(rowIndex, colLRNumber))
                End If
                If forecastNumber <= 0 Then
                    TaskPassesForecastFilterFromData = True
                ElseIf lifeNumber < 0# Then
                    TaskPassesForecastFilterFromData = True
                Else
                    TaskPassesForecastFilterFromData = (lifeNumber <= forecastNumber)
                End If
            Else
                TaskPassesForecastFilterFromData = True
            End If
    End Select
End Function

Private Function FormatHorizonHours(ByVal hoursValue As Double) As String
    If hoursValue = Fix(hoursValue) Then
        FormatHorizonHours = CStr(CLng(hoursValue)) & "h"
    Else
        FormatHorizonHours = Format$(hoursValue, "0.##") & "h"
    End If
End Function

Private Function ParseHorizonHours(ByVal textValue As String, _
                                   ByRef hoursValue As Double, _
                                   ByRef failReason As String) As Boolean
    Dim trimmed As String

    failReason = ""
    trimmed = Trim$(textValue)
    If Len(trimmed) = 0 Then
        failReason = "Hours is required (use 0 to show all)."
        Exit Function
    End If
    If Not IsNumeric(trimmed) Then
        failReason = "Hours must be a number (use 0 to show all)."
        Exit Function
    End If
    If CDbl(trimmed) < 0# Then
        failReason = "Hours cannot be negative."
        Exit Function
    End If

    hoursValue = CDbl(trimmed)
    ParseHorizonHours = True
End Function

Private Function ParseHorizonWholeNumber(ByVal textValue As String, _
                                         ByVal fieldName As String, _
                                         ByRef wholeValue As Long, _
                                         ByRef failReason As String) As Boolean
    Dim trimmed As String
    Dim parsed As Double

    failReason = ""
    trimmed = Trim$(textValue)
    If Len(trimmed) = 0 Then
        failReason = fieldName & " is required (use 0 to show all)."
        Exit Function
    End If
    If Not IsNumeric(trimmed) Then
        failReason = fieldName & " must be a number (use 0 to show all)."
        Exit Function
    End If

    parsed = CDbl(trimmed)
    If parsed < 0# Then
        failReason = fieldName & " cannot be negative."
        Exit Function
    End If
    If parsed <> Fix(parsed) Then
        failReason = fieldName & " must be a whole number."
        Exit Function
    End If

    wholeValue = CLng(parsed)
    ParseHorizonWholeNumber = True
End Function

' Clear header + data rows without touching rows 1–6 (shell / buttons).
Private Sub ClearSITSViewDataArea(ByVal ws As Worksheet, ByVal columnCount As Long)
    Dim lastRow As Long
    Dim clearCols As Long

    clearCols = columnCount
    If clearCols < SITS_VISIBLE_COLUMN_COUNT Then clearCols = SITS_VISIBLE_COLUMN_COUNT
    If clearCols < 1 Then clearCols = 1

    lastRow = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row
    If lastRow < SITS_VIEW_HEADER_ROW Then lastRow = SITS_VIEW_HEADER_ROW

    On Error Resume Next
    ' Prefer UsedRange bottom so leftover rows beyond col A are cleared too.
    If ws.UsedRange.Rows.Count + ws.UsedRange.Row - 1 > lastRow Then
        lastRow = ws.UsedRange.Rows.Count + ws.UsedRange.Row - 1
    End If
    On Error GoTo 0

    If lastRow >= SITS_VIEW_HEADER_ROW Then
        ws.Range(ws.Cells(SITS_VIEW_HEADER_ROW, 1), ws.Cells(lastRow, clearCols)).Clear
    End If
End Sub

Private Sub ClearSITSViewSelectorCheckboxes(ByVal ws As Worksheet)

    Dim i As Long

    On Error Resume Next
    For i = ws.CheckBoxes.Count To 1 Step -1
        Set cb = ws.CheckBoxes(i)
        actionName = Trim$(CStr(cb.OnAction))
        bangPos = InStrRev(actionName, "!")

        If shp.Type = msoFormControl Then
            If Left$(shp.Name, 8) = "SITS_SEL" Then shp.Delete
        End If
    Next i
    On Error GoTo 0
End Sub

Private Function FindSheetButtonForMacro(ByVal ws As Worksheet, ByVal macroName As String) As Button
    Dim btn As Button
    Dim actionName As String
    Dim bangPos As Long

    Set FindSheetButtonForMacro = Nothing
    On Error Resume Next
    For Each btn In ws.Buttons
        actionName = Trim$(CStr(btn.OnAction))
        bangPos = InStrRev(actionName, "!")
        If bangPos > 0 Then actionName = Mid$(actionName, bangPos + 1)
        If StrComp(actionName, macroName, vbTextCompare) = 0 Then
            Set FindSheetButtonForMacro = btn
            Exit Function
        End If
    Next btn
    On Error GoTo 0

' Reuse existing Form Control checkboxes when the visible row set changes.
' Creating/deleting hundreds of CheckBoxes dominates RefreshSITSView wall time.
Private Sub SyncSITSViewSelectorCheckboxes(ByVal ws As Worksheet, _
                                           ByVal firstDataRow As Long, _
                                           ByVal lastDataRow As Long)
    Dim pool As Collection
    Dim cb As CheckBox
    Dim linkCell As Range
    Dim rowIndex As Long
    Dim i As Long
    Dim actionName As String
    Dim bangPos As Long
    Dim leftPos As Double
    Dim widthPos As Double
    Dim heightPos As Double
    Dim cellText As String
    Dim poolIndex As Long

    If ws Is Nothing Then Exit Sub

    If lastDataRow < firstDataRow Then
        ClearSITSViewSelectorCheckboxes ws
        Exit Sub
    End If

    Set pool = New Collection
    On Error Resume Next
    For i = ws.CheckBoxes.Count To 1 Step -1
        Set cb = ws.CheckBoxes(i)
        actionName = Trim$(CStr(cb.OnAction))
        bangPos = InStrRev(actionName, "!")
        If bangPos > 0 Then actionName = Mid$(actionName, bangPos + 1)

        If Left$(cb.Name, 8) = "SITS_SEL" Or _
           StrComp(actionName, "SITSSelectorCheckboxChanged", vbTextCompare) = 0 Then
            ' Free SITS_SEL_* names so rows can reclaim them without collisions.
            cb.Name = "SITS_POOL_" & CStr(pool.Count + 1)
            pool.Add cb
        End If
    Next i
    On Error GoTo 0

    leftPos = ws.Cells(firstDataRow, SITS_SELECTOR_COL).Left
    widthPos = ws.Cells(firstDataRow, SITS_SELECTOR_COL).Width
    heightPos = ws.Cells(firstDataRow, SITS_SELECTOR_COL).Height
    poolIndex = 1

    For rowIndex = firstDataRow To lastDataRow
        Set linkCell = ws.Cells(rowIndex, SITS_SELECTOR_COL)
        cellText = UCase$(Trim$(CStr(linkCell.Value)))

        ' Tasks already in a work package show IN WORK — no checkbox.
        If StrComp(cellText, SITS_IN_WORK_TEXT, vbTextCompare) = 0 Then
            linkCell.NumberFormat = FORMAT_GENERAL
            linkCell.HorizontalAlignment = xlCenter
            linkCell.Font.Bold = True
            linkCell.Font.Color = RGB(192, 0, 0)
            GoTo NextSyncRow
        End If

        If poolIndex <= pool.Count Then
            Set cb = pool(poolIndex)
            poolIndex = poolIndex + 1
            cb.Left = leftPos
            cb.Top = linkCell.Top
            cb.Width = widthPos
            cb.Height = heightPos
        Else
            Set cb = ws.CheckBoxes.Add(leftPos, linkCell.Top, widthPos, heightPos)
        End If

        linkCell.Value = False
        linkCell.NumberFormat = SITS_CHECKBOX_HIDE_FORMAT
        linkCell.Font.Bold = False
        linkCell.Font.Color = vbBlack
        linkCell.Locked = False

        cb.Name = SITSCheckboxName(rowIndex)
        cb.LinkedCell = "'" & ws.Name & "'!" & linkCell.Address(False, False)
        cb.OnAction = "SITSSelectorCheckboxChanged"
        cb.Caption = ""
        cb.Placement = xlMove
        cb.Display3DShading = False
        cb.Locked = False
NextSyncRow:
    Next rowIndex

    ' Delete unused pooled controls (fewer visible rows, or more IN WORK rows).
    On Error Resume Next
    For i = pool.Count To poolIndex Step -1
        pool(i).Delete
    Next i
    On Error GoTo 0
End Sub

Private Function SITSCheckboxName(ByVal viewRow As Long) As String
    SITSCheckboxName = "SITS_SEL_" & CStr(viewRow)
End Function

Private Function BuildSITSViewRowIdMap(ByVal wsView As Worksheet) As Object
    Dim map As Object
    Dim lastRow As Long
    Dim colRowIdView As Long
    Dim headerCell As Range
    Dim viewIds As Variant
    Dim i As Long
    Dim rowId As String

    Set map = CreateObject("Scripting.Dictionary")
    map.CompareMode = 1
    Set BuildSITSViewRowIdMap = map

    If wsView Is Nothing Then Exit Function

    colRowIdView = 0
    For Each headerCell In wsView.Range(wsView.Cells(SITS_VIEW_HEADER_ROW, 1), _
                                        wsView.Cells(SITS_VIEW_HEADER_ROW, 40))
        If StrComp(Trim$(CStr(headerCell.Value)), HDR_SITS_ROW_ID, vbTextCompare) = 0 Then
            colRowIdView = headerCell.Column
            Exit For
        End If
    Next headerCell
    If colRowIdView = 0 Then Exit Function

    lastRow = wsView.Cells(wsView.Rows.Count, colRowIdView).End(xlUp).Row
    If lastRow < SITS_VIEW_DATA_START_ROW Then Exit Function

    viewIds = wsView.Range(wsView.Cells(SITS_VIEW_DATA_START_ROW, colRowIdView), _
                           wsView.Cells(lastRow, colRowIdView)).Value

    If Not IsArray(viewIds) Then
        rowId = Trim$(CStr(viewIds))
        If Len(rowId) > 0 Then map(rowId) = SITS_VIEW_DATA_START_ROW
        Exit Function
    End If

    For i = LBound(viewIds, 1) To UBound(viewIds, 1)
        rowId = Trim$(CStr(viewIds(i, 1)))
        If Len(rowId) > 0 Then map(rowId) = SITS_VIEW_DATA_START_ROW + i - 1
    Next i
End Function

Private Sub RemapSITSViewSelectorLinkedCells(ByVal ws As Worksheet)
    Dim lastRow As Long
    Dim rowIndex As Long
    Dim cb As CheckBox
    Dim i As Long
    Dim colRowIdView As Long
    Dim headerCell As Range
    Dim linkCell As Range
    Dim cellText As String
    Dim firstTop As Double
    Dim rowHeightPx As Double
    Dim estimatedRow As Long

    colRowIdView = 0
    For Each headerCell In ws.Range(ws.Cells(SITS_VIEW_HEADER_ROW, 1), _
                                    ws.Cells(SITS_VIEW_HEADER_ROW, 40))
        If StrComp(Trim$(CStr(headerCell.Value)), HDR_SITS_ROW_ID, vbTextCompare) = 0 Then
            colRowIdView = headerCell.Column
            Exit For
        End If
    Next headerCell

    If colRowIdView > 0 Then
        lastRow = ws.Cells(ws.Rows.Count, colRowIdView).End(xlUp).Row
    Else
        lastRow = ws.Cells(ws.Rows.Count, SITS_SELECTOR_COL).End(xlUp).Row
    End If
    If lastRow < SITS_VIEW_DATA_START_ROW Then Exit Sub

    firstTop = ws.Cells(SITS_VIEW_DATA_START_ROW, SITS_SELECTOR_COL).Top
    rowHeightPx = ws.Cells(SITS_VIEW_DATA_START_ROW, SITS_SELECTOR_COL).Height
    If rowHeightPx <= 0# Then rowHeightPx = 15#

    ' xlMove keeps checkbox Top aligned with its row after deletes.
    ' Remap LinkedCell/Name only — do not recreate hundreds of controls.
    On Error Resume Next
    For i = ws.CheckBoxes.Count To 1 Step -1
        Set cb = ws.CheckBoxes(i)

        estimatedRow = SITS_VIEW_DATA_START_ROW + CLng((cb.Top - firstTop) / rowHeightPx)
        If estimatedRow < SITS_VIEW_DATA_START_ROW Then estimatedRow = SITS_VIEW_DATA_START_ROW
        If estimatedRow > lastRow Then estimatedRow = lastRow

        ' Nudge one row either side if Top is between rows.
        rowIndex = estimatedRow
        If Abs(cb.Top - ws.Cells(estimatedRow, SITS_SELECTOR_COL).Top) > (rowHeightPx / 2#) Then
            If estimatedRow < lastRow Then
                If Abs(cb.Top - ws.Cells(estimatedRow + 1, SITS_SELECTOR_COL).Top) < _
                   Abs(cb.Top - ws.Cells(estimatedRow, SITS_SELECTOR_COL).Top) Then
                    rowIndex = estimatedRow + 1
                End If
            End If
            If estimatedRow > SITS_VIEW_DATA_START_ROW Then
                If Abs(cb.Top - ws.Cells(estimatedRow - 1, SITS_SELECTOR_COL).Top) < _
                   Abs(cb.Top - ws.Cells(rowIndex, SITS_SELECTOR_COL).Top) Then
                    rowIndex = estimatedRow - 1
                End If
            End If
        End If

        Set linkCell = ws.Cells(rowIndex, SITS_SELECTOR_COL)
        cellText = UCase$(Trim$(CStr(linkCell.Value)))
        If StrComp(cellText, SITS_IN_WORK_TEXT, vbTextCompare) = 0 Then
            cb.Delete
            linkCell.NumberFormat = FORMAT_GENERAL
            linkCell.HorizontalAlignment = xlCenter
            linkCell.Font.Bold = True
            linkCell.Font.Color = RGB(192, 0, 0)
        Else
            cb.Name = SITSCheckboxName(rowIndex)
            cb.LinkedCell = "'" & ws.Name & "'!" & linkCell.Address(False, False)
            cb.OnAction = "SITSSelectorCheckboxChanged"
            cb.Locked = False
            If Not (VarType(linkCell.Value) = vbBoolean Or IsNumeric(linkCell.Value)) Then
                linkCell.Value = False
            End If
            linkCell.NumberFormat = SITS_CHECKBOX_HIDE_FORMAT
            linkCell.Font.Bold = False
            linkCell.Font.Color = vbBlack
            linkCell.Locked = False
            ApplySITSViewRowSelectionHighlight ws, rowIndex, ConvertToBoolean(linkCell.Value)
        End If
    Next i
    On Error GoTo 0
End Sub

    ' Kept as a full rebuild fallback (manual repair / rare paths).
    Dim lastRow As Long
    Dim rowIndex As Long
    Dim linkCell As Range

    Dim i As Long
    Dim colRowIdView As Long
    Dim headerCell As Range

    colRowIdView = 0

                                    ws.Cells(SITS_VIEW_HEADER_ROW, 40))

Private Sub SetSITSViewSelectorCheckbox(ByVal ws As Worksheet, ByVal viewRow As Long)
    Dim linkCell As Range
    Dim cb As CheckBox

    DeleteSITSViewCheckboxOnRow ws, viewRow
    Set linkCell = ws.Cells(viewRow, SITS_SELECTOR_COL)
    linkCell.Value = False
    linkCell.NumberFormat = SITS_CHECKBOX_HIDE_FORMAT
    linkCell.Font.Bold = False
    linkCell.Font.Color = vbBlack

    linkCell.Locked = False
    Set cb = ws.CheckBoxes.Add(linkCell.Left, linkCell.Top, linkCell.Width, linkCell.Height)
    cb.Name = SITSCheckboxName(viewRow)
    cb.LinkedCell = "'" & ws.Name & "'!" & linkCell.Address(False, False)
    cb.OnAction = "SITSSelectorCheckboxChanged"
    cb.Caption = ""
    cb.Placement = xlMove
    cb.Display3DShading = False
    cb.Locked = False
End Sub

Private Sub DeleteSITSViewCheckboxOnRow(ByVal ws As Worksheet, ByVal viewRow As Long)
    Dim cb As CheckBox
    Dim i As Long
    Dim linkAddr As String
    Dim targetAddr As String

    On Error Resume Next
    ws.CheckBoxes(SITSCheckboxName(viewRow)).Delete
    On Error GoTo 0

    ' Fallback: match by linked cell in case the control was renamed.
    targetAddr = UCase$("'" & ws.Name & "'!" & ws.Cells(viewRow, SITS_SELECTOR_COL).Address(False, False))
    On Error Resume Next
    For i = ws.CheckBoxes.Count To 1 Step -1
        Set cb = ws.CheckBoxes(i)
        linkAddr = UCase$(Replace(CStr(cb.LinkedCell), "$", ""))
        If StrComp(linkAddr, targetAddr, vbTextCompare) = 0 Then
            cb.Delete
        ElseIf InStr(1, linkAddr, ws.Cells(viewRow, SITS_SELECTOR_COL).Address(False, False), vbTextCompare) > 0 Then
            cb.Delete
        End If
    Next i
    On Error GoTo 0
End Sub

Private Sub RewriteSITSViewDataRow(ByVal wsView As Worksheet, _
                                   ByVal viewRow As Long, _
                                   ByVal sitsRow As ListRow, _
                                   ByVal tbl As ListObject, _
                                   ByVal intervalCode As String, _
                                   ByVal viewColumns As Collection, _
                                   Optional ByVal colMap As Object = Nothing)
    Dim columnName As Variant
    Dim outputCol As Long

    If colMap Is Nothing Then Set colMap = BuildTableColumnIndexMap(tbl)

    outputCol = 1
    For Each columnName In viewColumns
        ' Selector handled separately by caller.

            WriteSITSViewCell sitsRow, tbl, intervalCode, CStr(columnName), _
                wsView.Cells(viewRow, outputCol), colMap
        End If
        outputCol = outputCol + 1
    Next columnName
End Sub

Private Sub SortLongCollectionDescending(ByRef values As Collection)
    Dim i As Long
    Dim j As Long
    Dim arr() As Long
    Dim tmp As Long
    Dim n As Long

    If values Is Nothing Then Exit Sub
    n = values.Count
    If n < 2 Then Exit Sub

    ReDim arr(1 To n)
    For i = 1 To n
        arr(i) = CLng(values(i))
    Next i

    For i = 1 To n - 1
        For j = i + 1 To n
            If arr(j) > arr(i) Then
                tmp = arr(i)
                arr(i) = arr(j)
                arr(j) = tmp
            End If
        Next j
    Next i

    Do While values.Count > 0
        values.Remove 1
    Loop
    For i = 1 To n
        values.Add arr(i)
    Next i
End Sub

        .VerticalAlignment = xlTop

        .WrapText = False

    ApplyColumnTextAlign ws.Range(ws.Cells(SITS_VIEW_HEADER_ROW, 2), ws.Cells(layoutLastRow, 2)), xlCenter
    ApplyColumnTextAlign ws.Range(ws.Cells(SITS_VIEW_HEADER_ROW, 3), ws.Cells(layoutLastRow, 3)), xlCenter
    ApplyColumnTextAlign ws.Range(ws.Cells(SITS_VIEW_HEADER_ROW, 4), ws.Cells(layoutLastRow, 4)), xlCenter
    ApplyColumnTextAlign ws.Range(ws.Cells(SITS_VIEW_HEADER_ROW, 5), ws.Cells(layoutLastRow, 5)), xlCenter
    ApplyColumnTextAlign ws.Range(ws.Cells(SITS_VIEW_HEADER_ROW, 6), ws.Cells(layoutLastRow, 6)), xlLeft

    ApplyColumnTextAlign ws.Range(ws.Cells(SITS_VIEW_HEADER_ROW, 8), ws.Cells(layoutLastRow, 8)), xlLeft

        ws.Columns(col).Hidden = False

Private Function SITSViewUsesHourFormat(ByVal intervalCode As String) As Boolean
    Select Case UCase$(Trim$(intervalCode))
        Case TYPE_HH, TYPE_E1, TYPE_E2
            SITSViewUsesHourFormat = True
    End Select
End Function

' Pure value/format builder for bulk view writes and single-cell writers.
Private Sub BuildSITSViewCellPayload(ByVal sourceRow As ListRow, _
                                     ByVal tbl As ListObject, _
                                     ByVal intervalCode As String, _
                                     ByVal columnName As String, _
                                     ByVal colMap As Object, _
                                     ByRef outValue As Variant, _
                                     ByRef outFormat As String, _
                                     ByRef outComment As String)
    Dim sourceCell As Range
    Dim colIndex As Long
    Dim value As Variant

    outValue = Empty
    outFormat = FORMAT_GENERAL
    outComment = ""

    If StrComp(columnName, HDR_VIEW_LIFE_REMAINING, vbTextCompare) = 0 Then
        BuildSITSViewLifeRemainingPayload sourceRow, tbl, intervalCode, colMap, outValue, outFormat
        Exit Sub
    End If

    If StrComp(columnName, HDR_SELECTOR, vbTextCompare) = 0 Then
        BuildSITSViewSelectorPayload sourceRow, tbl, colMap, outValue, outFormat
        Exit Sub
    End If

    colIndex = ResolveTableColumnIndex(tbl, columnName, colMap)
    If colIndex = 0 Then Exit Sub
    Set sourceCell = sourceRow.Range.Cells(1, colIndex)

    If StrComp(columnName, HDR_SERIAL, vbTextCompare) = 0 Or _
       StrComp(columnName, HDR_END_ITEM_SERIAL, vbTextCompare) = 0 Then
        outValue = NormalizeSerialText(sourceCell.Value, sourceCell.Text)
        outFormat = FORMAT_TEXT
        Exit Sub
    End If

    If StrComp(columnName, HDR_ACTUAL_INTERVAL, vbTextCompare) = 0 Or _
       StrComp(columnName, HDR_EXTENSION, vbTextCompare) = 0 Or _
       StrComp(columnName, HDR_WARNING_INTERVAL, vbTextCompare) = 0 Then
        BuildSITSViewIntervalPayload sourceCell, intervalCode, outValue, outFormat
        Exit Sub
    End If

    If StrComp(columnName, HDR_CURRENT_USAGE_HOURS, vbTextCompare) = 0 Then
        If SITSViewUsesHourFormat(intervalCode) Then
            BuildSITSViewIntervalPayload sourceCell, intervalCode, outValue, outFormat
        End If
        Exit Sub
    End If

    If StrComp(columnName, HDR_CURRENT_USAGE_NUMBER, vbTextCompare) = 0 Then
        If SITSViewUsesHourFormat(intervalCode) Then
            Exit Sub
        ElseIf IsEmpty(sourceCell.Value) Or Len(Trim$(CStr(sourceCell.Value))) = 0 Then
            Exit Sub
        Else
            outValue = CDbl(NzSITSNumber(sourceCell.Value))
            outFormat = FORMAT_DECIMAL
        End If
        Exit Sub
    End If

    If StrComp(columnName, HDR_PROJECTED_DUE_DATE, vbTextCompare) = 0 Then
        value = sourceCell.Value
        If IsDate(value) Then
            outValue = CDate(value)
            outFormat = FORMAT_DATE
        End If
        Exit Sub
    End If

    If StrComp(columnName, HDR_TASK_DESCRIPTION, vbTextCompare) = 0 Then
        outValue = sourceCell.Value
        outFormat = FORMAT_GENERAL
        outComment = GetCellCommentText(sourceCell)
        Exit Sub
    End If

    outValue = sourceCell.Value
    outFormat = FORMAT_GENERAL
End Sub

Private Sub BuildSITSViewCellPayloadFromData(ByRef data As Variant, _
                                             ByVal rowIndex As Long, _
                                             ByVal intervalCode As String, _
                                             ByVal columnName As String, _
                                             ByVal colMap As Object, _
                                             ByVal colOrn As Long, _
                                             ByRef outValue As Variant, _
                                             ByRef outFormat As String, _
                                             ByRef outComment As String)
    Dim colIndex As Long
    Dim value As Variant

    outValue = Empty
    outFormat = FORMAT_GENERAL
    outComment = ""

    If StrComp(columnName, HDR_VIEW_LIFE_REMAINING, vbTextCompare) = 0 Then
        BuildSITSViewLifeRemainingPayloadFromData data, rowIndex, intervalCode, colMap, outValue, outFormat
        Exit Sub
    End If

    If StrComp(columnName, HDR_SELECTOR, vbTextCompare) = 0 Then
        If colOrn > 0 Then
            If Len(Trim$(CStr(data(rowIndex, colOrn)))) > 0 Then
                outValue = SITS_IN_WORK_TEXT
                outFormat = FORMAT_GENERAL
                Exit Sub
            End If
        End If
        outValue = False
        outFormat = SITS_CHECKBOX_HIDE_FORMAT
        Exit Sub
    End If

    colIndex = LookupColumnIndex(colMap, columnName)
    If colIndex = 0 Then Exit Sub
    value = data(rowIndex, colIndex)

    If StrComp(columnName, HDR_SERIAL, vbTextCompare) = 0 Or _
       StrComp(columnName, HDR_END_ITEM_SERIAL, vbTextCompare) = 0 Then
        outValue = NormalizeSerialText(value)
        outFormat = FORMAT_TEXT
        Exit Sub
    End If

    If StrComp(columnName, HDR_ACTUAL_INTERVAL, vbTextCompare) = 0 Or _
       StrComp(columnName, HDR_EXTENSION, vbTextCompare) = 0 Or _
       StrComp(columnName, HDR_WARNING_INTERVAL, vbTextCompare) = 0 Then
        BuildSITSViewIntervalPayloadFromValue value, intervalCode, outValue, outFormat
        Exit Sub
    End If

    If StrComp(columnName, HDR_CURRENT_USAGE_HOURS, vbTextCompare) = 0 Then
        If SITSViewUsesHourFormat(intervalCode) Then
            BuildSITSViewIntervalPayloadFromValue value, intervalCode, outValue, outFormat
        End If
        Exit Sub
    End If

    If StrComp(columnName, HDR_CURRENT_USAGE_NUMBER, vbTextCompare) = 0 Then
        If SITSViewUsesHourFormat(intervalCode) Then
            Exit Sub
        ElseIf IsEmpty(value) Or Len(Trim$(CStr(value))) = 0 Then
            Exit Sub
        Else
            outValue = CDbl(NzSITSNumber(value))
            outFormat = FORMAT_DECIMAL
        End If
        Exit Sub
    End If

    If StrComp(columnName, HDR_PROJECTED_DUE_DATE, vbTextCompare) = 0 Then
        If IsDate(value) Then
            outValue = CDate(value)
            outFormat = FORMAT_DATE
        End If
        Exit Sub
    End If

    If StrComp(columnName, HDR_TASK_DESCRIPTION, vbTextCompare) = 0 Then
        outValue = value
        outFormat = FORMAT_GENERAL
        Exit Sub
    End If

    outValue = value
    outFormat = FORMAT_GENERAL
End Sub

Private Sub BuildSITSViewLifeRemainingPayloadFromData(ByRef data As Variant, _
                                                      ByVal rowIndex As Long, _
                                                      ByVal intervalCode As String, _
                                                      ByVal colMap As Object, _
                                                      ByRef outValue As Variant, _
                                                      ByRef outFormat As String)
    Dim colIndex As Long
    Dim value As Variant

    outValue = Empty
    outFormat = FORMAT_GENERAL

    If SITSViewUsesHourFormat(intervalCode) Then
        colIndex = LookupColumnIndex(colMap, HDR_HOURS_LIFE_REMAINING)
        If colIndex = 0 Then Exit Sub
        value = data(rowIndex, colIndex)
        If IsEmpty(value) Or Len(Trim$(CStr(value))) = 0 Then
            Exit Sub
        ElseIf IsNumeric(value) And CDbl(value) < 0# Then
            outValue = FormatElapsedHoursFromDayFraction(CDbl(value))
            outFormat = FORMAT_GENERAL
        Else
            BuildSITSViewIntervalPayloadFromValue value, intervalCode, outValue, outFormat
        End If
        Exit Sub
    End If

    If intervalCode = TYPE_DD Then
        colIndex = LookupColumnIndex(colMap, HDR_DAYS_WHEN_DUE)
        If colIndex = 0 Then Exit Sub
        value = data(rowIndex, colIndex)
        If IsEmpty(value) Or Len(Trim$(CStr(value))) = 0 Then
            Exit Sub
        ElseIf LooksLikeAbsoluteDueDate(value) Then
            outValue = CDbl(DateDiff("d", Date, CDate(value)))
            outFormat = FORMAT_DECIMAL
        ElseIf IsNumeric(value) Then
            outValue = CDbl(value)
            outFormat = FORMAT_DECIMAL
        End If
        Exit Sub
    End If

    colIndex = LookupColumnIndex(colMap, HDR_NUMBER_LIFE_REMAINING)
    If colIndex = 0 Then Exit Sub
    value = data(rowIndex, colIndex)
    If IsEmpty(value) Or Len(Trim$(CStr(value))) = 0 Then Exit Sub
    outValue = CDbl(NzSITSNumber(value))
    outFormat = FORMAT_DECIMAL
End Sub

Private Sub BuildSITSViewIntervalPayloadFromValue(ByVal value As Variant, _
                                                  ByVal intervalCode As String, _
                                                  ByRef outValue As Variant, _
                                                  ByRef outFormat As String)
    Dim storedValue As Double

    outValue = Empty
    outFormat = FORMAT_GENERAL

    If IsEmpty(value) Or Len(Trim$(CStr(value))) = 0 Then Exit Sub

    If SITSViewUsesHourFormat(intervalCode) Then
        storedValue = CoerceUsageStorageValue(value, FORMAT_TYPE_HOURS)
        outValue = storedValue
        outFormat = UsageNumberFormatPattern(FORMAT_TYPE_HOURS, storedValue)
    Else
        outValue = CDbl(NzSITSNumber(value))
        outFormat = FORMAT_DECIMAL
    End If
End Sub

Private Sub SaveForecastSettingsFromView()
    Dim ws As Worksheet
    Dim hoursValue As Variant
    Dim daysValue As Variant
    Dim numberValue As Variant

    Set ws = GetWorksheet(SHEET_SITS, showError:=False)
    If ws Is Nothing Then Exit Sub

    ' Empty cells are IsNumeric in VBA and coerce to 0 — ignore blanks so a
    ' mid-refresh clear cannot overwrite the real config with zeros.
    hoursValue = ws.Range(SITS_FORECAST_HOURS_CELL).Value
    If IsPresentNumericSetting(hoursValue) Then
        SetConfigValue CFG_FORECAST_HOURS, CDbl(hoursValue)
    End If

    daysValue = ws.Range(SITS_FORECAST_DAYS_CELL).Value
    If IsPresentNumericSetting(daysValue) Then
        SetConfigValue CFG_FORECAST_DAYS, CLng(daysValue)
    End If

    numberValue = ws.Range(SITS_FORECAST_NUMBER_CELL).Value
    If IsPresentNumericSetting(numberValue) Then
        SetConfigValue CFG_FORECAST_NUMBER, CLng(numberValue)
    End If
End Sub

Public Function GetForecastHoursSetting() As Double
    Dim value As Variant

    value = GetConfigValue(CFG_FORECAST_HOURS, 50)
    If IsPresentNumericSetting(value) Then
        GetForecastHoursSetting = CDbl(value)
    Else
        GetForecastHoursSetting = 50#
    End If
End Function

Public Function GetForecastDaysSetting() As Long
    Dim value As Variant

    value = GetConfigValue(CFG_FORECAST_DAYS, 90)
    If IsPresentNumericSetting(value) Then
        GetForecastDaysSetting = CLng(value)
    Else
        GetForecastDaysSetting = 90
    End If
End Function

Public Function GetForecastNumberSetting() As Long
    Dim value As Variant

    value = GetConfigValue(CFG_FORECAST_NUMBER, 500)
    If IsPresentNumericSetting(value) Then
        GetForecastNumberSetting = CLng(value)
    Else
        GetForecastNumberSetting = 500
    End If
End Function

' True only for a real entered number — Empty/blank must not count (VBA
' IsNumeric(Empty) is True and CDbl(Empty)=0, which corrupted B3/B4).
' Note: 0 itself is a valid setting (show all tasks).
Private Function IsPresentNumericSetting(ByVal value As Variant) As Boolean
    If IsError(value) Then Exit Function
    If IsEmpty(value) Then Exit Function
    If IsNull(value) Then Exit Function
    If VarType(value) = vbString Then
        If Len(Trim$(CStr(value))) = 0 Then Exit Function
    End If
    IsPresentNumericSetting = IsNumeric(value)
End Function

' Seed missing forecast config to defaults (50 hours / 90 days / 500 number).
' Does not overwrite an intentional 0 (show-all) value.
Public Sub EnsureForecastConfigDefaults()
    Dim hoursValue As Variant
    Dim daysValue As Variant
    Dim numberValue As Variant

    hoursValue = GetConfigValue(CFG_FORECAST_HOURS, Empty)
    If Not IsPresentNumericSetting(hoursValue) Then
        SetConfigValue CFG_FORECAST_HOURS, 50
    End If

    daysValue = GetConfigValue(CFG_FORECAST_DAYS, Empty)
    If Not IsPresentNumericSetting(daysValue) Then
        SetConfigValue CFG_FORECAST_DAYS, 90
    End If

    numberValue = GetConfigValue(CFG_FORECAST_NUMBER, Empty)
    If Not IsPresentNumericSetting(numberValue) Then
        SetConfigValue CFG_FORECAST_NUMBER, 500
    End If
End Sub

Private Function ParseWarnHoursSetting() As Double
    Dim rawValue As Variant
    Dim parsed As Variant

    rawValue = GetConfigValue(CFG_WARN_HOURS, "10:00")
    parsed = ParseTimeUsageValue(rawValue, CStr(rawValue))
    If IsEmpty(parsed) Then
        ParseWarnHoursSetting = 10# / 24#
    Else
        ParseWarnHoursSetting = CDbl(parsed)
    End If
End Function

Private Sub ApplySITSViewRowWarnings(ByVal ws As Worksheet, _
                                     ByVal rowIndex As Long, _
                                     ByVal colCount As Long, _
                                     ByVal intervalCode As String, _
                                     ByVal lifeHoursFraction As Double, _
                                     ByVal daysUntilDue As Long, _
                                     ByVal warnHoursFraction As Double, _
                                     ByVal warnDays As Long, _
                                     ByVal sourceRow As ListRow, _
                                     ByVal colManualFlag As Long)
    Dim warnColor As Long
    Dim rowRange As Range

    Set rowRange = ws.Range(ws.Cells(rowIndex, 1), ws.Cells(rowIndex, colCount))

End Sub

' Light-blue selection highlight while a Selector checkbox is ticked.
' Deselect restores due-soon / overdue / manual-update colours (or clears).
Private Sub ApplySITSViewRowSelectionHighlight(ByVal ws As Worksheet, _
                                               ByVal viewRow As Long, _
                                               ByVal selected As Boolean)
    Dim rowRange As Range
    Dim wsData As Worksheet
    Dim tbl As ListObject
    Dim sitsRow As ListRow
    Dim colRowIdView As Long
    Dim colInterval As Long
    Dim colManualFlag As Long
    Dim headerCell As Range
    Dim rowId As String
    Dim intervalCode As String
    Dim warnColor As Long
    Dim warnHoursFraction As Double
    Dim warnDays As Long
    Dim sitsIndex As Object

    If ws Is Nothing Or viewRow < SITS_VIEW_DATA_START_ROW Then Exit Sub

    Set rowRange = ws.Range(ws.Cells(viewRow, 1), ws.Cells(viewRow, SITS_VISIBLE_COLUMN_COUNT))

    If selected Then
        rowRange.Interior.Color = COLOR_SELECTED_BLUE
        Exit Sub
    End If

    ' Deselected — clear selection colour, then restore any warning colour.
    rowRange.Interior.ColorIndex = xlNone

    colRowIdView = 0
    For Each headerCell In ws.Range(ws.Cells(SITS_VIEW_HEADER_ROW, 1), _
                                    ws.Cells(SITS_VIEW_HEADER_ROW, 40))
        If StrComp(Trim$(CStr(headerCell.Value)), HDR_SITS_ROW_ID, vbTextCompare) = 0 Then
            colRowIdView = headerCell.Column
            Exit For
        End If
    Next headerCell
    If colRowIdView = 0 Then Exit Sub

    rowId = Trim$(CStr(ws.Cells(viewRow, colRowIdView).Value))
    If Len(rowId) = 0 Then Exit Sub

    Set wsData = GetWorksheet(SHEET_DATA_SITS, showError:=False)
    If wsData Is Nothing Then Exit Sub
    Set tbl = GetTable(wsData, TBL_SITS, showError:=False)
    If tbl Is Nothing Then Exit Sub

    Set sitsIndex = BuildSITSRowIdIndex(tbl)
    Set sitsRow = ResolveSITSRowById(sitsIndex, tbl, rowId)
    If sitsRow Is Nothing Then Exit Sub

    colInterval = GetTableColumnIndex(tbl, HDR_INTERVAL_TYPE_CODE)
    colManualFlag = GetTableColumnIndex(tbl, HDR_NEEDS_MANUAL_UPDATE)
    If colInterval = 0 Then Exit Sub

    intervalCode = UCase$(Trim$(CStr(sitsRow.Range.Cells(1, colInterval).Value)))
    warnHoursFraction = ParseWarnHoursSetting()
    warnDays = CLng(NzSITSNumber(GetConfigValue(CFG_WARN_DAYS, 7)))
    warnColor = ResolveSITSViewWarningColor(intervalCode, warnHoursFraction, warnDays, _
                                           sitsRow, colManualFlag)
    If warnColor <> 0 Then
        rowRange.Interior.Color = warnColor
    End If
End Sub

Private Function ResolveSITSViewWarningColor(ByVal intervalCode As String, _
                                             ByVal warnHoursFraction As Double, _
                                             ByVal warnDays As Long, _
                                             ByVal sourceRow As ListRow, _
                                             ByVal colManualFlag As Long) As Long
    Dim lifeValue As Double
    Dim hasLife As Boolean

    ResolveSITSViewWarningColor = 0

    ' Manual-update flag takes priority over due-soon / overdue colours.
    If colManualFlag > 0 Then
        If ConvertToBoolean(sourceRow.Range.Cells(1, colManualFlag).Value) Then
            ResolveSITSViewWarningColor = COLOR_MANUAL_BLUE
            Exit Function
        End If
    End If

    hasLife = TryGetWarningLifeValue(sourceRow, intervalCode, lifeValue)
    If Not hasLife Then Exit Function

    Select Case intervalCode
        Case TYPE_HH, TYPE_E1, TYPE_E2
            If lifeValue < 0# Then
                ResolveSITSViewWarningColor = COLOR_WARN_PINK
            ElseIf lifeValue <= warnHoursFraction Then
                ResolveSITSViewWarningColor = COLOR_WARN_AMBER
            End If
        Case TYPE_DD
            If lifeValue < 0# Then
                ResolveSITSViewWarningColor = COLOR_WARN_PINK
            ElseIf lifeValue <= CDbl(warnDays) Then
                ResolveSITSViewWarningColor = COLOR_WARN_AMBER
            End If
    End Select
End Function

' Array-path warning colour — avoids ListRow COM during full refresh.
Private Function ResolveSITSViewWarningColorValues(ByVal intervalCode As String, _
                                                   ByVal warnHoursFraction As Double, _
                                                   ByVal warnDays As Long, _
                                                   ByVal isManual As Boolean, _
                                                   ByRef data As Variant, _
                                                   ByVal rowIndex As Long, _
                                                   ByVal colLRHours As Long, _
                                                   ByVal colDaysDue As Long) As Long
    Dim lifeValue As Double
    Dim hasLife As Boolean
    Dim rawValue As Variant

    ResolveSITSViewWarningColorValues = 0

    If isManual Then
        ResolveSITSViewWarningColorValues = COLOR_MANUAL_BLUE
        Exit Function
    End If

    hasLife = False
    Select Case intervalCode
        Case TYPE_HH, TYPE_E1, TYPE_E2
            If colLRHours > 0 Then
                rawValue = data(rowIndex, colLRHours)
                If Not IsEmpty(rawValue) And Len(Trim$(CStr(rawValue))) > 0 And IsNumeric(rawValue) Then
                    lifeValue = CDbl(rawValue)
                    hasLife = True
                End If
            End If
            If hasLife Then
                If lifeValue < 0# Then
                    ResolveSITSViewWarningColorValues = COLOR_WARN_PINK
                ElseIf lifeValue <= warnHoursFraction Then
                    ResolveSITSViewWarningColorValues = COLOR_WARN_AMBER
                End If
            End If

        Case TYPE_DD
            If colDaysDue > 0 Then
                rawValue = data(rowIndex, colDaysDue)
                If Not IsEmpty(rawValue) And Len(Trim$(CStr(rawValue))) > 0 Then
                    If LooksLikeAbsoluteDueDate(rawValue) Then
                        lifeValue = CDbl(DateDiff("d", Date, CDate(rawValue)))
                        hasLife = True
                    ElseIf IsNumeric(rawValue) Then
                        lifeValue = CDbl(rawValue)
                        hasLife = True
                    End If
                End If
            End If
            If hasLife Then
                If lifeValue < 0# Then
                    ResolveSITSViewWarningColorValues = COLOR_WARN_PINK
                ElseIf lifeValue <= CDbl(warnDays) Then
                    ResolveSITSViewWarningColorValues = COLOR_WARN_AMBER
                End If
            End If
    End Select
End Function

Private Function TryGetWarningLifeValue(ByVal sourceRow As ListRow, _
                                        ByVal intervalCode As String, _
                                        ByRef lifeValue As Double) As Boolean
    Dim tbl As ListObject
    Dim colIndex As Long
    Dim value As Variant

    Set tbl = sourceRow.Parent
    lifeValue = 0#

    Select Case intervalCode
        Case TYPE_HH, TYPE_E1, TYPE_E2
            colIndex = GetTableColumnIndex(tbl, HDR_HOURS_LIFE_REMAINING)
            If colIndex = 0 Then Exit Function
            value = sourceRow.Range.Cells(1, colIndex).Value
            If IsEmpty(value) Or Len(Trim$(CStr(value))) = 0 Or Not IsNumeric(value) Then Exit Function
            lifeValue = CDbl(value)
            TryGetWarningLifeValue = True

        Case TYPE_DD
            colIndex = GetTableColumnIndex(tbl, HDR_DAYS_WHEN_DUE)
            If colIndex = 0 Then Exit Function
            value = sourceRow.Range.Cells(1, colIndex).Value
            If IsEmpty(value) Or Len(Trim$(CStr(value))) = 0 Then Exit Function
            If LooksLikeAbsoluteDueDate(value) Then
                lifeValue = CDbl(DateDiff("d", Date, CDate(value)))
                TryGetWarningLifeValue = True
            ElseIf IsNumeric(value) Then
                lifeValue = CDbl(value)
                TryGetWarningLifeValue = True
            End If
    End Select
End Function

' Dashboard status counts — same amber/pink rules as SITS view highlighting.
' Manual-update tasks are excluded from both counts (same priority as the view).
Public Function CountWarningPeriodTasks() As Long
    CountWarningPeriodTasks = CountSITSTasksByWarnColor(COLOR_WARN_AMBER)
End Function

Public Function CountOverdueTasks() As Long
    CountOverdueTasks = CountSITSTasksByWarnColor(COLOR_WARN_PINK)
End Function

Private Function CountSITSTasksByWarnColor(ByVal targetColor As Long) As Long
    Dim ws As Worksheet
    Dim tbl As ListObject
    Dim row As ListRow
    Dim colInterval As Long
    Dim colManualFlag As Long
    Dim intervalCode As String
    Dim warnHoursFraction As Double
    Dim warnDays As Long
    Dim warnColor As Long
    Dim count As Long

    Set ws = GetWorksheet(SHEET_DATA_SITS, showError:=False)
    If ws Is Nothing Then Exit Function
    Set tbl = GetTable(ws, TBL_SITS, showError:=False)
    If tbl Is Nothing Or tbl.DataBodyRange Is Nothing Then Exit Function

    colInterval = GetTableColumnIndex(tbl, HDR_INTERVAL_TYPE_CODE)
    colManualFlag = GetTableColumnIndex(tbl, HDR_NEEDS_MANUAL_UPDATE)
    If colInterval = 0 Then Exit Function

    warnHoursFraction = ParseWarnHoursSetting()
    warnDays = CLng(NzSITSNumber(GetConfigValue(CFG_WARN_DAYS, 7)))

    For Each row In tbl.ListRows
        intervalCode = UCase$(Trim$(CStr(row.Range.Cells(1, colInterval).Value)))
        warnColor = ResolveSITSViewWarningColor(intervalCode, warnHoursFraction, warnDays, _
                                               row, colManualFlag)
        If warnColor = targetColor Then count = count + 1
    Next row

    CountSITSTasksByWarnColor = count
End Function

Private Function BuildUsageCurrentLookup() As Object
    Dim lookup As Object
    Dim ws As Worksheet
    Dim tbl As ListObject
    Dim row As ListRow
    Dim intervalCode As String

    Set lookup = CreateObject("Scripting.Dictionary")
    lookup.CompareMode = 1
    Set BuildUsageCurrentLookup = lookup

    Set ws = GetWorksheet(SHEET_DATA_USAGE_CURRENT, showError:=False)
    If ws Is Nothing Then Exit Function
    Set tbl = GetTable(ws, TBL_USAGE_CURRENT, showError:=False)
    If tbl Is Nothing Or tbl.DataBodyRange Is Nothing Then Exit Function

    For Each row In tbl.ListRows
        intervalCode = UCase$(Trim$(CStr(row.Range.Cells(1, 1).Value)))
        If Len(intervalCode) = 0 Then GoTo NextUsageLookup
        If lookup.Exists(intervalCode) Then GoTo NextUsageLookup
        lookup.Add intervalCode, Array(row.Range.Cells(1, 3).Value, _
                                       Trim$(CStr(row.Range.Cells(1, 2).Value)))
NextUsageLookup:
    Next row
End Function

Private Function BuildUsageSnapshotLookup() As Object
    Dim lookup As Object
    Dim ws As Worksheet
    Dim tbl As ListObject
    Dim row As ListRow
    Dim intervalCode As String

    Set lookup = CreateObject("Scripting.Dictionary")
    lookup.CompareMode = 1
    Set BuildUsageSnapshotLookup = lookup

    Set ws = GetWorksheet(SHEET_CONFIG, showError:=False)
    If ws Is Nothing Then Exit Function
    Set tbl = GetTable(ws, TBL_USAGE_SNAPSHOT, showError:=False)
    If tbl Is Nothing Or tbl.DataBodyRange Is Nothing Then Exit Function

    For Each row In tbl.ListRows
        intervalCode = UCase$(Trim$(CStr(row.Range.Cells(1, 1).Value)))
        If Len(intervalCode) = 0 Then GoTo NextSnapLookup
        If lookup.Exists(intervalCode) Then GoTo NextSnapLookup
        lookup.Add intervalCode, row.Range.Cells(1, 2).Value
NextSnapLookup:
    Next row
End Function

Private Function LookupDictUsage(ByVal lookup As Object, _
                                 ByVal intervalCode As String, _
                                 ByRef formatType As String) As Variant
    Dim entry As Variant

    If lookup Is Nothing Then Exit Function
    intervalCode = UCase$(Trim$(intervalCode))
    If Not lookup.Exists(intervalCode) Then Exit Function

    entry = lookup(intervalCode)
    formatType = ResolveUsageFormatType(intervalCode, CStr(entry(1)), formatType)
    LookupDictUsage = entry(0)
End Function

Private Function LookupDictSnapshot(ByVal lookup As Object, _
                                    ByVal intervalCode As String) As Variant
    If lookup Is Nothing Then Exit Function
    intervalCode = UCase$(Trim$(intervalCode))
    If Not lookup.Exists(intervalCode) Then Exit Function
    LookupDictSnapshot = lookup(intervalCode)
End Function

' Day-count remaining values are small; Excel date serials for modern dates are large.
' VBA IsDate(30) is True, so thresholding is required.
Private Function LooksLikeAbsoluteDueDate(ByVal value As Variant) As Boolean
    Const MIN_DATE_SERIAL As Double = 20000#  ' ~24/10/1954

    If IsEmpty(value) Or Len(Trim$(CStr(value))) = 0 Then Exit Function
    If VarType(value) = vbDate Then
        LooksLikeAbsoluteDueDate = True
    ElseIf IsNumeric(value) Then
        LooksLikeAbsoluteDueDate = (CDbl(value) >= MIN_DATE_SERIAL)
    End If
End Function

Private Sub WriteCurrentUsageHours(ByVal target As Range, _
                                   ByVal value As Variant, _
                                   ByVal formatType As String)

        target.ClearContents

    End If
    WriteFormattedUsageValue target, value, formatType

Private Sub WriteCurrentUsageNumber(ByVal target As Range, _
                                    ByVal value As Variant, _
                                    ByVal formatType As String)
    If IsMissingUsageValue(value) Then
        target.Value = "---"
        Exit Sub
    End If

    If StrComp(formatType, FORMAT_TYPE_NUMBER, vbTextCompare) = 0 Then
        target.Value = CDbl(NzSITSNumber(value))
        target.NumberFormat = FORMAT_DECIMAL
    Else
        WriteFormattedUsageValue target, value, formatType
    End If
End Sub

Private Sub EnsureTableColumn(ByVal tbl As ListObject, ByVal columnName As String)
    If GetTableColumnIndex(tbl, columnName) <> 0 Then Exit Sub

    SuspendWorkbookProtection
    On Error GoTo EnsureColFail
    tbl.ListColumns.Add
    tbl.ListColumns(tbl.ListColumns.Count).Name = columnName
    ResumeWorkbookProtection
    Exit Sub

EnsureColFail:
    ResumeWorkbookProtection
End Sub

Private Function IsEmptyBaselineValue(ByVal value As Variant) As Boolean
    If IsEmpty(value) Then
        IsEmptyBaselineValue = True
    ElseIf Len(Trim$(CStr(value))) = 0 Then
        IsEmptyBaselineValue = True
    ElseIf VarType(value) = vbString Then
        IsEmptyBaselineValue = (Trim$(CStr(value)) = "---")
    End If
End Function

Private Function IsMissingUsageValue(ByVal value As Variant) As Boolean
    If IsEmpty(value) Then
        IsMissingUsageValue = True
    ElseIf Len(Trim$(CStr(value))) = 0 Then
        IsMissingUsageValue = True
    End If
End Function

Private Function NzSITSNumber(ByVal value As Variant) As Double
    If IsEmpty(value) Or Len(Trim$(CStr(value))) = 0 Then
        NzSITSNumber = 0#
    ElseIf VarType(value) = vbString And Trim$(CStr(value)) = "---" Then
        NzSITSNumber = 0#
    Else
        NzSITSNumber = CDbl(value)
    End If
End Function
