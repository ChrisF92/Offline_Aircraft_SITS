Option Explicit

' ========================================================================
' SITS IMPORT MODULE — Offline Aircraft SITS v2
' ========================================================================
' Imports a GOLDesp Long Forecast export into tblSITS on _DATA_SITS.
' Strict header validation — aborts if any required column is missing.
' ========================================================================

Private Const OPT_PROJECTED_DUE As String = "Projected Due Date"
Private Const OPT_INTERVAL_TYPE As String = "Interval Type"
Private Const OPT_TASK_TYPE_CODE As String = "Task Type Code"
Private Const OPT_WARNING_INTERVAL As String = "Warning Interval"

Public Sub RunImportSITSFromFile()
    ImportSITSFromFile
End Sub

Public Function ImportSITSFromFile(Optional ByVal filePath As String = "", _
                                   Optional ByVal showDialogs As Boolean = True, _
                                   Optional ByVal skipOverwritePrompt As Boolean = False) As Long
    Const PROC_NAME As String = "ImportSITSFromFile"

    Dim failReason As String
    Dim srcWb As Workbook
    Dim srcWs As Worksheet
    Dim headerRow As Long
    Dim colMap As Object
    Dim optCols As Object
    Dim tailNumber As String
    Dim wsData As Worksheet
    Dim tbl As ListObject
    Dim lastRow As Long
    Dim firstDataRow As Long
    Dim rowIndex As Long
    Dim importedRows As Long
    Dim skippedRows As Long
    Dim skippedGenRows As Long
    Dim intervalCode As String
    Dim taskCode As String
    Dim lifeRemaining As Variant
    Dim newRow As ListRow
    Dim destColMap As Object
    Dim colSelector As Long
    Dim colIntervalDest As Long
    Dim colTaskCodeDest As Long
    Dim colTaskSequence As Long
    Dim colTaskDescription As Long
    Dim colSerial As Long
    Dim colEndItemSerial As Long
    Dim colExtension As Long
    Dim colActualInterval As Long
    Dim colSitsRowId As Long
    Dim ownsBusy As Boolean

    ImportSITSFromFile = -1

    If Not ValidateWorkbookStructure() Then Exit Function

    tailNumber = GetTailNumber()
    If Len(tailNumber) = 0 Then
        If showDialogs Then
            MsgBox "Tail number is not configured. Run the Setup Wizard or seed config first.", _
                   vbExclamation, APP_NAME
        End If
        Exit Function
    End If

    If Len(filePath) = 0 Then
        filePath = PromptImportFile("Select SITS Long Forecast export for " & tailNumber)
        If Len(filePath) = 0 Then Exit Function
    End If

    Set srcWb = ImportOpenReadOnly(filePath, failReason)
    If srcWb Is Nothing Then
        If showDialogs Then ShowImportFailure "SITS", failReason
        LogImportFailure "SITS", failReason
        Exit Function
    End If

    Set srcWs = srcWb.Worksheets(1)
    headerRow = FindHeaderRow(srcWs, SITSRequiredHeaders(), failReason)
    If headerRow = 0 Then
        srcWb.Close SaveChanges:=False
        If showDialogs Then ShowImportFailure "SITS", failReason
        LogImportFailure "SITS", failReason
        Exit Function
    End If

    Set colMap = BuildHeaderMap(srcWs, headerRow, SITSRequiredHeaders(), failReason)
    If colMap Is Nothing Then
        srcWb.Close SaveChanges:=False
        If showDialogs Then ShowImportFailure "SITS", failReason
        LogImportFailure "SITS", failReason
        Exit Function
    End If

    Set optCols = BuildOptionalColumnMap(srcWs, headerRow)

    firstDataRow = headerRow + 1
    lastRow = LastUsedRowAnyColumn(srcWs)
    If lastRow < firstDataRow Then
        srcWb.Close SaveChanges:=False
        failReason = "The file contains headers but no task rows."
        If showDialogs Then ShowImportFailure "SITS", failReason
        LogImportFailure "SITS", failReason
        Exit Function
    End If

    If Not ValidateSITSSourceRows(srcWs, firstDataRow, lastRow, colMap, tailNumber, failReason) Then
        srcWb.Close SaveChanges:=False
        If showDialogs Then ShowImportFailure "SITS", failReason
        LogImportFailure "SITS", failReason
        Exit Function
    End If

    Set wsData = GetWorksheet(SHEET_DATA_SITS)
    Set tbl = GetTable(wsData, TBL_SITS)
    If Not ValidateSITSColumns(tbl) Then
        srcWb.Close SaveChanges:=False
        Exit Function
    End If

    If Not skipOverwritePrompt And showDialogs Then
        If Not tbl.DataBodyRange Is Nothing Then
            If Not PromptOverwriteConfirm("SITS task data") Then
                srcWb.Close SaveChanges:=False
                Exit Function
            End If
        End If
    End If

    On Error GoTo ErrorHandler
    ownsBusy = Not BusyUIActive()
    If ownsBusy Then BeginBusyUI "Importing SITS…"
    SetExcelState "disable"
    ' ListRows.Add / ClearTableData / snapshot table ops need sheets unprotected.
    SuspendWorkbookProtection

    UpdateBusyUI "Clearing existing SITS data…"
    ClearTableData tbl

    ' Ensure structural columns exist before caching indexes for the import loop.
    EnsureSITSBaselineColumns tbl
    EnsureWorkPackageSITSColumns tbl

    Set destColMap = BuildTableColumnIndexMap(tbl)
    colSelector = LookupColumnIndex(destColMap, HDR_SELECTOR)
    colIntervalDest = LookupColumnIndex(destColMap, HDR_INTERVAL_TYPE_CODE)
    colTaskCodeDest = LookupColumnIndex(destColMap, HDR_TASK_CODE)
    colTaskSequence = LookupColumnIndex(destColMap, HDR_TASK_SEQUENCE)
    colTaskDescription = LookupColumnIndex(destColMap, HDR_TASK_DESCRIPTION)
    colSerial = LookupColumnIndex(destColMap, HDR_SERIAL)
    colEndItemSerial = LookupColumnIndex(destColMap, HDR_END_ITEM_SERIAL)
    colExtension = LookupColumnIndex(destColMap, HDR_EXTENSION)
    colActualInterval = LookupColumnIndex(destColMap, HDR_ACTUAL_INTERVAL)
    colSitsRowId = LookupColumnIndex(destColMap, HDR_SITS_ROW_ID)

    UpdateBusyUI "Importing SITS tasks…"
    For rowIndex = firstDataRow To lastRow
        If Not SITSRowHasData(srcWs, rowIndex, colMap) Then GoTo NextRow

        intervalCode = UCase$(Trim$(CStr(srcWs.Cells(rowIndex, colMap("Interval Type Code")).Value)))
        taskCode = Trim$(CStr(srcWs.Cells(rowIndex, colMap("Task Code")).Value))
        lifeRemaining = srcWs.Cells(rowIndex, colMap("Life Remaining")).Value

        If ShouldSkipGenPrefixedTaskCode(taskCode) Then
            skippedGenRows = skippedGenRows + 1
            GoTo NextRow
        End If

        If ShouldSkipNumberBasedSITSImportRow(intervalCode, _
                srcWs.Cells(rowIndex, colMap("Actual Interval")).Value, _
                srcWs.Cells(rowIndex, colMap("Actual Interval")).Text, _
                lifeRemaining, _
                srcWs.Cells(rowIndex, colMap("Life Remaining")).Text) Then
            skippedRows = skippedRows + 1
            GoTo NextRow
        End If

        Set newRow = tbl.ListRows.Add
        With newRow.Range
            .Cells(1, colSelector).Value = False
            .Cells(1, colIntervalDest).Value = intervalCode
            .Cells(1, colTaskCodeDest).Value = taskCode
            .Cells(1, colTaskSequence).Value = _
                srcWs.Cells(rowIndex, colMap("Task Sequence")).Value
            .Cells(1, colTaskDescription).Value = _
                srcWs.Cells(rowIndex, colMap("Task Description")).Value
            WriteSerialCell .Cells(1, colSerial), _
                srcWs.Cells(rowIndex, colMap("Serial")).Value, _
                srcWs.Cells(rowIndex, colMap("Serial")).Text
            WriteSerialCell .Cells(1, colEndItemSerial), _
                srcWs.Cells(rowIndex, colMap("End Item Serial")).Value, _
                srcWs.Cells(rowIndex, colMap("End Item Serial")).Text
            ApplyImportedExtension .Cells(1, colExtension), _
                intervalCode, taskCode, optCols, srcWs, rowIndex, _
                srcWs.Cells(rowIndex, colMap("Extension")).Value, _
                srcWs.Cells(rowIndex, colMap("Extension")).Text

            ApplyActualIntervalValue .Cells(1, colActualInterval), _
                srcWs.Cells(rowIndex, colMap("Actual Interval")).Value, intervalCode, _
                srcWs.Cells(rowIndex, colMap("Actual Interval")).Text
            ApplyLifeRemainingValues tbl, newRow, intervalCode, lifeRemaining, _
                srcWs.Cells(rowIndex, colMap("Life Remaining")).Text, destColMap
            ApplyOptionalSITSColumns tbl, newRow, optCols, srcWs, rowIndex, destColMap
            CaptureTaskBaselineFromImport tbl, newRow, intervalCode, destColMap
            .Cells(1, colSitsRowId).Value = _
                "SITS-" & Format$(importedRows + 1, "00000")
        End With

        importedRows = importedRows + 1

NextRow:
    Next rowIndex

    srcWb.Close SaveChanges:=False
    UpdateBusyUI "Applying formats…"
    ApplyWorkbookDataFormats
    UpdateBusyUI "Snapshotting usage…"
    SnapshotUsageForSITS
    UpdateBusyUI "Recalculating life remaining…"
    RecalculateSITS skipTableFormats:=True
    UpdateBusyUI "Refreshing SITS view…"
    RefreshSITSView

    LogAction "SITS_IMPORTED", "", "Imported " & importedRows & " rows (skipped " & _
              skippedRows & " number-based, " & skippedGenRows & " Gen/) from " & filePath
    ResumeWorkbookProtection
    SetExcelState "enable"
    If ownsBusy Then EndBusyUI

    ImportSITSFromFile = importedRows

    If showDialogs Then
        Dim msg As String
        msg = "Imported " & importedRows & " SITS task(s) for " & tailNumber & "."
        If skippedGenRows > 0 Then
            msg = msg & vbCrLf & skippedGenRows & " Gen/ task code row(s) were skipped."
        End If
        If skippedRows > 0 Then
            msg = msg & vbCrLf & skippedRows & " number-based row(s) were skipped " & _
                  "(actual interval and life remaining were not both zero)."
        End If
        MsgBox msg, vbInformation, APP_NAME
    End If
    Exit Function

ErrorHandler:
    Dim errNum As Long
    Dim errDesc As String
    errNum = Err.Number
    errDesc = Err.Description
    On Error Resume Next
    If Not srcWb Is Nothing Then srcWb.Close SaveChanges:=False
    ForceResumeWorkbookProtection
    SetExcelState "enable"
    If ownsBusy Then EndBusyUI
    On Error GoTo 0
    LogError PROC_NAME, errNum, errDesc
    If showDialogs Then
        MsgBox "SITS import error: " & errNum & " - " & errDesc, vbCritical, APP_NAME
    End If
End Function

Private Function BuildOptionalColumnMap(ByVal srcWs As Worksheet, _
                                        ByVal headerRow As Long) As Object
    Dim dict As Object
    Dim optionalHeaders As Variant
    Dim item As Variant
    Dim colIndex As Long

    Set dict = CreateObject("Scripting.Dictionary")
    optionalHeaders = Array(OPT_PROJECTED_DUE, OPT_INTERVAL_TYPE, OPT_TASK_TYPE_CODE, OPT_WARNING_INTERVAL)

    For Each item In optionalHeaders
        colIndex = FindHeaderColumn(srcWs, headerRow, CStr(item))
        If colIndex > 0 Then dict(CStr(item)) = colIndex
    Next item

    Set BuildOptionalColumnMap = dict
End Function

Private Function ValidateSITSSourceRows(ByVal srcWs As Worksheet, _
                                        ByVal firstDataRow As Long, _
                                        ByVal lastRow As Long, _
                                        ByVal colMap As Object, _
                                        ByVal expectedTail As String, _
                                        ByRef failReason As String) As Boolean
    Dim rowIndex As Long
    Dim validRows As Long
    Dim mismatchCount As Long
    Dim mismatches As String
    Dim sourceSerial As String
    Dim expected As String

    expected = NormaliseTailText(expectedTail)

    For rowIndex = firstDataRow To lastRow
        If Not SITSRowHasData(srcWs, rowIndex, colMap) Then GoTo NextRow

        validRows = validRows + 1
        sourceSerial = NormaliseTailText(CStr(srcWs.Cells(rowIndex, colMap("End Item Serial")).Value))

        If Len(sourceSerial) = 0 Then
            mismatchCount = mismatchCount + 1
            If mismatchCount <= 10 Then
                mismatches = mismatches & "Row " & rowIndex & ": blank End Item Serial" & vbCrLf
            End If
        ElseIf sourceSerial <> expected Then
            mismatchCount = mismatchCount + 1
            If mismatchCount <= 10 Then
                mismatches = mismatches & "Row " & rowIndex & ": End Item Serial '" & _
                             CStr(srcWs.Cells(rowIndex, colMap("End Item Serial")).Value) & _
                             "' does not match '" & expectedTail & "'" & vbCrLf
            End If
        End If

NextRow:
    Next rowIndex

    If validRows = 0 Then
        failReason = "No task rows were found below the header row."
        Exit Function
    End If

    If mismatchCount > 0 Then
        If mismatchCount > 10 Then
            mismatches = mismatches & "...and " & CStr(mismatchCount - 10) & " more mismatch(es)."
        End If
        failReason = "Tail validation failed for " & expectedTail & "." & vbCrLf & vbCrLf & mismatches
        Exit Function
    End If

    ValidateSITSSourceRows = True
End Function

Private Function SITSRowHasData(ByVal srcWs As Worksheet, _
                                ByVal rowIndex As Long, _
                                ByVal colMap As Object) As Boolean
    SITSRowHasData = RowHasAnyValue(srcWs, rowIndex, _
                                    colMap("Task Code"), colMap("Interval Type Code"))
End Function

Private Sub ApplyImportedExtension(ByVal targetCell As Range, _
                                   ByVal intervalCode As String, _
                                   ByVal taskCode As String, _
                                   ByVal optCols As Object, _
                                   ByVal srcWs As Worksheet, _
                                   ByVal rowIndex As Long, _
                                   ByVal rawValue As Variant, _
                                   ByVal displayText As String)
    Dim taskTypeCode As String

    taskTypeCode = ""
    If Not optCols Is Nothing Then
        If optCols.Exists(OPT_TASK_TYPE_CODE) Then
            taskTypeCode = UCase$(Trim$(CStr(srcWs.Cells(rowIndex, optCols(OPT_TASK_TYPE_CODE)).Value)))
        End If
    End If

    If IsDeferredFaultTaskCode(taskCode) Or IsDeferredFaultTaskCode(taskTypeCode) Then
        If SITSViewUsesHourFormatForImport(intervalCode) Then
            WriteFormattedUsageValue targetCell, 0#, FORMAT_TYPE_HOURS
        Else
            targetCell.Value = 0#
            targetCell.NumberFormat = FORMAT_DECIMAL
        End If
        Exit Sub
    End If

    ApplyTimeOrNumberCell targetCell, rawValue, displayText, intervalCode
End Sub

Private Function SITSViewUsesHourFormatForImport(ByVal intervalCode As String) As Boolean
    Select Case UCase$(Trim$(intervalCode))
        Case TYPE_HH, TYPE_E1, TYPE_E2
            SITSViewUsesHourFormatForImport = True
    End Select
End Function

Private Sub ApplyActualIntervalValue(ByVal targetCell As Range, _
                                     ByVal rawValue As Variant, _
                                     ByVal intervalCode As String, _
                                     Optional ByVal displayText As String = "")
    ApplyTimeOrNumberCell targetCell, rawValue, displayText, intervalCode
End Sub

Private Sub ApplyTimeOrNumberCell(ByVal targetCell As Range, _
                                  ByVal rawValue As Variant, _
                                  ByVal displayText As String, _
                                  ByVal intervalCode As String)
    Dim parsedTime As Variant

    If InStr(displayText, ":") > 0 Or (IsHourBasedIntervalType(intervalCode) And intervalCode <> TYPE_DD) Then
        parsedTime = ParseTimeUsageValue(rawValue, displayText)
        If Not IsEmpty(parsedTime) Then
            WriteFormattedUsageValue targetCell, parsedTime, FORMAT_TYPE_HOURS
            Exit Sub
        End If
    End If

    targetCell.Value = rawValue
    targetCell.NumberFormat = FORMAT_DECIMAL
End Sub

Private Sub ApplyLifeRemainingValues(ByVal tbl As ListObject, _
                                     ByVal newRow As ListRow, _
                                     ByVal intervalCode As String, _
                                     ByVal rawValue As Variant, _
                                     ByVal displayText As String, _
                                     Optional ByVal colMap As Object = Nothing)
    Dim colHours As Long
    Dim colNumber As Long
    Dim colDays As Long
    Dim parsedTime As Variant

    colHours = ResolveTableColumnIndex(tbl, HDR_HOURS_LIFE_REMAINING, colMap)
    colNumber = ResolveTableColumnIndex(tbl, HDR_NUMBER_LIFE_REMAINING, colMap)
    colDays = ResolveTableColumnIndex(tbl, HDR_DAYS_WHEN_DUE, colMap)

    If InStr(displayText, ":") > 0 Or (IsHourBasedIntervalType(intervalCode) And intervalCode <> TYPE_DD) Then
        parsedTime = ParseTimeUsageValue(rawValue, displayText)
        If Not IsEmpty(parsedTime) Then
            WriteFormattedUsageValue newRow.Range.Cells(1, colHours), parsedTime, FORMAT_TYPE_HOURS
            Exit Sub
        End If
    End If

    newRow.Range.Cells(1, colNumber).Value = rawValue
    newRow.Range.Cells(1, colNumber).NumberFormat = FORMAT_DECIMAL
    If intervalCode = TYPE_DD Then
        newRow.Range.Cells(1, colDays).Value = rawValue
        newRow.Range.Cells(1, colDays).NumberFormat = FORMAT_DECIMAL
    End If
End Sub

Private Sub ApplyOptionalSITSColumns(ByVal tbl As ListObject, _
                                     ByVal newRow As ListRow, _
                                     ByVal optCols As Object, _
                                     ByVal srcWs As Worksheet, _
                                     ByVal rowIndex As Long, _
                                     Optional ByVal colMap As Object = Nothing)
    Dim parsedDate As Variant
    Dim colIndex As Long

    If optCols Is Nothing Then Exit Sub

    If optCols.Exists(OPT_INTERVAL_TYPE) Then
        colIndex = ResolveTableColumnIndex(tbl, HDR_INTERVAL_TYPE, colMap)
        If colIndex > 0 Then
            newRow.Range.Cells(1, colIndex).Value = srcWs.Cells(rowIndex, optCols(OPT_INTERVAL_TYPE)).Value
        End If
    End If

    If optCols.Exists(OPT_TASK_TYPE_CODE) Then
        colIndex = ResolveTableColumnIndex(tbl, HDR_TASK_TYPE_CODE, colMap)
        If colIndex > 0 Then
            newRow.Range.Cells(1, colIndex).Value = srcWs.Cells(rowIndex, optCols(OPT_TASK_TYPE_CODE)).Value
        End If
    End If

    If optCols.Exists(OPT_WARNING_INTERVAL) Then
        colIndex = ResolveTableColumnIndex(tbl, HDR_WARNING_INTERVAL, colMap)
        If colIndex > 0 Then
            newRow.Range.Cells(1, colIndex).Value = srcWs.Cells(rowIndex, optCols(OPT_WARNING_INTERVAL)).Value
        End If
    End If

    If optCols.Exists(OPT_PROJECTED_DUE) Then
        colIndex = ResolveTableColumnIndex(tbl, HDR_PROJECTED_DUE_DATE, colMap)
        If colIndex > 0 Then
            parsedDate = ParseDMYDateValue( _
                srcWs.Cells(rowIndex, optCols(OPT_PROJECTED_DUE)).Value, _
                srcWs.Cells(rowIndex, optCols(OPT_PROJECTED_DUE)).Text)
            If IsDate(parsedDate) Then
                newRow.Range.Cells(1, colIndex).Value = CDate(parsedDate)
                newRow.Range.Cells(1, colIndex).NumberFormat = FORMAT_DATE
            End If
        End If
    End If
End Sub

Private Function ShouldSkipNumberBasedSITSImportRow(ByVal intervalCode As String, _
                                                    ByVal actualIntervalValue As Variant, _
                                                    ByVal actualIntervalText As String, _
                                                    ByVal lifeRemainingValue As Variant, _
                                                    ByVal lifeRemainingText As String) As Boolean
    If Not IsNumberBasedSITSInterval(intervalCode) Then Exit Function

    ShouldSkipNumberBasedSITSImportRow = Not ( _
        IsImportedScalarZero(actualIntervalValue, actualIntervalText) And _
        IsImportedScalarZero(lifeRemainingValue, lifeRemainingText))
End Function

' GOLDesp generic / generated task codes (e.g. Gen/XXXX) are not imported.
Private Function ShouldSkipGenPrefixedTaskCode(ByVal taskCode As String) As Boolean
    ShouldSkipGenPrefixedTaskCode = _
        (StrComp(Left$(Trim$(taskCode), 4), "Gen/", vbTextCompare) = 0)
End Function

Private Function IsImportedScalarZero(ByVal rawValue As Variant, ByVal displayText As String) As Boolean
    Dim parsed As Variant

    If IsError(rawValue) Then Exit Function

    If IsEmpty(rawValue) Or IsNull(rawValue) Then
        IsImportedScalarZero = True
        Exit Function
    End If

    If VarType(rawValue) = vbString And Len(Trim$(CStr(rawValue))) = 0 Then
        IsImportedScalarZero = True
        Exit Function
    End If

    If InStr(displayText, ":") > 0 Then
        parsed = ParseTimeUsageValue(rawValue, displayText)
        If Not IsEmpty(parsed) Then
            IsImportedScalarZero = (CDbl(parsed) = 0#)
            Exit Function
        End If
    End If

    If IsNumeric(rawValue) Then
        IsImportedScalarZero = (CDbl(rawValue) = 0#)
    End If
End Function
