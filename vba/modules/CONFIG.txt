' ========================================================================
' Config sheet accessors
' ========================================================================

Public Function GetConfigValue(ByVal settingName As String, _
                               Optional ByVal defaultValue As Variant) As Variant
    Dim ws As Worksheet
    Dim tbl As ListObject
    Dim row As ListRow

    Set ws = GetWorksheet(SHEET_CONFIG, showError:=False)
    If ws Is Nothing Then
        GetConfigValue = defaultValue
        Exit Function
    End If

    Set tbl = GetTable(ws, TBL_CONFIG, showError:=False)
    If tbl Is Nothing Or tbl.DataBodyRange Is Nothing Then
        GetConfigValue = defaultValue
        Exit Function
    End If

    For Each row In tbl.ListRows
        If StrComp(CStr(row.Range.Cells(1, 1).Value), settingName, vbTextCompare) = 0 Then
            GetConfigValue = row.Range.Cells(1, 2).Value
            Exit Function
        End If
    Next row

    GetConfigValue = defaultValue
End Function

Public Sub SetConfigValue(ByVal settingName As String, ByVal newValue As Variant)
    Dim ws As Worksheet
    Dim tbl As ListObject
    Dim row As ListRow
    Dim found As Boolean

    Set ws = GetWorksheet(SHEET_CONFIG)
    Set tbl = GetTable(ws, TBL_CONFIG)

    found = False
    If Not tbl.DataBodyRange Is Nothing Then
        For Each row In tbl.ListRows
            If StrComp(CStr(row.Range.Cells(1, 1).Value), settingName, vbTextCompare) = 0 Then
                row.Range.Cells(1, 2).Value = newValue
                found = True
                Exit For
            End If
        Next row
    End If

    If Not found Then
        Set row = tbl.ListRows.Add
        row.Range.Cells(1, 1).Value = settingName
        row.Range.Cells(1, 2).Value = newValue
    End If
End Sub

Public Function GetTailNumber() As String
    GetTailNumber = Trim$(CStr(GetConfigValue(CFG_TAIL_NUMBER, "")))
End Function

Public Function IsSetupComplete() As Boolean
    IsSetupComplete = CBool(GetConfigValue(CFG_SETUP_COMPLETE, False))
End Function

Public Function IsMasterTemplate() As Boolean
    IsMasterTemplate = CBool(GetConfigValue(CFG_MASTER_TEMPLATE, False))
End Function
