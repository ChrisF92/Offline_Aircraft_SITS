' Nested disable/enable so inner helpers cannot turn ScreenUpdating back on
' while an outer operation is still running.
Private mExcelStateDepth As Long
Private mSavedCalculation As XlCalculation
Private mSavedStatusBar As Variant

Public Sub SetExcelState(ByVal action As String)
    Select Case LCase$(Trim$(action))
        Case "disable", "fast"
            If mExcelStateDepth = 0 Then
                On Error Resume Next
                mSavedCalculation = Application.Calculation
                mSavedStatusBar = Application.StatusBar
                On Error GoTo 0

                Application.ScreenUpdating = False
                Application.EnableEvents = False
                Application.DisplayAlerts = False
                Application.Calculation = xlCalculationManual
                Application.Cursor = xlWait
            End If
            mExcelStateDepth = mExcelStateDepth + 1

        Case "enable"
            If mExcelStateDepth > 0 Then mExcelStateDepth = mExcelStateDepth - 1
            If mExcelStateDepth = 0 Then
                On Error Resume Next
                If mSavedCalculation = 0 Then
                    Application.Calculation = xlCalculationAutomatic
                Else
                    Application.Calculation = mSavedCalculation
                End If
                Application.StatusBar = False
                On Error GoTo 0

                Application.Cursor = xlDefault
                Application.DisplayAlerts = True
                Application.EnableEvents = True
                Application.ScreenUpdating = True
            End If

        Case "force-enable"
            mExcelStateDepth = 0
            On Error Resume Next
            Application.Calculation = xlCalculationAutomatic
            Application.StatusBar = False
            On Error GoTo 0
            Application.Cursor = xlDefault
            Application.DisplayAlerts = True
            Application.EnableEvents = True
            Application.ScreenUpdating = True
    End Select
End Sub

Public Function GetTableColumnIndex(ByVal tbl As ListObject, _
                                    ByVal columnName As String) As Long
    Dim col As ListColumn

    GetTableColumnIndex = 0
    If tbl Is Nothing Then Exit Function

    For Each col In tbl.ListColumns
        If StrComp(col.Name, columnName, vbTextCompare) = 0 Then
            GetTableColumnIndex = col.Index
            Exit Function
        End If
    Next col
End Function

Public Function CleanText(ByVal text As String) As String
    CleanText = Trim$(Replace(Replace(text, Chr$(160), " "), vbTab, " "))
    Do While InStr(CleanText, "  ") > 0
        CleanText = Replace(CleanText, "  ", " ")
    Loop
End Function

Public Function NormaliseHeaderText(ByVal headerText As String) As String
    NormaliseHeaderText = LCase$(CleanText(CStr(headerText)))
End Function

Public Function NormaliseTailText(ByVal tail As String) As String
    NormaliseTailText = UCase$(Replace(CleanText(CStr(tail)), " ", ""))
End Function

' ORN format: ####TAILDDMMYY  e.g. 7034ZZ383081125
' First 4 digits any number; middle segment must match aircraft tail; last 6 must be a real date.
Public Function ValidateORNForAircraft(ByVal orn As String, _
                                       ByVal tailNumber As String, _
                                       ByRef failReason As String) As Boolean
    Dim cleanOrn As String
    Dim cleanTail As String
    Dim expectedLen As Long
    Dim prefix As String
    Dim midTail As String
    Dim datePart As String
    Dim dayPart As Long
    Dim monthPart As Long
    Dim yearPart As Long
    Dim parsedDate As Date
    Dim i As Long
    Dim ch As String

    failReason = ""
    cleanOrn = UCase$(Trim$(orn))
    cleanTail = NormaliseTailText(tailNumber)

    If Len(cleanOrn) = 0 Then
        failReason = "Enter an ORN."
        Exit Function
    End If

    If Len(cleanTail) = 0 Then
        failReason = "Aircraft tail number is not configured."
        Exit Function
    End If

    expectedLen = ORN_PREFIX_DIGITS + Len(cleanTail) + ORN_DATE_DIGITS
    If Len(cleanOrn) <> expectedLen Then
        failReason = "ORN must be " & expectedLen & " characters: ####" & cleanTail & "DDMMYY"
        Exit Function
    End If

    prefix = Left$(cleanOrn, ORN_PREFIX_DIGITS)
    midTail = Mid$(cleanOrn, ORN_PREFIX_DIGITS + 1, Len(cleanTail))
    datePart = Right$(cleanOrn, ORN_DATE_DIGITS)

    For i = 1 To Len(prefix)
        ch = Mid$(prefix, i, 1)
        If Not ch Like "#" Then
            failReason = "ORN must start with 4 digits."
            Exit Function
        End If
    Next i

    If StrComp(midTail, cleanTail, vbTextCompare) <> 0 Then
        failReason = "ORN must include aircraft tail " & cleanTail & " after the first 4 digits."
        Exit Function
    End If

    For i = 1 To Len(datePart)
        ch = Mid$(datePart, i, 1)
        If Not ch Like "#" Then
            failReason = "ORN date segment (DDMMYY) must be 6 digits."
            Exit Function
        End If
    Next i

    dayPart = CLng(Left$(datePart, 2))
    monthPart = CLng(Mid$(datePart, 3, 2))
    yearPart = 2000 + CLng(Right$(datePart, 2))

    On Error Resume Next
    parsedDate = DateSerial(yearPart, monthPart, dayPart)
    If Err.Number <> 0 Then
        Err.Clear
        On Error GoTo 0
        failReason = "ORN date segment is not a valid calendar date (DDMMYY)."
        Exit Function
    End If
    On Error GoTo 0

    If Day(parsedDate) <> dayPart Or Month(parsedDate) <> monthPart Or Year(parsedDate) <> yearPart Then
        failReason = "ORN date segment is not a valid calendar date (DDMMYY)."
        Exit Function
    End If

    ValidateORNForAircraft = True
End Function

Public Sub SetCellCommentText(ByVal target As Range, ByVal commentText As String)
    On Error Resume Next
    If Not target.Comment Is Nothing Then target.Comment.Delete
    On Error GoTo 0

    If Len(Trim$(commentText)) = 0 Then Exit Sub

    On Error Resume Next
    target.AddComment Text:=commentText
    If Not target.Comment Is Nothing Then
        target.Comment.Visible = False
        target.Comment.Shape.TextFrame.AutoSize = True
    End If
    On Error GoTo 0
End Sub

    On Error Resume Next
    If Not source.Comment Is Nothing Then
        GetCellCommentText = source.Comment.Text
    End If
    On Error GoTo 0
End Function

Public Function ConvertToBoolean(ByVal value As Variant) As Boolean
    If IsError(value) Then Exit Function
    If IsEmpty(value) Then Exit Function

    Select Case VarType(value)
        Case vbBoolean
            ConvertToBoolean = value
        Case vbString
            Select Case UCase$(Trim$(CStr(value)))
                Case "TRUE", "YES", "Y", "1"
                    ConvertToBoolean = True
            End Select
        Case vbInteger, vbLong, vbSingle, vbDouble, vbCurrency, vbByte
            ConvertToBoolean = (CLng(value) <> 0)
    End Select
End Function

Public Function LastUsedRowAnyColumn(ByVal ws As Worksheet) As Long
    Dim lastCell As Range

    On Error Resume Next
    Set lastCell = ws.Cells.Find(What:="*", _
                                 After:=ws.Range("A1"), _
                                 LookIn:=xlFormulas, _
                                 LookAt:=xlPart, _
                                 SearchOrder:=xlByRows, _
                                 SearchDirection:=xlPrevious)
    On Error GoTo 0

    If lastCell Is Nothing Then
        LastUsedRowAnyColumn = 0
    Else
        LastUsedRowAnyColumn = lastCell.Row
    End If
End Function

Public Sub DeleteSheetIfExists(ByVal sheetName As String)
    Dim ws As Worksheet

    ' Excel requires at least one worksheet — never delete the last sheet.
    If ThisWorkbook.Worksheets.Count <= 1 Then Exit Sub

    On Error Resume Next
    Set ws = ThisWorkbook.Worksheets(sheetName)
    On Error GoTo 0

    If Not ws Is Nothing Then
        Application.DisplayAlerts = False
        ws.Delete
        Application.DisplayAlerts = True
    End If
End Sub

' Excel requires at least one worksheet — delete extras, clear the last one.
Public Sub PrepareWorkbookForSetup()
    Application.DisplayAlerts = False

    Do While ThisWorkbook.Worksheets.Count > 1
        ThisWorkbook.Worksheets(1).Delete
    Loop

    With ThisWorkbook.Worksheets(1)
        .Cells.Clear
        .Visible = xlSheetVisible
    End With

    Application.DisplayAlerts = True
End Sub

Public Sub ReuseSheetAs(ByVal sheetName As String)
    Dim ws As Worksheet

    Set ws = ThisWorkbook.Worksheets(1)

    If ws.Name <> sheetName And SheetExists(sheetName) Then
        DeleteSheetIfExists sheetName
    End If

    ws.Name = sheetName
    ws.Visible = xlSheetVisible
    ws.Cells.Clear
End Sub

Public Sub AddSetupSheet(ByVal sheetName As String)
    Dim ws As Worksheet

    If SheetExists(sheetName) Then DeleteSheetIfExists sheetName

    Set ws = ThisWorkbook.Worksheets.Add(After:=ThisWorkbook.Worksheets(ThisWorkbook.Worksheets.Count))
    ws.Name = sheetName
    ws.Visible = xlSheetVisible
End Sub

Public Sub OrderSetupSheets()
    Dim order As Variant
    Dim i As Long
    Dim ws As Worksheet
    Dim targetIndex As Long

    order = Array( _
        SHEET_DASHBOARD, SHEET_SITS, SHEET_USAGE, SHEET_AUDIT, _
        SHEET_CONFIG, SHEET_DATA_SITS, SHEET_DATA_USAGE_BASELINE, _
        SHEET_DATA_USAGE_LOG, SHEET_DATA_USAGE_CURRENT, _
        SHEET_DATA_WP, SHEET_DATA_WP_TASKS, SHEET_WP_STATE _
    )

    For i = LBound(order) To UBound(order)
        targetIndex = i + 1
        Set ws = ThisWorkbook.Worksheets(CStr(order(i)))

        ' Avoid Move Before self — that can trigger Excel error 1004.
        If ws.Index <> targetIndex Then
            ws.Move Before:=ThisWorkbook.Worksheets(targetIndex)
        End If
    Next i
End Sub

Public Sub FinalizeSetupSheetLayout()
    ' Call once at the very end of setup, after tables/config/dashboard exist.
    ' Phase 0: unhide everything first, then hide data sheets with xlSheetHidden only
    ' (not xlSheetVeryHidden — that is hard to recover from if something goes wrong).

    Dim ws As Worksheet

    ' Recover from any hidden state left by a partial or earlier failed run.
    For Each ws In ThisWorkbook.Worksheets
        ws.Visible = xlSheetVisible
    Next ws

    ThisWorkbook.Worksheets(SHEET_DASHBOARD).Tab.Color = COLOR_TAB_VISIBLE
    ThisWorkbook.Worksheets(SHEET_SITS).Tab.Color = COLOR_TAB_VISIBLE
    ThisWorkbook.Worksheets(SHEET_USAGE).Tab.Color = COLOR_TAB_VISIBLE
    ThisWorkbook.Worksheets(SHEET_AUDIT).Tab.Color = COLOR_TAB_SYSTEM
    ThisWorkbook.Worksheets(SHEET_CONFIG).Tab.Color = COLOR_TAB_DATA
    ThisWorkbook.Worksheets(SHEET_DATA_SITS).Tab.Color = COLOR_TAB_DATA
    ThisWorkbook.Worksheets(SHEET_DATA_USAGE_BASELINE).Tab.Color = COLOR_TAB_DATA
    ThisWorkbook.Worksheets(SHEET_DATA_USAGE_LOG).Tab.Color = COLOR_TAB_DATA
    ThisWorkbook.Worksheets(SHEET_DATA_USAGE_CURRENT).Tab.Color = COLOR_TAB_DATA
    ThisWorkbook.Worksheets(SHEET_DATA_WP).Tab.Color = COLOR_TAB_DATA
    ThisWorkbook.Worksheets(SHEET_DATA_WP_TASKS).Tab.Color = COLOR_TAB_DATA
    ThisWorkbook.Worksheets(SHEET_WP_STATE).Tab.Color = COLOR_TAB_SYSTEM

    ThisWorkbook.Worksheets(SHEET_DASHBOARD).Activate

    HideDataSetupSheet SHEET_CONFIG
    HideDataSetupSheet SHEET_DATA_SITS
    HideDataSetupSheet SHEET_DATA_USAGE_BASELINE
    HideDataSetupSheet SHEET_DATA_USAGE_LOG
    HideDataSetupSheet SHEET_DATA_USAGE_CURRENT
    HideDataSetupSheet SHEET_DATA_WP
    HideDataSetupSheet SHEET_DATA_WP_TASKS
    HideDataSetupSheet SHEET_WP_STATE

    ThisWorkbook.Worksheets(SHEET_DASHBOARD).Activate
End Sub

' Run manually from VBA if sheets are stuck hidden after a failed setup.
Public Sub RepairSetupSheetVisibility()
    Dim ws As Worksheet

    For Each ws In ThisWorkbook.Worksheets
        ws.Visible = xlSheetVisible
    Next ws

    On Error Resume Next
    ThisWorkbook.Worksheets(SHEET_DASHBOARD).Activate
    On Error GoTo 0

    MsgBox "All worksheets are now visible." & vbCrLf & vbCrLf & _
           "Run Setup_RunAll again, or run FinalizeSetupSheetLayout to apply tab colours.", _
           vbInformation, APP_NAME
End Sub

Public Function BuildUsageKey(ByVal intervalCode As String, _
                                ByVal taskCode As String) As String
    BuildUsageKey = UCase$(Trim$(intervalCode)) & "|" & Trim$(taskCode)
End Function

Public Function StripAssociatedSuffix(ByVal taskCode As String) As String
    taskCode = Trim$(taskCode)
    If Right$(taskCode, Len(ASSOCIATED_TASK_SUFFIX)) = ASSOCIATED_TASK_SUFFIX Then
        StripAssociatedSuffix = Left$(taskCode, Len(taskCode) - Len(ASSOCIATED_TASK_SUFFIX))
    Else
        StripAssociatedSuffix = taskCode
    End If
End Function

' Associated tasks share base Task Code + Sequence on the SAME Serial
' (e.g. WC/PMP/2511002 and WC/PMP/2511002/1 on serial 2337014).
' Same task code on different serials (e.g. each engine's WC/PMP/7200005)
' must NOT claim each other — Serial is required in the key.
Public Function BuildAssociatedLinkKey(ByVal taskCode As String, _
                                       ByVal taskSequence As String, _
                                       ByVal serialNumber As String) As String
    BuildAssociatedLinkKey = StripAssociatedSuffix(taskCode) & "|" & _
                             Trim$(CStr(taskSequence)) & "|" & _
                             UCase$(Trim$(CStr(serialNumber)))
End Function

End Function

Public Function IsHourBasedIntervalType(ByVal intervalCode As String) As Boolean
    Select Case UCase$(Trim$(intervalCode))
        Case TYPE_HH, TYPE_E1, TYPE_E2, TYPE_DD
            IsHourBasedIntervalType = True
        Case Else
            IsHourBasedIntervalType = (InStr(UCase$(intervalCode), "E") = 1 And Len(intervalCode) <= 2)
    End Select
End Function

Public Function ParseDMYDateValue(ByVal rawVal As Variant, _
                                  Optional ByVal displayText As String = "") As Variant
    Dim s As String
    Dim parts() As String
    Dim d As Long
    Dim m As Long
    Dim y As Long
    Dim parsedDate As Date

    s = Trim$(CStr(displayText))

    If Len(s) > 0 And s <> "#####" Then
        If InStr(s, " ") > 0 Then s = Left$(s, InStr(s, " ") - 1)
        s = Replace(Replace(s, "-", "/"), ".", "/")

        If InStr(s, "/") > 0 Then
            parts = Split(s, "/")
            If UBound(parts) = 2 Then
                If IsNumeric(parts(0)) And IsNumeric(parts(1)) And IsNumeric(parts(2)) Then
                    d = CLng(parts(0))
                    m = CLng(parts(1))
                    y = CLng(parts(2))
                    If y < 100 Then y = 2000 + y

                    On Error Resume Next
                    parsedDate = DateSerial(y, m, d)
                    On Error GoTo 0

                    If Day(parsedDate) = d And Month(parsedDate) = m And Year(parsedDate) = y Then
                        ParseDMYDateValue = parsedDate
                        Exit Function
                    End If
                End If
            End If
        End If
    End If

    If IsDate(rawVal) Then
        ParseDMYDateValue = DateValue(CDate(rawVal))
    End If
End Function

Public Function ParseTimeUsageValue(ByVal rawVal As Variant, _
                                    Optional ByVal displayText As String = "") As Variant
    Dim s As String
    Dim parts() As String
    Dim hoursPart As Double
    Dim minutesPart As Double

    If IsEmpty(rawVal) Then Exit Function

    s = Trim$(CStr(displayText))
    If Len(s) = 0 Or s = "#####" Then s = Trim$(CStr(rawVal))

    If InStr(s, ":") > 0 Then
        parts = Split(s, ":")
        If UBound(parts) >= 1 Then
            If IsNumeric(parts(0)) And IsNumeric(parts(1)) Then
                hoursPart = CDbl(parts(0))
                minutesPart = CDbl(parts(1))
                ParseTimeUsageValue = (hoursPart / 24#) + (minutesPart / 1440#)
                Exit Function
            End If
        End If
    End If

    If IsNumeric(rawVal) Then
        ParseTimeUsageValue = CDbl(rawVal)
    End If
End Function

Public Function DetectUsageFormat(ByVal sampleValue As Variant, _
                                  Optional ByVal intervalCode As String = "", _
                                  Optional ByVal taskCode As String = "", _
                                  Optional ByVal description As String = "", _
                                  Optional ByVal displayText As String = "") As String
    If IsSecondsCountInterval(intervalCode) Then
        DetectUsageFormat = FORMAT_TYPE_NUMBER
        Exit Function
    End If

    If IsHourUsageInterval(intervalCode) Then
        DetectUsageFormat = FORMAT_TYPE_HOURS
        Exit Function
    End If

    If Len(Trim$(displayText)) > 0 And InStr(displayText, ":") > 0 Then
        DetectUsageFormat = FORMAT_TYPE_HOURS
        Exit Function
    End If

    If IsEmpty(sampleValue) Then
        DetectUsageFormat = FORMAT_TYPE_NUMBER
        Exit Function
    End If

    ' Do not use IsDate() here — VBA returns True for any numeric serial (e.g. 3.38).

    If InStr(CStr(sampleValue), ":") > 0 Then
        DetectUsageFormat = FORMAT_TYPE_HOURS
    Else
        DetectUsageFormat = FORMAT_TYPE_NUMBER
    End If
End Function

Public Function IsHourUsageInterval(ByVal intervalCode As String) As Boolean
    Select Case UCase$(Trim$(intervalCode))
        Case TYPE_HH, TYPE_E1, TYPE_E2, "MX", "OH", "XJ", "DA", "HRS", "AF"
            IsHourUsageInterval = True
    End Select
End Function

Public Function IsMinutesUsageCounter(ByVal intervalCode As String, _
                                      ByVal taskCode As String, _
                                      Optional ByVal description As String = "") As Boolean
    Dim code As String
    Dim task As String
    Dim desc As String

    code = UCase$(Trim$(intervalCode))
    task = UCase$(Trim$(taskCode))
    desc = UCase$(Trim$(description))

    ' 1E/1G/2E/2G are decimal second totals (Number), not elapsed m:ss times.
    If IsSecondsCountInterval(code) Then Exit Function

    If InStr(task, "/SEC") > 0 Or InStr(task, "/MIN/") > 0 Then IsMinutesUsageCounter = True
    If InStr(desc, "SECOND") > 0 And InStr(desc, "HOUR") = 0 Then IsMinutesUsageCounter = True
End Function

Public Function IsSecondsCountInterval(ByVal intervalCode As String) As Boolean
    Select Case UCase$(Trim$(intervalCode))
        Case "1E", "1G", "2E", "2G"
            IsSecondsCountInterval = True
    End Select
End Function

' Number-based SITS intervals (starts, cycles, etc.) — not DD or hour-based types.
Public Function IsNumberBasedSITSInterval(ByVal intervalCode As String) As Boolean
    intervalCode = UCase$(Trim$(intervalCode))
    If intervalCode = TYPE_DD Then Exit Function
    If IsHourUsageInterval(intervalCode) Then Exit Function
    If IsHourBasedIntervalType(intervalCode) And Not IsSecondsCountInterval(intervalCode) Then Exit Function
    IsNumberBasedSITSInterval = True
End Function

Public Function GetFormatPatternForType(ByVal formatType As String) As String
    Select Case UCase$(Trim$(formatType))
        Case UCase$(FORMAT_TYPE_HOURS)
            GetFormatPatternForType = FORMAT_TIME_HOURS
        Case UCase$(FORMAT_TYPE_MINUTES)
            GetFormatPatternForType = FORMAT_TIME_MINUTES
        Case Else
            GetFormatPatternForType = FORMAT_DECIMAL
    End Select
End Function

Public Function NormalizeSerialText(ByVal rawVal As Variant, _
                                    Optional ByVal displayText As String = "") As String
    Dim s As String

    s = Trim$(displayText)
    If Len(s) > 0 And s <> "#####" Then
        NormalizeSerialText = s
        Exit Function
    End If

    If IsEmpty(rawVal) Then Exit Function

    If IsNumeric(rawVal) Then
        If CDbl(rawVal) = Fix(CDbl(rawVal)) Then
            NormalizeSerialText = Format$(rawVal, "0")
        Else
            NormalizeSerialText = Trim$(CStr(rawVal))
        End If
    Else
        NormalizeSerialText = Trim$(CStr(rawVal))
    End If
End Function

Public Sub WriteSerialCell(ByVal target As Range, _
                           ByVal rawVal As Variant, _
                           Optional ByVal displayText As String = "")
    Dim serialText As String

    serialText = NormalizeSerialText(rawVal, displayText)
    target.NumberFormat = FORMAT_TEXT
    target.Value = serialText
End Sub

' Convert an imported Life Used value based on interval code (Hours vs Number).
Public Function CoerceImportedUsageValue(ByVal value As Variant, _
                                         ByVal displayText As String, _
                                         ByVal intervalCode As String) As Double
    Dim textValue As String
    Dim parsed As Variant

    If IsHourUsageInterval(intervalCode) Then
        textValue = Trim$(displayText)
        If Len(textValue) = 0 Or textValue = "#####" Then textValue = Trim$(CStr(value))
        If Len(textValue) > 0 And InStr(textValue, ":") > 0 Then
            parsed = ParseTimeUsageValue(value, textValue)
            If Not IsEmpty(parsed) Then
                CoerceImportedUsageValue = CDbl(parsed)
                Exit Function
            End If
        End If
        If IsNumeric(value) Then
            CoerceImportedUsageValue = CDbl(value)
        Else
            CoerceImportedUsageValue = 0#
        End If
    ElseIf IsNumeric(value) Then
        CoerceImportedUsageValue = CDbl(value)
    Else
        CoerceImportedUsageValue = 0#
    End If
End Function

Public Function UsageFormatPatternForInterval(ByVal intervalCode As String, _
                                              Optional ByVal storedValue As Variant) As String
    If IsHourUsageInterval(intervalCode) Then
        UsageFormatPatternForInterval = FORMAT_TIME_HOURS
    ElseIf IsSecondsCountInterval(intervalCode) Then
        UsageFormatPatternForInterval = FORMAT_DECIMAL
    ElseIf IsNumeric(storedValue) And CDbl(storedValue) <> Fix(CDbl(storedValue)) Then
        UsageFormatPatternForInterval = FORMAT_DECIMAL
    Else
        UsageFormatPatternForInterval = FORMAT_INTEGER
    End If
End Function

Public Function StoredUsageFormatLabel(ByVal intervalCode As String) As String
    If IsHourUsageInterval(intervalCode) Then
        StoredUsageFormatLabel = FORMAT_TYPE_HOURS
    Else
        StoredUsageFormatLabel = FORMAT_TYPE_NUMBER
    End If
End Function

' Convert a stored usage value for display/write when format label is already known.
Public Function CoerceUsageStorageValue(ByVal value As Variant, _
                                        ByVal formatType As String, _
                                        Optional ByVal displayText As String = "") As Double
    Dim parsed As Variant
    Dim textValue As String

    Select Case UCase$(Trim$(formatType))
        Case UCase$(FORMAT_TYPE_HOURS)
            textValue = Trim$(displayText)
            If Len(textValue) > 0 And InStr(textValue, ":") > 0 Then
                parsed = ParseTimeUsageValue(value, textValue)
                If Not IsEmpty(parsed) Then
                    CoerceUsageStorageValue = CDbl(parsed)
                    Exit Function
                End If
            End If
            If VarType(value) = vbString And InStr(CStr(value), ":") > 0 Then
                parsed = ParseTimeUsageValue(value, CStr(value))
                If Not IsEmpty(parsed) Then
                    CoerceUsageStorageValue = CDbl(parsed)
                    Exit Function
                End If
            End If
            If IsNumeric(value) Then
                CoerceUsageStorageValue = CDbl(value)
            Else
                CoerceUsageStorageValue = 0#
            End If
        Case Else
            If IsNumeric(value) Then
                CoerceUsageStorageValue = CDbl(value)
            Else
                CoerceUsageStorageValue = 0#
            End If
    End Select
End Function

Public Function UsageNumberFormatPattern(ByVal formatType As String, _
                                         ByVal storedValue As Variant) As String
    If StrComp(UCase$(Trim$(formatType)), UCase$(FORMAT_TYPE_HOURS), vbTextCompare) = 0 Then
        UsageNumberFormatPattern = FORMAT_TIME_HOURS
    ElseIf IsNumeric(storedValue) And CDbl(storedValue) <> Fix(CDbl(storedValue)) Then
        UsageNumberFormatPattern = FORMAT_DECIMAL
    Else
        UsageNumberFormatPattern = FORMAT_INTEGER
    End If
End Function

Public Sub WriteUsageCell(ByVal target As Range, _
                          ByVal value As Variant, _
                          ByVal formatType As String, _
                          Optional ByVal displayText As String = "")
    Dim storedValue As Double
    Dim pattern As String

    storedValue = CoerceUsageStorageValue(value, formatType, displayText)
    pattern = UsageNumberFormatPattern(formatType, storedValue)
    target.Value = storedValue
    target.NumberFormat = pattern
End Sub

Public Function NormalizeUsageImportValue(ByVal rawVal As Variant, _
                                          ByVal formatType As String, _
                                          Optional ByVal displayText As String = "", _
                                          Optional ByVal intervalCode As String = "") As Variant
    If Len(Trim$(intervalCode)) > 0 Then
        NormalizeUsageImportValue = CoerceImportedUsageValue(rawVal, displayText, intervalCode)
    Else
        NormalizeUsageImportValue = CoerceUsageStorageValue(rawVal, formatType, displayText)
    End If
End Function

Public Function ResolveUsageFormatType(ByVal intervalCode As String, _
                                       ByVal taskCode As String, _
                                       Optional ByVal storedFormat As String = "", _
                                       Optional ByVal description As String = "") As String
    If IsSecondsCountInterval(intervalCode) Then
        ResolveUsageFormatType = FORMAT_TYPE_NUMBER
    ElseIf IsHourUsageInterval(intervalCode) Then
        ResolveUsageFormatType = FORMAT_TYPE_HOURS
    ElseIf Len(Trim$(storedFormat)) > 0 Then
        ResolveUsageFormatType = Trim$(storedFormat)
    Else
        ResolveUsageFormatType = FORMAT_TYPE_NUMBER
    End If
End Function

Public Function FormatUsageValueForDisplay(ByVal value As Variant, _
                                           ByVal formatType As String) As String
    If IsEmpty(value) Or Len(Trim$(CStr(value))) = 0 Then
        FormatUsageValueForDisplay = ""
        Exit Function
    End If

    Select Case UCase$(Trim$(formatType))
        Case UCase$(FORMAT_TYPE_HOURS)
            FormatUsageValueForDisplay = FormatElapsedHoursFromDayFraction(CDbl(value))
        Case FORMAT_TYPE_MINUTES
            FormatUsageValueForDisplay = FormatElapsedMinutesFromDayFraction(CDbl(value))
        Case Else
            If IsNumeric(value) And CDbl(value) = Fix(CDbl(value)) Then
                FormatUsageValueForDisplay = Format$(value, FORMAT_INTEGER)
            Else
                FormatUsageValueForDisplay = CStr(value)
            End If
    End Select
End Function

' Elapsed hours for display — avoids VBA Format$ limitations with [h]:mm above 24:00.
Public Function FormatElapsedHoursFromDayFraction(ByVal dayFraction As Double) As String
    Dim totalMinutes As Long
    Dim hours As Long
    Dim minutes As Long

    totalMinutes = CLng(Round(dayFraction * 24# * 60#, 0))
    If totalMinutes < 0 Then totalMinutes = 0

    hours = totalMinutes \ 60
    minutes = totalMinutes Mod 60
    FormatElapsedHoursFromDayFraction = CStr(hours) & ":" & Format$(minutes, "00")
End Function

' Elapsed minutes:seconds for display — supports totals above 59:59.
Public Function FormatElapsedMinutesFromDayFraction(ByVal dayFraction As Double) As String
    Dim totalSeconds As Long
    Dim minutes As Long
    Dim seconds As Long

    totalSeconds = CLng(Round(dayFraction * 86400#, 0))
    If totalSeconds < 0 Then totalSeconds = 0

    minutes = totalSeconds \ 60
    seconds = totalSeconds Mod 60
    FormatElapsedMinutesFromDayFraction = CStr(minutes) & ":" & Format$(seconds, "00")
End Function

Public Function ParseMinutesSecondsValue(ByVal rawVal As Variant, _
                                         Optional ByVal displayText As String = "") As Variant
    Dim s As String
    Dim parts() As String
    Dim minutesPart As Double
    Dim secondsPart As Double
    Dim totalSeconds As Double

    If IsEmpty(rawVal) Then Exit Function

    s = Trim$(displayText)
    If Len(s) = 0 Or s = "#####" Then s = Trim$(CStr(rawVal))

    If InStr(s, ":") > 0 Then
        parts = Split(s, ":")
        If UBound(parts) >= 1 Then
            If IsNumeric(parts(0)) And IsNumeric(parts(1)) Then
                minutesPart = CDbl(parts(0))
                secondsPart = CDbl(parts(1))
                totalSeconds = (minutesPart * 60#) + secondsPart
                ParseMinutesSecondsValue = totalSeconds / 86400#
                Exit Function
            End If
        End If
    End If

    If IsNumeric(rawVal) Then
        ParseMinutesSecondsValue = CDbl(rawVal) / 86400#
    End If
End Function

Public Sub FormatAllUsageDataTables()
    Dim ws As Worksheet
    Dim tbl As ListObject
    Dim row As ListRow
    Dim intervalCode As String
    Dim taskCode As String
    Dim description As String
    Dim formatType As String
    Dim baselineValue As Variant

    Set ws = GetWorksheet(SHEET_DATA_USAGE_BASELINE, showError:=False)
    If Not ws Is Nothing Then
        Set tbl = GetTable(ws, TBL_USAGE_BASELINE, showError:=False)
        If Not tbl Is Nothing And Not tbl.DataBodyRange Is Nothing Then
            For Each row In tbl.ListRows
                intervalCode = Trim$(CStr(row.Range.Cells(1, 1).Value))
                taskCode = Trim$(CStr(row.Range.Cells(1, 2).Value))
                description = Trim$(CStr(row.Range.Cells(1, 3).Value))
                baselineValue = row.Range.Cells(1, 4).Value
                formatType = ResolveUsageFormatType(intervalCode, taskCode, _
                    CStr(row.Range.Cells(1, 5).Value), description)
                row.Range.Cells(1, 5).Value = formatType
                If IsSecondsCountInterval(intervalCode) Then
                    WriteUsageBaselineCell row.Range.Cells(1, 4), _
                        NormalizeSecondsCountValue(baselineValue), formatType
                Else
                    WriteUsageBaselineCell row.Range.Cells(1, 4), baselineValue, formatType, _
                        CStr(row.Range.Cells(1, 4).Text)
                End If
            Next row
        End If
    End If

    Set ws = GetWorksheet(SHEET_DATA_USAGE_CURRENT, showError:=False)
    If Not ws Is Nothing Then
        Set tbl = GetTable(ws, TBL_USAGE_CURRENT, showError:=False)
        If Not tbl Is Nothing And Not tbl.DataBodyRange Is Nothing Then
            For Each row In tbl.ListRows
                intervalCode = Trim$(CStr(row.Range.Cells(1, 1).Value))
                taskCode = Trim$(CStr(row.Range.Cells(1, 2).Value))
                formatType = LookupUsageFormatMetadata(intervalCode, taskCode, description)
                If IsSecondsCountInterval(intervalCode) Then
                    WriteUsageCurrentCell row.Range.Cells(1, 3), _
                        NormalizeSecondsCountValue(row.Range.Cells(1, 3).Value), formatType
                ElseIf IsCorruptUsageStoredValue(row.Range.Cells(1, 3).Value, formatType) Then
                    baselineValue = LookupUsageBaselineValue(intervalCode, taskCode)
                    If Not IsEmpty(baselineValue) Then
                        WriteUsageCurrentCell row.Range.Cells(1, 3), baselineValue, formatType
                    Else
                        WriteUsageCurrentCell row.Range.Cells(1, 3), row.Range.Cells(1, 3).Value, _
                            formatType, CStr(row.Range.Cells(1, 3).Text)
                    End If
                Else
                    WriteUsageCurrentCell row.Range.Cells(1, 3), row.Range.Cells(1, 3).Value, _
                        formatType, CStr(row.Range.Cells(1, 3).Text)
                End If
            Next row
        End If
    End If

    Set ws = GetWorksheet(SHEET_DATA_USAGE_LOG, showError:=False)
    If Not ws Is Nothing Then
        Set tbl = GetTable(ws, TBL_USAGE_LOG, showError:=False)
        If Not tbl Is Nothing And Not tbl.DataBodyRange Is Nothing Then
            For Each row In tbl.ListRows
                intervalCode = Trim$(CStr(row.Range.Cells(1, 3).Value))
                taskCode = Trim$(CStr(row.Range.Cells(1, 4).Value))
                formatType = LookupUsageFormatMetadata(intervalCode, taskCode, description)
                WriteFormattedUsageValue row.Range.Cells(1, 5), row.Range.Cells(1, 5).Value, _
                    formatType, CStr(row.Range.Cells(1, 5).Text)
            Next row
        End If
    End If
End Sub

Private Function LookupUsageFormatMetadata(ByVal intervalCode As String, _
                                           ByVal taskCode As String, _
                                           ByRef description As String) As String
    Dim ws As Worksheet
    Dim tbl As ListObject
    Dim row As ListRow

    description = ""
    LookupUsageFormatMetadata = ResolveUsageFormatType(intervalCode, taskCode)

    Set ws = GetWorksheet(SHEET_DATA_USAGE_BASELINE, showError:=False)
    If ws Is Nothing Then Exit Function

    Set tbl = GetTable(ws, TBL_USAGE_BASELINE, showError:=False)
    If tbl Is Nothing Or tbl.DataBodyRange Is Nothing Then Exit Function

    For Each row In tbl.ListRows
        If StrComp(CStr(row.Range.Cells(1, 1).Value), intervalCode, vbTextCompare) = 0 And _
           StrComp(CStr(row.Range.Cells(1, 2).Value), taskCode, vbTextCompare) = 0 Then
            description = Trim$(CStr(row.Range.Cells(1, 3).Value))
            LookupUsageFormatMetadata = ResolveUsageFormatType(intervalCode, taskCode, _
                CStr(row.Range.Cells(1, 5).Value), description)
            Exit Function
        End If
    Next row
End Function

Private Function LookupUsageBaselineValue(ByVal intervalCode As String, _
                                          ByVal taskCode As String) As Variant
    Dim ws As Worksheet
    Dim tbl As ListObject
    Dim row As ListRow

    Set ws = GetWorksheet(SHEET_DATA_USAGE_BASELINE, showError:=False)
    If ws Is Nothing Then Exit Function

    Set tbl = GetTable(ws, TBL_USAGE_BASELINE, showError:=False)
    If tbl Is Nothing Or tbl.DataBodyRange Is Nothing Then Exit Function

    For Each row In tbl.ListRows
        If StrComp(CStr(row.Range.Cells(1, 1).Value), intervalCode, vbTextCompare) = 0 And _
           StrComp(CStr(row.Range.Cells(1, 2).Value), taskCode, vbTextCompare) = 0 Then
            LookupUsageBaselineValue = row.Range.Cells(1, 4).Value
            Exit Function
        End If
    Next row
End Function

Private Function IsCorruptUsageStoredValue(ByVal value As Variant, _
                                           ByVal formatType As String) As Boolean
    If Not IsNumeric(value) Then Exit Function

    Select Case UCase$(Trim$(formatType))
        Case FORMAT_TYPE_MINUTES
            ' Normalised m:ss values are tiny day fractions; large numbers are corrupt.
            IsCorruptUsageStoredValue = (CDbl(value) >= 1#)
    End Select
End Function

' 1E/1G/2E/2G hold a decimal second count (e.g. 3.38), matching original Item Usage.
Private Function NormalizeSecondsCountValue(ByVal value As Variant) As Double
    Dim v As Double

    If Not IsNumeric(value) Then
        NormalizeSecondsCountValue = 0#
        Exit Function
    End If

    v = Abs(CDbl(value))
    If v <= 10000# Then
        NormalizeSecondsCountValue = CDbl(value)
        Exit Function
    End If

    ' Undo a prior bad [m]:ss repair that re-parsed corrupt display text (e.g. 3.38 -> 730080).
    If v / 216000# <= 10000# Then
        NormalizeSecondsCountValue = CDbl(value) / 216000#
        Exit Function
    End If

    NormalizeSecondsCountValue = CDbl(value)
End Function

Private Sub ApplyColumnFormat(ByVal tbl As ListObject, _
                              ByVal columnName As String, _
                              ByVal numberFormat As String)
    Dim col As ListColumn

    If tbl Is Nothing Then Exit Sub

    For Each col In tbl.ListColumns
        If StrComp(col.Name, columnName, vbTextCompare) = 0 Then
            If Not col.DataBodyRange Is Nothing Then
                col.DataBodyRange.NumberFormat = numberFormat
            End If
            Exit Sub
        End If
    Next col
End Sub

Private Sub RepairTableSerialColumn(ByVal sheetName As String, _
                                    ByVal tableName As String, _
                                    ByVal columnName As String)
    Dim ws As Worksheet
    Dim tbl As ListObject
    Dim colIndex As Long
    Dim row As ListRow
    Dim cell As Range

    Set ws = GetWorksheet(sheetName, showError:=False)
    If ws Is Nothing Then Exit Sub

    Set tbl = GetTable(ws, tableName, showError:=False)
    If tbl Is Nothing Or tbl.DataBodyRange Is Nothing Then Exit Sub

    colIndex = GetTableColumnIndex(tbl, columnName)
    If colIndex = 0 Then Exit Sub

    For Each row In tbl.ListRows
        Set cell = row.Range.Cells(1, colIndex)
        If Not IsEmpty(cell.Value) Then
            WriteSerialCell cell, cell.Value, cell.Text
        End If
    Next row
End Sub
