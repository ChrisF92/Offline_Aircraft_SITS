Option Explicit

' ========================================================================
' SETUP WIZARD MODULE — Offline Aircraft SITS v2 (Phase 1)
' ========================================================================
' One-time guided setup: tail number, SITS import, airframe + engine usage,
' then commit merged usage baseline and mark setup complete.
' ========================================================================

Public Sub RunSetupWizard()
    Const PROC_NAME As String = "RunSetupWizard"

    Dim tailNumber As String
    Dim sitsCount As Long
    Dim airframeCount As Long
    Dim engineCount As Long
    Dim usageCount As Long
    Dim summary As String
    Dim userName As String
    Dim failReason As String
    Dim errNum As Long
    Dim errDesc As String
    Dim protectionSuspended As Boolean
    Dim busyStarted As Boolean

    On Error GoTo ErrorHandler

    If Not ValidateWorkbookStructure() Then Exit Sub

    If IsMasterTemplate() Then
        If MsgBox("This file is marked as the master template." & vbCrLf & vbCrLf & _
                  "Best practice: File > Save As a new workbook named for the aircraft tail," & _
                  " then run Setup Wizard on that copy." & vbCrLf & vbCrLf & _
                  "Continue setup on THIS workbook anyway?", _
                  vbQuestion + vbYesNo + vbDefaultButton2, "Setup Wizard") <> vbYes Then
            Exit Sub
        End If
    End If

    If IsSetupComplete() Then
        If MsgBox("This workbook has already been set up." & vbCrLf & vbCrLf & _
                  "Re-running the wizard will replace SITS and usage baseline data." & vbCrLf & _
                  "Continue?", vbQuestion + vbYesNo + vbDefaultButton2, "Setup Wizard") <> vbYes Then
            Exit Sub
        End If
    End If

    tailNumber = PromptTailNumber()
    If Len(tailNumber) = 0 Then Exit Sub

    SetConfigValue CFG_TAIL_NUMBER, tailNumber
    UpdateDashboardTailDisplay tailNumber

    If MsgBox("Step 2 of 5 — SITS Long Forecast" & vbCrLf & vbCrLf & _
              "Select the GOLDesp Long Forecast export for " & tailNumber & ".", _
              vbInformation + vbOKCancel, "Setup Wizard") <> vbOK Then Exit Sub

    ' Import shows frmProgress after the file picker (keeps focus stable).
    sitsCount = ImportSITSFromFile(, False, True)
    If sitsCount < 0 Then
        MsgBox "Setup cancelled — SITS import failed or was not selected.", vbExclamation, "Setup Wizard"
        Exit Sub
    End If

    BeginUsageImportSession

    If MsgBox("Step 3 of 5 — Airframe usage" & vbCrLf & vbCrLf & _
              "Select the airframe Item Usage export for " & tailNumber & ".", _
              vbInformation + vbOKCancel, "Setup Wizard") <> vbOK Then
        ResetUsageImportSession
        Exit Sub
    End If

    airframeCount = ImportUsageAirframeFromFile(, False)
    If airframeCount < 0 Then
        ResetUsageImportSession
        MsgBox "Setup cancelled — airframe usage import failed or was not selected.", _
               vbExclamation, "Setup Wizard"
        Exit Sub
    End If

    If MsgBox("Step 4 of 5 — Engine usage" & vbCrLf & vbCrLf & _
              "Select the engine usage export for " & tailNumber & ".", _
              vbInformation + vbOKCancel, "Setup Wizard") <> vbOK Then
        ResetUsageImportSession
        Exit Sub
    End If

    engineCount = ImportUsageEngineFromFile(, False)
    If engineCount < 0 Then
        ResetUsageImportSession
        MsgBox "Setup cancelled — engine usage import failed or was not selected.", _
               vbExclamation, "Setup Wizard"
        Exit Sub
    End If

    usageCount = StagedUsageRowCount()
    summary = "Step 5 of 5 — Confirm setup" & vbCrLf & vbCrLf & _
              "Tail: " & tailNumber & vbCrLf & _
              "SITS tasks imported: " & sitsCount & vbCrLf & _
              "Airframe usage rows staged: " & airframeCount & vbCrLf & _
              "Engine usage rows staged: " & engineCount & vbCrLf & _
              "Merged usage baseline rows: " & usageCount & vbCrLf & vbCrLf & _
              "Commit these imports and mark setup complete?"

    If MsgBox(summary, vbQuestion + vbYesNo, "Setup Wizard") <> vbYes Then
        ResetUsageImportSession
        Exit Sub
    End If

    ' One busy session for commit + refresh so Audit/ListRows cannot leave the
    ' user on a jumping sheet (looks like a crash without the progress form).
    BeginBusyUI "Committing usage baseline…"
    busyStarted = True

    If Not CommitUsageImport(False, failReason) Then
        EndBusyUI
        busyStarted = False
        ResetUsageImportSession
        If Len(failReason) = 0 Then failReason = "Usage validation or commit failed."
        MsgBox "Setup cancelled — usage validation or commit failed." & vbCrLf & vbCrLf & _
               failReason, vbExclamation, "Setup Wizard"
        Exit Sub
    End If

    userName = Environ$("USERNAME")
    If Len(userName) = 0 Then userName = "Unknown"

    ' Keep sheets unlocked through config writes, snapshot, view refresh, and
    ' Dashboard button swap. Nested Suspend/Resume in helpers must not re-lock
    ' before Buttons.Add / CheckBoxes.Add (that surfaced as a bogus "error 0").
    SuspendWorkbookProtection
    protectionSuspended = True

    UpdateBusyUI "Marking setup complete…"
    SetConfigValue CFG_SETUP_COMPLETE, True
    SetConfigValue CFG_SETUP_DATE, Now
    SetConfigValue CFG_SETUP_BY, userName
    SetConfigValue CFG_MASTER_TEMPLATE, False
    ' Re-assert tail after imports in case any helper touched config.
    SetConfigValue CFG_TAIL_NUMBER, tailNumber

    ResetUsageImportSession
    LogSetupComplete tailNumber, sitsCount, usageCount

    UpdateBusyUI "Refreshing usage…"
    SnapshotUsageForSITS
    RefreshUsageView

    UpdateBusyUI "Refreshing SITS…"
    RefreshSITS

    UpdateBusyUI "Updating Dashboard…"
    RefreshDashboardAfterSetup

    ResumeWorkbookProtection
    protectionSuspended = False

    EndBusyUI
    busyStarted = False

    MsgBox "Setup complete for " & tailNumber & "." & vbCrLf & vbCrLf & _
           "SITS tasks: " & sitsCount & vbCrLf & _
           "Usage baseline rows: " & usageCount & vbCrLf & vbCrLf & _
           "Save the workbook before closing.", vbInformation, "Setup Wizard"
    Exit Sub

ErrorHandler:
    errNum = Err.Number
    errDesc = Err.Description
    On Error Resume Next
    If busyStarted Then EndBusyUI
    ResetUsageImportSession
    If protectionSuspended Then ForceResumeWorkbookProtection
    On Error GoTo 0

    If errNum = 0 And Len(errDesc) = 0 Then
        errDesc = "Unknown error after import commit (protection/button refresh)."
    End If

    LogError PROC_NAME, errNum, errDesc
    MsgBox "Setup Wizard error: " & errNum & " - " & errDesc, vbCritical, "Setup Wizard"
End Sub

Private Function PromptTailNumber() As String
    Dim currentTail As String
    Dim enteredTail As String

    currentTail = GetTailNumber()
    If Len(currentTail) = 0 Then currentTail = "ZZ383"

    enteredTail = InputBox("Step 1 of 5 — Aircraft tail number" & vbCrLf & vbCrLf & _
                           "Enter the tail number for this workbook:", _
                           "Setup Wizard", currentTail)

    enteredTail = UCase$(Trim$(enteredTail))
    If Len(enteredTail) = 0 Then Exit Function

    If MsgBox("Use tail number " & enteredTail & "?", vbQuestion + vbYesNo, "Setup Wizard") <> vbYes Then
        Exit Function
    End If

    PromptTailNumber = enteredTail
End Function

Private Sub UpdateDashboardTailDisplay(ByVal tailNumber As String)
    Dim ws As Worksheet

    Set ws = GetWorksheet(SHEET_SITS, showError:=False)
    If Not ws Is Nothing Then
        ws.Range("B2").Value = tailNumber
    End If

    Set ws = GetWorksheet(SHEET_USAGE, showError:=False)
    If Not ws Is Nothing Then
        ws.Range("B2").Value = tailNumber
    End If
End Sub

Private Sub RefreshDashboardAfterSetup()
    ' Do not rebuild layout after setup — preserve Master column widths / buttons.
    ' EnsureDashboardActionButtons swaps Setup Wizard for the post-setup user actions.
    EnsureDashboardActionButtons
    RefreshDashboardStatusCounts
    RepairDashboardConfigFormulas
End Sub
