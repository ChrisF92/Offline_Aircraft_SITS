Option Explicit

' ========================================================================
' SITS EDIT MODULE — Offline Aircraft SITS v2
' ========================================================================
' Manual Add Task / Remove Task from the SITS view.
' Add: sequential InputBox prompts for import-required fields, validate,
'      confirm summary, insert into tblSITS, recalc + refresh (forecast
'      filter decides visibility).
' Remove: reuse Selector checkboxes; confirm irreversible delete of the
'         whole selection (listed); delete from tblSITS (+ orphan WP links).
' Manual Update: zero-interval tasks only — step-by-step InputBox for due
'         date or hours/number until due (workbook cells stay protected).
' ========================================================================

Private Type SITSManualTaskDraft
    IntervalTypeCode As String
    LifeRemainingText As String
    TaskCode As String
    TaskSequence As String
    TaskDescription As String
    Serial As String
    ActualIntervalText As String
    ExtensionText As String
    EndItemSerial As String
End Type

Public Sub RunAddSITSTask()
    AddSITSTaskManually
End Sub

Public Sub RunRemoveSITSTasks()
    RemoveSelectedSITSTasks
End Sub

Public Sub RunManualUpdateSITSTasks()
    ManualUpdateSelectedZeroIntervalTasks
End Sub

Public Function AddSITSTaskManually() As Boolean
    Const PROC_NAME As String = "AddSITSTaskManually"

    Dim draft As SITSManualTaskDraft
    Dim failReason As String
    Dim tbl As ListObject
    Dim newRow As ListRow
    Dim rowId As String
    Dim tailNumber As String

    AddSITSTaskManually = False

    If Not ValidateSetupReady() Then Exit Function

    Set tbl = GetSITSEditTable(True)
    If tbl Is Nothing Then Exit Function
    If Not ValidateSITSColumns(tbl) Then Exit Function

    tailNumber = GetTailNumber()

    If Not PromptManualSITSTaskFields(draft, tailNumber, failReason) Then
        If Len(failReason) > 0 And StrComp(failReason, "Cancelled.", vbTextCompare) <> 0 Then
            MsgBox failReason, vbExclamation, APP_NAME
        End If
        Exit Function
    End If

    If Not ConfirmManualSITSTask(draft) Then Exit Function

    On Error GoTo ErrorHandler
    BeginBusyUI "Adding SITS task…"
    SetExcelState "disable"
    SuspendWorkbookProtection

    rowId = GenerateNextSITSRowId(tbl)
    Set newRow = tbl.ListRows.Add
    WriteManualSITSTaskRow tbl, newRow, draft, rowId

    UpdateBusyUI "Recalculating SITS…"
    ApplyWorkbookDataFormats
    RecalculateSITS skipTableFormats:=True
    UpdateBusyUI "Refreshing SITS view…"
    RefreshSITSView

    LogAction "SITS_TASK_ADDED", "", _
              "RowID=" & rowId & "; Code=" & draft.TaskCode & _
              "; Seq=" & draft.TaskSequence & "; Interval=" & draft.IntervalTypeCode
    ResumeWorkbookProtection
    SetExcelState "enable"
    EndBusyUI

    AddSITSTaskManually = True
    MsgBox "Task added (" & rowId & ")." & vbCrLf & vbCrLf & _
           "It appears on the SITS view only if it is within the current forecast filter.", _
           vbInformation, APP_NAME
    Exit Function

ErrorHandler:
    Dim errNum As Long
    Dim errDesc As String
    errNum = Err.Number
    errDesc = Err.Description
    EndBusyUI
    ForceResumeWorkbookProtection
    SetExcelState "force-enable"
    LogError PROC_NAME, errNum, errDesc
    MsgBox "Add task error: " & errNum & " - " & errDesc, vbCritical, APP_NAME
End Function

Public Function RemoveSelectedSITSTasks() As Boolean
    Const PROC_NAME As String = "RemoveSelectedSITSTasks"

    Dim tbl As ListObject
    Dim selectedIds As Object
    Dim rowIndex As Object
    Dim rowId As Variant
    Dim sitsRow As ListRow
    Dim rowsToDelete As Collection
    Dim deleteRow As Variant
    Dim confirmList As String
    Dim deletedCount As Long
    Dim detailParts As String
    Dim colTaskCode As Long
    Dim colSeq As Long
    Dim colDesc As Long
    Dim taskCode As String
    Dim taskSeq As String
    Dim taskDesc As String
    Dim line As String
    Dim maxLines As Long
    Dim listed As Long

    RemoveSelectedSITSTasks = False

    If Not ValidateSetupReady() Then Exit Function

    Set tbl = GetSITSEditTable(True)
    If tbl Is Nothing Then Exit Function
    If Not ValidateSITSColumns(tbl) Then Exit Function

    Set rowIndex = BuildSITSRowIdIndex(tbl)
    Set selectedIds = CollectSelectedFromSITSView(tbl, rowIndex)

    If selectedIds Is Nothing Or selectedIds.Count = 0 Then
        MsgBox "Tick one or more tasks on the SITS view, then click Remove Task.", _
               vbExclamation, APP_NAME
        Exit Function
    End If

    colTaskCode = GetTableColumnIndex(tbl, HDR_TASK_CODE)
    colSeq = GetTableColumnIndex(tbl, HDR_TASK_SEQUENCE)
    colDesc = GetTableColumnIndex(tbl, HDR_TASK_DESCRIPTION)

    maxLines = 25
    listed = 0
    confirmList = ""
    detailParts = ""

    For Each rowId In selectedIds.Keys
        Set sitsRow = ResolveSITSRowById(rowIndex, tbl, CStr(rowId))
        If sitsRow Is Nothing Then GoTo NextConfirmRow

        taskCode = ""
        taskSeq = ""
        taskDesc = ""
        If colTaskCode > 0 Then taskCode = Trim$(CStr(sitsRow.Range.Cells(1, colTaskCode).Value))
        If colSeq > 0 Then taskSeq = Trim$(CStr(sitsRow.Range.Cells(1, colSeq).Value))
        If colDesc > 0 Then taskDesc = Trim$(CStr(sitsRow.Range.Cells(1, colDesc).Value))

        line = "• " & CStr(rowId)
        If Len(taskCode) > 0 Then line = line & "  " & taskCode
        If Len(taskSeq) > 0 Then line = line & " / " & taskSeq
        If Len(taskDesc) > 0 Then line = line & " — " & Left$(taskDesc, 60)

        listed = listed + 1
        If listed <= maxLines Then
            If Len(confirmList) > 0 Then confirmList = confirmList & vbCrLf
            confirmList = confirmList & line
        End If

        If Len(detailParts) > 0 Then detailParts = detailParts & "; "
        detailParts = detailParts & CStr(rowId)
NextConfirmRow:
    Next rowId

    If listed = 0 Then
        MsgBox "Tick one or more tasks on the SITS view, then click Remove Task.", _
               vbExclamation, APP_NAME
        Exit Function
    End If

    If listed > maxLines Then
        confirmList = confirmList & vbCrLf & "• … and " & (listed - maxLines) & " more"
    End If

    If MsgBox( _
        "Delete " & listed & " selected task(s) from SITS?" & vbCrLf & vbCrLf & _
        confirmList & vbCrLf & vbCrLf & _
        "This cannot be undone.", _
        vbQuestion + vbYesNo + vbDefaultButton2, _
        APP_NAME) <> vbYes Then Exit Function

    On Error GoTo ErrorHandler
    BeginBusyUI "Removing SITS task(s)…"
    SetExcelState "disable"
    SuspendWorkbookProtection

    ' Delete highest ListRow.Index first so earlier indices stay valid.
    Set rowIndex = BuildSITSRowIdIndex(tbl)
    Set rowsToDelete = New Collection

    For Each rowId In selectedIds.Keys
        Set sitsRow = ResolveSITSRowById(rowIndex, tbl, CStr(rowId))
        If Not sitsRow Is Nothing Then
            InsertListRowDescending rowsToDelete, sitsRow
        End If
        RemoveWPTaskLinksForSITSRowId CStr(rowId)
    Next rowId

    For Each deleteRow In rowsToDelete
        deleteRow.Delete
        deletedCount = deletedCount + 1
    Next deleteRow

    UpdateBusyUI "Refreshing SITS view…"
    RefreshSITSView

    LogAction "SITS_TASKS_REMOVED", "", _
              "Count=" & deletedCount & "; " & Left$(detailParts, 180)
    ResumeWorkbookProtection
    SetExcelState "enable"
    EndBusyUI

    RemoveSelectedSITSTasks = True
    MsgBox "Removed " & deletedCount & " task(s) from SITS.", vbInformation, APP_NAME
    Exit Function

ErrorHandler:
    Dim errNum As Long
    Dim errDesc As String
    errNum = Err.Number
    errDesc = Err.Description
    EndBusyUI
    ForceResumeWorkbookProtection
    SetExcelState "force-enable"
    LogError PROC_NAME, errNum, errDesc
    MsgBox "Remove task error: " & errNum & " - " & errDesc, vbCritical, APP_NAME
End Function

' Manual Update — zero-interval life remaining (protected sheet)
' ------------------------------------------------------------------------

Public Function ManualUpdateSelectedZeroIntervalTasks() As Boolean
    Const PROC_NAME As String = "ManualUpdateSelectedZeroIntervalTasks"

    Dim tbl As ListObject
    Dim selectedIds As Object
    Dim rowIndex As Object
    Dim rowId As Variant
    Dim sitsRow As ListRow
    Dim failReason As String
    Dim updatedCount As Long
    Dim skippedCount As Long
    Dim taskCode As String
    Dim intervalCode As String
    Dim lifeText As String
    Dim cancelled As Boolean

    ManualUpdateSelectedZeroIntervalTasks = False

    If Not ValidateSetupReady() Then Exit Function

    Set tbl = GetSITSEditTable(True)
    If tbl Is Nothing Then Exit Function
    If Not ValidateSITSColumns(tbl) Then Exit Function
    EnsureWorkPackageSITSColumns tbl
    EnsureSITSBaselineColumns tbl

    Set rowIndex = BuildSITSRowIdIndex(tbl)
    Set selectedIds = CollectSelectedFromSITSView(tbl, rowIndex)

    If selectedIds Is Nothing Or selectedIds.Count = 0 Then
        MsgBox "Tick one or more zero-interval tasks on the SITS view, then click Manual Update.", _
               vbExclamation, APP_NAME
        Exit Function
    End If

    If Not ValidateSelectionForManualUpdate(selectedIds, tbl, rowIndex, failReason) Then
        MsgBox failReason, vbExclamation, APP_NAME
        Exit Function
    End If

    On Error GoTo ErrorHandler

    For Each rowId In selectedIds.Keys
        Set sitsRow = ResolveSITSRowById(rowIndex, tbl, CStr(rowId))
        If sitsRow Is Nothing Then
            skippedCount = skippedCount + 1
            GoTo NextManualTask
        End If

        taskCode = Trim$(CStr(sitsRow.Range.Cells(1, GetTableColumnIndex(tbl, HDR_TASK_CODE)).Value))
        intervalCode = UCase$(Trim$(CStr( _
            sitsRow.Range.Cells(1, GetTableColumnIndex(tbl, HDR_INTERVAL_TYPE_CODE)).Value)))

        lifeText = PromptManualLifeRemainingForTask(taskCode, intervalCode, cancelled)
        If cancelled Then
            If updatedCount = 0 Then Exit Function
            Exit For
        End If

        If Len(Trim$(lifeText)) = 0 Then
            skippedCount = skippedCount + 1
            GoTo NextManualTask
        End If

        failReason = ""
        If Not ValidateLifeRemainingText(intervalCode, lifeText, failReason) Then
            MsgBox failReason & vbCrLf & vbCrLf & "Skipping " & taskCode & ".", _
                   vbExclamation, APP_NAME
            skippedCount = skippedCount + 1
            GoTo NextManualTask
        End If

        If MsgBox("Update " & taskCode & " (" & intervalCode & ")?" & vbCrLf & vbCrLf & _
                  "New life remaining / due: " & lifeText, _
                  vbQuestion + vbYesNo, APP_NAME) <> vbYes Then
            skippedCount = skippedCount + 1
            GoTo NextManualTask
        End If

        BeginBusyUI "Updating " & taskCode & "…"
        SetExcelState "disable"
        SuspendWorkbookProtection

        ApplyManualUpdateLifeRemaining sitsRow, tbl, intervalCode, lifeText
        ClearManualUpdateFlag sitsRow, tbl
        ClearTaskSelectorEdit sitsRow, tbl

        updatedCount = updatedCount + 1

        ResumeWorkbookProtection
        SetExcelState "enable"
        EndBusyUI
NextManualTask:
    Next rowId

    If updatedCount = 0 Then
        MsgBox "No tasks were updated.", vbInformation, APP_NAME
        Exit Function
    End If

    BeginBusyUI "Recalculating SITS…"
    SetExcelState "disable"
    SuspendWorkbookProtection
    RecalculateSITS skipTableFormats:=True
    UpdateBusyUI "Refreshing SITS view…"
    RefreshSITSView
    UpdateBusyUI "Updating Dashboard…"
    RefreshDashboardStatusCounts
    ResumeWorkbookProtection
    SetExcelState "enable"
    EndBusyUI

    LogAction "SITS_MANUAL_UPDATE", "", _
              "Updated=" & updatedCount & "; Skipped=" & skippedCount

    ManualUpdateSelectedZeroIntervalTasks = True
    MsgBox "Manual update applied to " & updatedCount & " task(s)." & _
           IIf(skippedCount > 0, vbCrLf & skippedCount & " skipped/cancelled.", ""), _
           vbInformation, APP_NAME
    Exit Function

ErrorHandler:
    Dim errNum As Long
    Dim errDesc As String
    errNum = Err.Number
    errDesc = Err.Description
    EndBusyUI
    ForceResumeWorkbookProtection
    SetExcelState "force-enable"
    LogError PROC_NAME, errNum, errDesc
    MsgBox "Manual update error: " & errNum & " - " & errDesc, vbCritical, APP_NAME
End Function

Private Function ValidateSelectionForManualUpdate(ByVal selectedIds As Object, _
                                                  ByVal tbl As ListObject, _
                                                  ByVal rowIndex As Object, _
                                                  ByRef failReason As String) As Boolean
    Dim rowId As Variant
    Dim sitsRow As ListRow
    Dim taskCode As String
    Dim inWorkOrn As String
    Dim haveAny As Boolean

    failReason = ""
    haveAny = False

    For Each rowId In selectedIds.Keys
        Set sitsRow = ResolveSITSRowById(rowIndex, tbl, CStr(rowId))
        If sitsRow Is Nothing Then
            failReason = "A selected SITS task is no longer present."
            Exit Function
        End If

        taskCode = Trim$(CStr(sitsRow.Range.Cells(1, GetTableColumnIndex(tbl, HDR_TASK_CODE)).Value))
        inWorkOrn = ""
        If GetTableColumnIndex(tbl, HDR_ORN_INWORK) > 0 Then
            inWorkOrn = Trim$(CStr(sitsRow.Range.Cells(1, GetTableColumnIndex(tbl, HDR_ORN_INWORK)).Value))
        End If
        If Len(inWorkOrn) > 0 Then
            failReason = "Task " & taskCode & " is in open work package " & inWorkOrn & "." & vbCrLf & _
                         "Close or remove it from the work package before manual update."
            Exit Function
        End If

        If Not IsZeroActualIntervalRow(sitsRow, tbl) Then
            failReason = "Manual Update is only for zero Actual Interval tasks." & vbCrLf & _
                         "Task " & taskCode & " does not have Actual Interval = 0."
            Exit Function
        End If

        haveAny = True
    Next rowId

    If Not haveAny Then
        failReason = "No valid tasks selected."
        Exit Function
    End If

    ValidateSelectionForManualUpdate = True
End Function

Private Function IsZeroActualIntervalRow(ByVal sitsRow As ListRow, ByVal tbl As ListObject) As Boolean
    Dim colInterval As Long
    Dim value As Variant

    colInterval = GetTableColumnIndex(tbl, HDR_ACTUAL_INTERVAL)
    If colInterval = 0 Then Exit Function

    value = sitsRow.Range.Cells(1, colInterval).Value
    If IsEmpty(value) Or Len(Trim$(CStr(value))) = 0 Then Exit Function

    If IsNumeric(value) Then
        IsZeroActualIntervalRow = (CDbl(value) = 0#)
    End If
End Function

Private Function PromptManualLifeRemainingForTask(ByVal taskCode As String, _
                                                  ByVal intervalCode As String, _
                                                  ByRef cancelled As Boolean) As String
    Dim hint As String
    Dim entered As String
    Dim fieldFail As String

    cancelled = False
    hint = ManualUpdatePromptHint(intervalCode)

    Do
        entered = PromptFieldValue( _
            "Life Remaining / Due", _
            "Manual update for " & taskCode & " (" & intervalCode & ")." & vbCrLf & vbCrLf & hint, _
            "", cancelled)
        If cancelled Then Exit Function
        If Len(Trim$(entered)) = 0 Then
            MsgBox "A value is required.", vbExclamation, APP_NAME
            GoTo NextPrompt
        End If
        fieldFail = ""
        If ValidateLifeRemainingText(intervalCode, Trim$(entered), fieldFail) Then
            PromptManualLifeRemainingForTask = Trim$(entered)
            Exit Function
        End If
        MsgBox fieldFail, vbExclamation, APP_NAME
NextPrompt:
    Loop
End Function

Private Function ManualUpdatePromptHint(ByVal intervalCode As String) As String
    If UsesHourFormatForManualEntry(intervalCode) Then
        ManualUpdatePromptHint = "Enter hours until due as h:mm (e.g. 25:30) or a decimal day-fraction."
    ElseIf UCase$(Trim$(intervalCode)) = TYPE_DD Then
        ManualUpdatePromptHint = "Enter date due as dd/mm/yyyy, or days until due as a number."
    Else
        ManualUpdatePromptHint = "Enter number until due."
    End If
End Function

Private Sub ApplyManualUpdateLifeRemaining(ByVal sitsRow As ListRow, _
                                           ByVal tbl As ListObject, _
                                           ByVal intervalCode As String, _
                                           ByVal lifeText As String)
    Dim targetLife As Double
    Dim parsedTime As Variant
    Dim parsedDate As Variant
    Dim dueDate As Date
    Dim colLRHours As Long
    Dim colLRNumber As Long
    Dim colDaysDue As Long
    Dim colBaseHours As Long
    Dim colBaseNumber As Long
    Dim colBaseDue As Long

    colLRHours = GetTableColumnIndex(tbl, HDR_HOURS_LIFE_REMAINING)
    colLRNumber = GetTableColumnIndex(tbl, HDR_NUMBER_LIFE_REMAINING)
    colDaysDue = GetTableColumnIndex(tbl, HDR_DAYS_WHEN_DUE)
    colBaseHours = GetTableColumnIndex(tbl, HDR_BASELINE_LR_HOURS)
    colBaseNumber = GetTableColumnIndex(tbl, HDR_BASELINE_LR_NUMBER)
    colBaseDue = GetTableColumnIndex(tbl, HDR_BASELINE_DAYS_DUE)

    If UsesHourFormatForManualEntry(intervalCode) Then
        parsedTime = ParseTimeUsageValue(lifeText, lifeText)
        If Not IsEmpty(parsedTime) Then
            targetLife = CDbl(parsedTime)
        Else
            targetLife = CDbl(lifeText)
        End If

        If colLRHours > 0 Then
            WriteFormattedUsageValue sitsRow.Range.Cells(1, colLRHours), targetLife, FORMAT_TYPE_HOURS
        End If
        If colLRNumber > 0 Then sitsRow.Range.Cells(1, colLRNumber).ClearContents
        If colDaysDue > 0 Then sitsRow.Range.Cells(1, colDaysDue).ClearContents
        If colBaseNumber > 0 Then sitsRow.Range.Cells(1, colBaseNumber).ClearContents
        If colBaseDue > 0 Then sitsRow.Range.Cells(1, colBaseDue).ClearContents
        AdjustManualUpdateBaseline sitsRow, tbl, intervalCode, targetLife, FORMAT_TYPE_HOURS
        Exit Sub
    End If

    If intervalCode = TYPE_DD Then
        parsedDate = ParseDMYDateValue(lifeText, lifeText)
        If IsDate(parsedDate) Then
            dueDate = CDate(parsedDate)
        Else
            dueDate = DateAdd("d", CLng(CDbl(lifeText)), Date)
        End If

        If colDaysDue > 0 Then
            sitsRow.Range.Cells(1, colDaysDue).Value = CDbl(DateDiff("d", Date, dueDate))
            sitsRow.Range.Cells(1, colDaysDue).NumberFormat = FORMAT_DECIMAL
        End If
        If colBaseDue > 0 Then
            sitsRow.Range.Cells(1, colBaseDue).Value = dueDate
            sitsRow.Range.Cells(1, colBaseDue).NumberFormat = FORMAT_DATE
        End If
        SyncProjectedDueDate sitsRow, tbl, dueDate
        If colLRHours > 0 Then sitsRow.Range.Cells(1, colLRHours).ClearContents
        If colLRNumber > 0 Then sitsRow.Range.Cells(1, colLRNumber).ClearContents
        If colBaseHours > 0 Then sitsRow.Range.Cells(1, colBaseHours).ClearContents
        If colBaseNumber > 0 Then sitsRow.Range.Cells(1, colBaseNumber).ClearContents
        Exit Sub
    End If

    targetLife = CDbl(lifeText)
    If colLRNumber > 0 Then
        sitsRow.Range.Cells(1, colLRNumber).Value = targetLife
        sitsRow.Range.Cells(1, colLRNumber).NumberFormat = FORMAT_DECIMAL
    End If
    If colLRHours > 0 Then sitsRow.Range.Cells(1, colLRHours).ClearContents
    If colDaysDue > 0 Then sitsRow.Range.Cells(1, colDaysDue).ClearContents
    If colBaseHours > 0 Then sitsRow.Range.Cells(1, colBaseHours).ClearContents
    If colBaseDue > 0 Then sitsRow.Range.Cells(1, colBaseDue).ClearContents
    AdjustManualUpdateBaseline sitsRow, tbl, intervalCode, targetLife, FORMAT_TYPE_NUMBER
End Sub

Private Sub AdjustManualUpdateBaseline(ByVal sitsRow As ListRow, _
                                       ByVal tbl As ListObject, _
                                       ByVal intervalCode As String, _
                                       ByVal targetLife As Double, _
                                       ByVal formatType As String)
    Dim currentUsage As Variant
    Dim snapshotUsage As Variant
    Dim usageDelta As Double
    Dim adjusted As Double
    Dim colBaseHours As Long
    Dim colBaseNumber As Long

    colBaseHours = GetTableColumnIndex(tbl, HDR_BASELINE_LR_HOURS)
    colBaseNumber = GetTableColumnIndex(tbl, HDR_BASELINE_LR_NUMBER)

    currentUsage = LookupEditCurrentUsage(intervalCode)
    snapshotUsage = LookupEditSnapshotUsage(intervalCode)

    If IsEmpty(currentUsage) Or IsEmpty(snapshotUsage) Then
        ' No usage delta available — store target as baseline so LR stays put until sorties.
        If formatType = FORMAT_TYPE_HOURS Then
            If colBaseHours > 0 Then
                WriteFormattedUsageValue sitsRow.Range.Cells(1, colBaseHours), targetLife, FORMAT_TYPE_HOURS
            End If
        ElseIf colBaseNumber > 0 Then
            sitsRow.Range.Cells(1, colBaseNumber).Value = targetLife
            sitsRow.Range.Cells(1, colBaseNumber).NumberFormat = FORMAT_DECIMAL
        End If
        Exit Sub
    End If

    If formatType = FORMAT_TYPE_HOURS Then
        usageDelta = UsageValueAsDayFraction(currentUsage, FORMAT_TYPE_HOURS) - _
                     UsageValueAsDayFraction(snapshotUsage, FORMAT_TYPE_HOURS)
        adjusted = targetLife + usageDelta
        If colBaseHours > 0 Then
            WriteFormattedUsageValue sitsRow.Range.Cells(1, colBaseHours), adjusted, FORMAT_TYPE_HOURS
        End If
    Else
        If Not IsNumeric(currentUsage) Or Not IsNumeric(snapshotUsage) Then
            If colBaseNumber > 0 Then
                sitsRow.Range.Cells(1, colBaseNumber).Value = targetLife
                sitsRow.Range.Cells(1, colBaseNumber).NumberFormat = FORMAT_DECIMAL
            End If
            Exit Sub
        End If
        usageDelta = CDbl(currentUsage) - CDbl(snapshotUsage)
        adjusted = targetLife + usageDelta
        If colBaseNumber > 0 Then
            sitsRow.Range.Cells(1, colBaseNumber).Value = adjusted
            sitsRow.Range.Cells(1, colBaseNumber).NumberFormat = FORMAT_DECIMAL
        End If
    End If
End Sub

Private Function LookupEditCurrentUsage(ByVal intervalCode As String) As Variant
    Dim ws As Worksheet
    Dim tbl As ListObject
    Dim row As ListRow

    Set ws = GetWorksheet(SHEET_DATA_USAGE_CURRENT, showError:=False)
    If ws Is Nothing Then Exit Function
    Set tbl = GetTable(ws, TBL_USAGE_CURRENT, showError:=False)
    If tbl Is Nothing Or tbl.DataBodyRange Is Nothing Then Exit Function

    For Each row In tbl.ListRows
        If StrComp(CStr(row.Range.Cells(1, 1).Value), intervalCode, vbTextCompare) = 0 Then
            LookupEditCurrentUsage = row.Range.Cells(1, 3).Value
            Exit Function
        End If
    Next row
End Function

Private Function LookupEditSnapshotUsage(ByVal intervalCode As String) As Variant
    Dim ws As Worksheet
    Dim tbl As ListObject
    Dim row As ListRow

    Set ws = GetWorksheet(SHEET_CONFIG, showError:=False)
    If ws Is Nothing Then Exit Function
    Set tbl = GetTable(ws, TBL_USAGE_SNAPSHOT, showError:=False)
    If tbl Is Nothing Or tbl.DataBodyRange Is Nothing Then Exit Function

    For Each row In tbl.ListRows
        If StrComp(CStr(row.Range.Cells(1, 1).Value), intervalCode, vbTextCompare) = 0 Then
            LookupEditSnapshotUsage = row.Range.Cells(1, 2).Value
            Exit Function
        End If
    Next row
End Function

Private Sub ClearManualUpdateFlag(ByVal sitsRow As ListRow, ByVal tbl As ListObject)
    Dim colFlag As Long
    colFlag = GetTableColumnIndex(tbl, HDR_NEEDS_MANUAL_UPDATE)
    If colFlag > 0 Then sitsRow.Range.Cells(1, colFlag).Value = False
End Sub

Private Sub ClearTaskSelectorEdit(ByVal sitsRow As ListRow, ByVal tbl As ListObject)
    Dim colSelector As Long
    colSelector = GetTableColumnIndex(tbl, HDR_SELECTOR)
    If colSelector > 0 Then sitsRow.Range.Cells(1, colSelector).Value = False
End Sub

Private Function PromptManualSITSTaskFields(ByRef draft As SITSManualTaskDraft, _
                                            ByVal tailNumber As String, _
                                            ByRef failReason As String) As Boolean
    Dim entered As String
    Dim normalisedTail As String
    Dim fieldFail As String
    Dim cancelled As Boolean

    failReason = ""
    normalisedTail = NormaliseTailText(tailNumber)

    Do
        entered = PromptFieldValue( _
            "Interval Type Code", _
            "Enter Interval Type Code (e.g. HH, DD, E1, E2, SM, SN).", _
            draft.IntervalTypeCode, cancelled)
        If cancelled Then failReason = "Cancelled.": Exit Function
        If Len(Trim$(entered)) = 0 Then
            MsgBox "Interval Type Code is required.", vbExclamation, APP_NAME
            GoTo NextIntervalCode
        End If
        draft.IntervalTypeCode = UCase$(Trim$(entered))
        fieldFail = ""
        If ValidateIntervalTypeCode(draft.IntervalTypeCode, fieldFail) Then Exit Do
        MsgBox fieldFail, vbExclamation, APP_NAME
NextIntervalCode:
    Loop

    Do
        entered = PromptFieldValue( _
            "Life Remaining", _
            LifeRemainingPromptHint(draft.IntervalTypeCode), _
            draft.LifeRemainingText, cancelled)
        If cancelled Then failReason = "Cancelled.": Exit Function
        If Len(Trim$(entered)) = 0 Then
            MsgBox "Life Remaining is required.", vbExclamation, APP_NAME
            GoTo NextLifeRemaining
        End If
        draft.LifeRemainingText = Trim$(entered)
        fieldFail = ""
        If ValidateLifeRemainingText(draft.IntervalTypeCode, draft.LifeRemainingText, fieldFail) Then Exit Do
        MsgBox fieldFail, vbExclamation, APP_NAME
NextLifeRemaining:
    Loop

    Do
        entered = PromptFieldValue("Task Code", "Enter Task Code.", draft.TaskCode, cancelled)
        If cancelled Then failReason = "Cancelled.": Exit Function
        If Len(Trim$(entered)) = 0 Then
            MsgBox "Task Code is required.", vbExclamation, APP_NAME
        Else
            draft.TaskCode = Trim$(entered)
            Exit Do
        End If
    Loop

    Do
        entered = PromptFieldValue("Task Sequence", "Enter Task Sequence.", draft.TaskSequence, cancelled)
        If cancelled Then failReason = "Cancelled.": Exit Function
        If Len(Trim$(entered)) = 0 Then
            MsgBox "Task Sequence is required.", vbExclamation, APP_NAME
        Else
            draft.TaskSequence = Trim$(entered)
            Exit Do
        End If
    Loop

    Do
        entered = PromptFieldValue("Task Description", "Enter Task Description.", _
                                   draft.TaskDescription, cancelled)
        If cancelled Then failReason = "Cancelled.": Exit Function
        If Len(Trim$(entered)) = 0 Then
            MsgBox "Task Description is required.", vbExclamation, APP_NAME
        Else
            draft.TaskDescription = Trim$(entered)
            Exit Do
        End If
    Loop

    Do
        entered = PromptFieldValue( _
            "Serial", _
            "Enter Serial (component / end-item serial for this task).", _
            draft.Serial, cancelled)
        If cancelled Then failReason = "Cancelled.": Exit Function
        If Len(Trim$(entered)) = 0 Then
            MsgBox "Serial is required.", vbExclamation, APP_NAME
        Else
            draft.Serial = Trim$(entered)
            Exit Do
        End If
    Loop

    Do
        entered = PromptFieldValue( _
            "Actual Interval", _
            ActualIntervalPromptHint(draft.IntervalTypeCode), _
            draft.ActualIntervalText, cancelled)
        If cancelled Then failReason = "Cancelled.": Exit Function
        If Len(Trim$(entered)) = 0 Then
            MsgBox "Actual Interval is required.", vbExclamation, APP_NAME
            GoTo NextActualInterval
        End If
        draft.ActualIntervalText = Trim$(entered)
        fieldFail = ""
        If ValidateScalarIntervalText(draft.IntervalTypeCode, draft.ActualIntervalText, _
                                      "Actual Interval", fieldFail) Then Exit Do
        MsgBox fieldFail, vbExclamation, APP_NAME
NextActualInterval:
    Loop

    Do
        entered = PromptFieldValue( _
            "Extension", _
            ExtensionPromptHint(draft.IntervalTypeCode), _
            IIf(Len(draft.ExtensionText) > 0, draft.ExtensionText, "0"), _
            cancelled)
        If cancelled Then failReason = "Cancelled.": Exit Function
        If Len(Trim$(entered)) = 0 Then
            MsgBox "Extension is required (use 0 if none).", vbExclamation, APP_NAME
            GoTo NextExtension
        End If
        draft.ExtensionText = Trim$(entered)
        fieldFail = ""
        If ValidateScalarIntervalText(draft.IntervalTypeCode, draft.ExtensionText, _
                                      "Extension", fieldFail) Then Exit Do
        MsgBox fieldFail, vbExclamation, APP_NAME
NextExtension:
    Loop

    Do
        entered = PromptFieldValue( _
            "End Item Serial", _
            "Enter End Item Serial (must match configured tail " & tailNumber & ").", _
            IIf(Len(draft.EndItemSerial) > 0, draft.EndItemSerial, normalisedTail), _
            cancelled)
        If cancelled Then failReason = "Cancelled.": Exit Function
        If Len(Trim$(entered)) = 0 Then
            MsgBox "End Item Serial is required.", vbExclamation, APP_NAME
            GoTo NextEndItem
        End If
        draft.EndItemSerial = Trim$(entered)
        If NormaliseTailText(draft.EndItemSerial) = normalisedTail Then Exit Do
        MsgBox "End Item Serial must match the configured tail (" & tailNumber & ").", _
               vbExclamation, APP_NAME
NextEndItem:
    Loop

    PromptManualSITSTaskFields = True
End Function

Private Function PromptFieldValue(ByVal fieldName As String, _
                                  ByVal promptText As String, _
                                  ByVal defaultValue As String, _
                                  ByRef cancelled As Boolean) As String
    Dim entered As String

    cancelled = False
    entered = InputBox(promptText & vbCrLf & vbCrLf & "Field: " & fieldName, _
                       APP_NAME, defaultValue)

    If StrPtr(entered) = 0 Then
        cancelled = True
        Exit Function
    End If

    PromptFieldValue = entered
End Function

Private Function ValidateIntervalTypeCode(ByVal intervalCode As String, _
                                          ByRef failReason As String) As Boolean
    If Len(Trim$(intervalCode)) = 0 Then
        failReason = "Interval Type Code is required."
        Exit Function
    End If

    ' Accept known codes and other short alphanumeric interval codes used in exports.
    If Not intervalCode Like "[A-Z0-9][A-Z0-9]*" Then
        failReason = "Interval Type Code must be letters/digits (e.g. HH, DD, E1, SM)."
        Exit Function
    End If

    If Len(intervalCode) > 8 Then
        failReason = "Interval Type Code is too long."
        Exit Function
    End If

    ValidateIntervalTypeCode = True
End Function

Private Function ValidateLifeRemainingText(ByVal intervalCode As String, _
                                           ByVal textValue As String, _
                                           ByRef failReason As String) As Boolean
    Dim parsedDate As Variant

    If UCase$(Trim$(intervalCode)) = TYPE_DD Then
        If Len(Trim$(textValue)) = 0 Then
            failReason = "Life Remaining is required."
            Exit Function
        End If
        parsedDate = ParseDMYDateValue(textValue, textValue)
        If IsDate(parsedDate) Then
            ValidateLifeRemainingText = True
            Exit Function
        End If
        If IsNumeric(textValue) Then
            ValidateLifeRemainingText = True
            Exit Function
        End If
        failReason = "Life Remaining must be a number of days or a date (dd/mm/yyyy)."
        Exit Function
    End If

    ValidateLifeRemainingText = ValidateScalarIntervalText(intervalCode, textValue, _
                                                           "Life Remaining", failReason)
End Function

Private Function ValidateScalarIntervalText(ByVal intervalCode As String, _
                                            ByVal textValue As String, _
                                            ByVal fieldName As String, _
                                            ByRef failReason As String) As Boolean
    Dim parsedTime As Variant

    If Len(Trim$(textValue)) = 0 Then
        failReason = fieldName & " is required."
        Exit Function
    End If

    If UsesHourFormatForManualEntry(intervalCode) Then
        If InStr(textValue, ":") > 0 Then
            parsedTime = ParseTimeUsageValue(textValue, textValue)
            If IsEmpty(parsedTime) Then
                failReason = fieldName & " must be a valid time (h:mm) or number."
                Exit Function
            End If
            ValidateScalarIntervalText = True
            Exit Function
        End If
    End If

    If Not IsNumeric(textValue) Then
        failReason = fieldName & " must be numeric" & _
                     IIf(UsesHourFormatForManualEntry(intervalCode), " or h:mm time", "") & "."
        Exit Function
    End If

    ValidateScalarIntervalText = True
End Function

Private Function UsesHourFormatForManualEntry(ByVal intervalCode As String) As Boolean
    UsesHourFormatForManualEntry = IsHourUsageInterval(intervalCode)
End Function

Private Function LifeRemainingPromptHint(ByVal intervalCode As String) As String
    If UsesHourFormatForManualEntry(intervalCode) Then
        LifeRemainingPromptHint = "Enter Life Remaining as hours (h:mm) or a decimal day-fraction number."
    ElseIf UCase$(Trim$(intervalCode)) = TYPE_DD Then
        LifeRemainingPromptHint = "Enter Life Remaining / days when due (number of days, or a due date as dd/mm/yyyy)."
    Else
        LifeRemainingPromptHint = "Enter Life Remaining as a number."
    End If
End Function

Private Function ActualIntervalPromptHint(ByVal intervalCode As String) As String
    If UsesHourFormatForManualEntry(intervalCode) Then
        ActualIntervalPromptHint = "Enter Actual Interval as hours (h:mm) or a number."
    Else
        ActualIntervalPromptHint = "Enter Actual Interval as a number."
    End If
End Function

Private Function ExtensionPromptHint(ByVal intervalCode As String) As String
    If UsesHourFormatForManualEntry(intervalCode) Then
        ExtensionPromptHint = "Enter Extension as hours (h:mm) or a number (use 0 if none)."
    Else
        ExtensionPromptHint = "Enter Extension as a number (use 0 if none)."
    End If
End Function

Private Sub WriteManualSITSTaskRow(ByVal tbl As ListObject, _
                                   ByVal newRow As ListRow, _
                                   ByRef draft As SITSManualTaskDraft, _
                                   ByVal rowId As String)
    Dim intervalCode As String

    intervalCode = draft.IntervalTypeCode

    With newRow.Range
        .Cells(1, GetTableColumnIndex(tbl, HDR_SELECTOR)).Value = False
        .Cells(1, GetTableColumnIndex(tbl, HDR_INTERVAL_TYPE_CODE)).Value = intervalCode
        .Cells(1, GetTableColumnIndex(tbl, HDR_TASK_CODE)).Value = draft.TaskCode
        .Cells(1, GetTableColumnIndex(tbl, HDR_TASK_SEQUENCE)).Value = draft.TaskSequence
        .Cells(1, GetTableColumnIndex(tbl, HDR_TASK_DESCRIPTION)).Value = draft.TaskDescription

        WriteSerialCell .Cells(1, GetTableColumnIndex(tbl, HDR_SERIAL)), draft.Serial, draft.Serial
        WriteSerialCell .Cells(1, GetTableColumnIndex(tbl, HDR_END_ITEM_SERIAL)), _
            draft.EndItemSerial, draft.EndItemSerial

        ApplyManualTimeOrNumberCell .Cells(1, GetTableColumnIndex(tbl, HDR_EXTENSION)), _
            draft.ExtensionText, intervalCode
        ApplyManualTimeOrNumberCell .Cells(1, GetTableColumnIndex(tbl, HDR_ACTUAL_INTERVAL)), _
            draft.ActualIntervalText, intervalCode

        ApplyManualLifeRemainingValues tbl, newRow, intervalCode, draft.LifeRemainingText
        CaptureTaskBaselineFromImport tbl, newRow, intervalCode

        .Cells(1, GetTableColumnIndex(tbl, HDR_SITS_ROW_ID)).Value = rowId
    End With
End Sub

Private Sub ApplyManualTimeOrNumberCell(ByVal target As Range, _
                                        ByVal textValue As String, _
                                        ByVal intervalCode As String)
    Dim parsedTime As Variant

    If InStr(textValue, ":") > 0 Or UsesHourFormatForManualEntry(intervalCode) Then
        parsedTime = ParseTimeUsageValue(textValue, textValue)
        If Not IsEmpty(parsedTime) Then
            WriteFormattedUsageValue target, parsedTime, FORMAT_TYPE_HOURS
            Exit Sub
        End If
    End If

    If IsNumeric(textValue) Then
        target.Value = CDbl(textValue)
    Else
        target.Value = textValue
    End If
    target.NumberFormat = FORMAT_DECIMAL
End Sub

Private Sub ApplyManualLifeRemainingValues(ByVal tbl As ListObject, _
                                           ByVal newRow As ListRow, _
                                           ByVal intervalCode As String, _
                                           ByVal textValue As String)
    Dim colHours As Long
    Dim colNumber As Long
    Dim colDays As Long
    Dim parsedTime As Variant
    Dim parsedDate As Variant

    colHours = GetTableColumnIndex(tbl, HDR_HOURS_LIFE_REMAINING)
    colNumber = GetTableColumnIndex(tbl, HDR_NUMBER_LIFE_REMAINING)
    colDays = GetTableColumnIndex(tbl, HDR_DAYS_WHEN_DUE)

    If UsesHourFormatForManualEntry(intervalCode) Then
        parsedTime = ParseTimeUsageValue(textValue, textValue)
        If Not IsEmpty(parsedTime) Then
            WriteFormattedUsageValue newRow.Range.Cells(1, colHours), parsedTime, FORMAT_TYPE_HOURS
            Exit Sub
        End If
        If IsNumeric(textValue) Then
            WriteFormattedUsageValue newRow.Range.Cells(1, colHours), CDbl(textValue), FORMAT_TYPE_HOURS
            Exit Sub
        End If
    End If

    If intervalCode = TYPE_DD Then
        parsedDate = ParseDMYDateValue(textValue, textValue)
        If IsDate(parsedDate) Then
            newRow.Range.Cells(1, colDays).Value = CDate(parsedDate)
            newRow.Range.Cells(1, colDays).NumberFormat = FORMAT_DATE
            Exit Sub
        End If
        If IsNumeric(textValue) Then
            newRow.Range.Cells(1, colNumber).Value = CDbl(textValue)
            newRow.Range.Cells(1, colNumber).NumberFormat = FORMAT_DECIMAL
            newRow.Range.Cells(1, colDays).Value = CDbl(textValue)
            newRow.Range.Cells(1, colDays).NumberFormat = FORMAT_DECIMAL
            Exit Sub
        End If
    End If

    If IsNumeric(textValue) Then
        newRow.Range.Cells(1, colNumber).Value = CDbl(textValue)
    Else
        newRow.Range.Cells(1, colNumber).Value = textValue
    End If
    newRow.Range.Cells(1, colNumber).NumberFormat = FORMAT_DECIMAL
End Sub

Private Function GenerateNextSITSRowId(ByVal tbl As ListObject) As String
    Dim colRowId As Long
    Dim row As ListRow
    Dim rawId As String
    Dim numPart As String
    Dim maxNum As Long
    Dim n As Long

    maxNum = 0
    colRowId = GetTableColumnIndex(tbl, HDR_SITS_ROW_ID)
    If colRowId > 0 And Not tbl.DataBodyRange Is Nothing Then
        For Each row In tbl.ListRows
            rawId = UCase$(Trim$(CStr(row.Range.Cells(1, colRowId).Value)))
            If Left$(rawId, 5) = "SITS-" Then
                numPart = Mid$(rawId, 6)
                If IsNumeric(numPart) Then
                    n = CLng(numPart)
                    If n > maxNum Then maxNum = n
                End If
            End If
        Next row
    End If

    GenerateNextSITSRowId = "SITS-" & Format$(maxNum + 1, "00000")
End Function

Private Sub RemoveWPTaskLinksForSITSRowId(ByVal sitsRowId As String)
    Dim ws As Worksheet
    Dim tbl As ListObject
    Dim colRowId As Long
    Dim row As ListRow
    Dim i As Long

    sitsRowId = Trim$(sitsRowId)
    If Len(sitsRowId) = 0 Then Exit Sub

    Set ws = GetWorksheet(SHEET_DATA_WP_TASKS, showError:=False)
    If ws Is Nothing Then Exit Sub
    Set tbl = GetTable(ws, TBL_WP_TASKS, showError:=False)
    If tbl Is Nothing Or tbl.DataBodyRange Is Nothing Then Exit Sub

    colRowId = GetTableColumnIndex(tbl, HDR_WP_TASK_ROW_ID)
    If colRowId = 0 Then Exit Sub

    ' Walk backwards so deletes do not skip rows.
    For i = tbl.ListRows.Count To 1 Step -1
        Set row = tbl.ListRows(i)
        If StrComp(Trim$(CStr(row.Range.Cells(1, colRowId).Value)), sitsRowId, vbTextCompare) = 0 Then
            row.Delete
        End If
    Next i
End Sub

' Keep collection ordered by descending ListRow.Index for safe multi-delete.
Private Sub InsertListRowDescending(ByVal rows As Collection, ByVal sitsRow As ListRow)
    Dim i As Long
    Dim existing As ListRow

    If sitsRow Is Nothing Then Exit Sub

    For i = 1 To rows.Count
        Set existing = rows(i)
        If sitsRow.Index > existing.Index Then
            rows.Add sitsRow, Before:=i
            Exit Sub
        End If
    Next i

    rows.Add sitsRow
End Sub

End Function
