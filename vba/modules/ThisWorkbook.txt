Option Explicit

Private Sub Workbook_Open()
    On Error Resume Next

    ' UserInterfaceOnly does not persist across closes — re-apply every open.
    ApplyWorkbookProtection

    If Not ValidateWorkbookStructure() Then
        MsgBox "Workbook structure validation found issues." & vbCrLf & vbCrLf & _
               "Run Setup_RunAll if this is a new master template, or use Rebuild Dashboard / " & _
               "Validate Structure on the Dashboard.", _
               vbExclamation, APP_NAME
    ElseIf IsMasterTemplate() Then
        MsgBox "Welcome to " & APP_NAME & " v" & APP_VERSION & " — master template." & vbCrLf & vbCrLf & _
               "Save a copy for each aircraft (e.g. ZZ383_SITS_v2.xlsm), then run " & _
               "Setup Wizard on the copy.", vbInformation, APP_NAME
    ElseIf Not IsSetupComplete() Then
        MsgBox "Welcome to " & APP_NAME & " v" & APP_VERSION & "." & vbCrLf & vbCrLf & _
               "This aircraft workbook has not been set up yet." & vbCrLf & _
               "Click Setup Wizard on the Dashboard to import SITS and usage.", _
               vbInformation, APP_NAME
    End If

    On Error GoTo 0
End Sub

' Form-control LinkedCell writes fire SheetChange — keep selection highlight in sync
' even when a checkbox OnAction was not assigned (e.g. view not refreshed yet).
Private Sub Workbook_SheetChange(ByVal Sh As Object, ByVal Target As Range)
    On Error Resume Next
    HandleSITSSelectorSheetChange Sh, Target
    On Error GoTo 0
End Sub

Private Sub Workbook_BeforeClose(Cancel As Boolean)
    SetExcelState "enable"
End Sub
